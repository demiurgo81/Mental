PROCEDURE df_gestion_rango_sp (
    p_fini IN DATE DEFAULT NULL,
    p_ffin IN DATE DEFAULT NULL,
    p_meses IN NUMBER DEFAULT NULL
)
IS
    v_fini DATE;
    v_ffin DATE;
    v_rows_updated NUMBER;

BEGIN
    -- Validación: Parámetro meses debe estar entre 1 y 12
    IF p_meses IS NOT NULL AND (p_meses < 1 OR p_meses > 12) THEN
        RAISE_APPLICATION_ERROR(-20003, 'Error: El parámetro meses debe estar entre 1 y 12');
    END IF;

    -- Validación: Si se proporciona FFIN, debe proporcionarse también FINI (solo cuando no hay meses)
    IF p_meses IS NULL AND p_ffin IS NOT NULL AND p_fini IS NULL THEN
        RAISE_APPLICATION_ERROR(-20001, 'Error: Si se proporciona FFIN, también debe proporcionarse FINI');
    END IF;

    -- Validación: FINI debe ser menor o igual a FFIN cuando ambos están presentes
    IF p_fini IS NOT NULL AND p_ffin IS NOT NULL AND p_fini > p_ffin THEN
        RAISE_APPLICATION_ERROR(-20002, 'Error: La fecha FINI debe ser menor o igual a FFIN');
    END IF;

    -- Lógica según los parámetros recibidos (orden de prioridad)
    IF p_meses IS NOT NULL THEN
        -- Escenario 1: Parámetro meses tiene prioridad
        v_fini := TRUNC(ADD_MONTHS(SYSDATE, -p_meses), 'MM');
        v_ffin := LAST_DAY(ADD_MONTHS(SYSDATE, -p_meses));

    ELSIF p_fini IS NOT NULL AND p_ffin IS NOT NULL THEN
        -- Escenario 2: Ambos parámetros FINI y FFIN proporcionados
        v_fini := p_fini;
        v_ffin := p_ffin;

    ELSIF p_fini IS NOT NULL THEN
        -- Escenario 3: Solo FINI proporcionado - usar p_fini en lugar de SYSDATE
        v_fini := TRUNC(p_fini, 'MM');
        v_ffin := LAST_DAY(p_fini);

    ELSE
        -- Escenario 4: Sin parámetros - usar SYSDATE
        -- Escenario 4: Sin parámetros - usar SYSDATE
        v_fini := TRUNC(SYSDATE, 'MM');
        v_ffin := LAST_DAY(SYSDATE);

    END IF;

    -- Ejecutar el UPDATE
    UPDATE df_rango 
    SET coord = 1, 
        ger = 346, 
        fini = v_fini, 
        ffin = v_ffin;

    -- Obtener número de filas actualizadas
    v_rows_updated := SQL%ROWCOUNT;

    -- Confirmar la transacción
    COMMIT;

    -- Mensaje informativo
    DBMS_OUTPUT.PUT_LINE('Procedimiento ejecutado exitosamente.');
    DBMS_OUTPUT.PUT_LINE('Filas actualizadas: ' || v_rows_updated);
    DBMS_OUTPUT.PUT_LINE('Fecha inicio (FINI): ' || TO_CHAR(v_fini, 'DD/MM/YYYY'));
    DBMS_OUTPUT.PUT_LINE('Fecha fin (FFIN): ' || TO_CHAR(v_ffin, 'DD/MM/YYYY'));

EXCEPTION
    WHEN OTHERS THEN
        -- Rollback en caso de error
        ROLLBACK;

        -- Capturar y mostrar el error
        DBMS_OUTPUT.PUT_LINE('Error en df_gestion_rango_sp: ' || SQLERRM);

        -- Re-lanzar la excepción
        RAISE;

END df_gestion_rango_sp;
