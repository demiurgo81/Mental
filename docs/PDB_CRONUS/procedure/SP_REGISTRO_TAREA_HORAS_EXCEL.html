PROCEDURE SP_REGISTRO_TAREA_HORAS_EXCEL (
    P_CODEMPLEADO IN NUMBER,
    P_PROYECTO IN NUMBER,
    P_TAREA_PADRE IN VARCHAR2,
    P_NOMBRE_TAREA IN VARCHAR2,
    P_FECHA_INI IN DATE, 
    P_FECHA_FIN IN DATE,
    P_FECHA_REGISTRO IN DATE,
    P_HORAS IN NUMBER,
    P_TIPO_TAREA IN NUMBER,
    P_DESCRIP_TAREA IN VARCHAR2,
    V_S_RESPUESTA OUT VARCHAR2,
    V_S_CODIGO OUT NUMBER
)
AS
    V_RESPUESTA VARCHAR2(4000);
    V_CODIGO NUMBER;
    V_MENSAJE VARCHAR2(4000);
    V_CODEMPLEADOLIDER NUMBER;
    V_CANT_TAREA_PADRE NUMBER;
    V_COD_TAREAESTADO NUMBER := 3;
    V_NIVEL_TAREA_PADRE NUMBER := 1;
    V_ID_TAREA_PADRE NUMBER;
    V_ESLOGRO NUMBER := 1;
    V_ESTADO VARCHAR(1) := 'A';
    V_ID_TAREA NUMBER := 0;
    V_ID_EMPLEADOCONTROL NUMBER;
    V_RESPUESTA_EMPLEADO VARCHAR2(4000);
    V_HORAS_INIT NUMBER:= 0.01;
    
BEGIN
    BEGIN
    
        BEGIN
        -- TRAE EL CODEMPLEADO DEL LIDER
            SELECT CODEMPLEADOLIDER INTO V_CODEMPLEADOLIDER FROM DF_EMPLEADO_VU WHERE CODEMPLEADO = P_CODEMPLEADO;
            DBMS_OUTPUT.PUT_LINE('ENCUENTRA EL CODEMPLEADOLIDER ' || V_CODEMPLEADOLIDER);
         EXCEPTION
            WHEN NO_DATA_FOUND THEN
                V_CODEMPLEADOLIDER := P_CODEMPLEADO;
                DBMS_OUTPUT.PUT_LINE('NO ENCUENTRA EL V_ID_TAREA_PADRE ' || V_ID_TAREA_PADRE);
        END;
        
        IF P_HORAS > 1 THEN 
            V_HORAS_INIT := P_HORAS;
        END IF;
        
        -- CONSULTA SI TAREA PADRE EXISTE, SINO LA CREA
        BEGIN
            SELECT ID INTO V_ID_TAREA_PADRE FROM DF_TAREA WHERE NOMBRE = P_TAREA_PADRE 
                AND CODPROYECTO = P_PROYECTO AND CODEMPLEADOASIGNADO = P_CODEMPLEADO 
                AND ROWNUM = 1;
            DBMS_OUTPUT.PUT_LINE('ENCUENTRA EL V_ID_TAREA_PADRE ' || V_ID_TAREA_PADRE);
        EXCEPTION
            WHEN NO_DATA_FOUND THEN
                V_ID_TAREA_PADRE := 0;
                V_CANT_TAREA_PADRE := 0;
                DBMS_OUTPUT.PUT_LINE('NO ENCUENTRA EL V_ID_TAREA_PADRE ' || V_ID_TAREA_PADRE);
        END;

        IF V_ID_TAREA_PADRE = 0  THEN
            --- CREA TAREA PADRE
            DBMS_OUTPUT.PUT_LINE('LLAMADO A CREAR TAREA_PADRE ' || V_ID_TAREA_PADRE);
            SP_SAVE_DF_TAREA(V_ID_TAREA_PADRE, V_ID_TAREA_PADRE, 
                P_TIPO_TAREA, V_COD_TAREAESTADO, V_CODEMPLEADOLIDER, 
                P_CODEMPLEADO, P_PROYECTO, '', 
                P_TAREA_PADRE, V_NIVEL_TAREA_PADRE, P_DESCRIP_TAREA, 
                P_FECHA_INI, P_FECHA_FIN, 
                P_FECHA_INI, P_FECHA_FIN, V_HORAS_INIT, V_HORAS_INIT, V_ESLOGRO, V_ESTADO,
                CURRENT_TIMESTAMP, CURRENT_TIMESTAMP, '', 0, V_RESPUESTA, V_CODIGO, V_MENSAJE );
            --RETORNAR EL ID DE LA TAREA PADRE CREADA
            V_ID_TAREA_PADRE := TO_NUMBER(V_RESPUESTA);
            DBMS_OUTPUT.PUT_LINE('CREA tareapadre ' || V_ID_TAREA_PADRE);
        END IF;
        
        
        -- CONSULTA SI TAREA HIJA EXISTE, SINO LA CREA
        BEGIN
            SELECT ID INTO V_ID_TAREA FROM DF_TAREA WHERE NOMBRE = P_NOMBRE_TAREA 
                AND CODPROYECTO = P_PROYECTO AND CODEMPLEADOASIGNADO = P_CODEMPLEADO
                AND FECHAINIESTIMADA BETWEEN P_FECHA_INI AND P_FECHA_FIN AND ROWNUM = 1;
            DBMS_OUTPUT.PUT_LINE('ENCUENTRA EL V_ID_TAREA ' || V_ID_TAREA);
        EXCEPTION
            WHEN NO_DATA_FOUND THEN
                V_ID_TAREA := 0;
                V_CANT_TAREA_PADRE := 0;
                DBMS_OUTPUT.PUT_LINE('NO ENCUENTRA EL V_ID_TAREA ' || V_ID_TAREA);
        END;
        
        --IF V_ID_TAREA = 0  THEN
        
        SP_SAVE_DF_TAREA(V_ID_TAREA, V_ID_TAREA_PADRE, P_TIPO_TAREA, V_COD_TAREAESTADO, 
            V_CODEMPLEADOLIDER, P_CODEMPLEADO, P_PROYECTO, '', 
                P_NOMBRE_TAREA, V_NIVEL_TAREA_PADRE, P_DESCRIP_TAREA, P_FECHA_INI, 
                P_FECHA_FIN, P_FECHA_INI, P_FECHA_FIN, V_HORAS_INIT, V_HORAS_INIT, V_ESLOGRO, V_ESTADO,
                CURRENT_TIMESTAMP, CURRENT_TIMESTAMP, '', 0, V_RESPUESTA, V_CODIGO, V_MENSAJE );
        --RETORNAR EL ID DE LA TAREA CREADA
        V_ID_TAREA := TO_NUMBER(V_RESPUESTA);
        DBMS_OUTPUT.PUT_LINE('CREA tarea HIJA ' || V_ID_TAREA);
    
        SP_SAVE_DF_EMPLEADOCONTROL(P_DESCRIP_TAREA, P_HORAS, P_CODEMPLEADO, V_ID_TAREA, 
            P_FECHA_REGISTRO, P_FECHA_REGISTRO, V_RESPUESTA_EMPLEADO, V_CODIGO);
        
        --V_ID_EMPLEADOCONTROL := FN_INSERT_DF_EMPLEADOCONTROL(P_CODEMPLEADO, V_ID_TAREA, P_DESCRIP_TAREA, P_FECHA_REGISTRO, P_FECHA_REGISTRO, P_HORAS);
        --DBMS_OUTPUT.PUT_LINE('CREA REGISTRO EMPLEADOCONTROL ' || V_ID_EMPLEADOCONTROL);
        
        --END IF;
        
        COMMIT;
        
        V_S_CODIGO := 200;
        V_S_RESPUESTA := 'REGISTRO Correcto Tarea: ' || V_ID_TAREA || ' ';
    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            ROLLBACK;
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'No Fue posible traer la data';
            RAISE;
    END;
END SP_REGISTRO_TAREA_HORAS_EXCEL;
