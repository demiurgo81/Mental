PROCEDURE SP_GET_TAREAS_HIJAS_FORMS (
    P_ID_CODTAREA_PADRE IN NUMBER,
    P_CODEMPLEADOASIGNADO IN NUMBER,
    P_CURSOR_GET_TAREAS_HIJAS_FORMS OUT SYS_REFCURSOR,
    V_S_RESPUESTA OUT VARCHAR2,
    V_S_CODIGO OUT NUMBER
)
AS
BEGIN
    BEGIN
        -- Primer cursor para obtener las iniciativas según el ID
        OPEN P_CURSOR_GET_TAREAS_HIJAS_FORMS FOR
            SELECT T.ID, T.NOMBRE AS NOMBRE_TAREA, P.NOMBRE AS NOMBRE_PROYECTO, T.FECHAINIREAL,
                T.DESCRIPCION, PER.NOMCOMPLETO 
            FROM DF_TAREA T
                LEFT JOIN DF_PROYECTO P ON T.CODPROYECTO=P.ID
                LEFT JOIN DF_EMPLEADO E ON E.ID=T.CODEMPLEADOASIGNADO
                LEFT JOIN DF_PERSONA  PER ON PER.ID=E.CODPERSONA
                WHERE (P_ID_CODTAREA_PADRE = 0 OR CODTAREAPADRE = P_ID_CODTAREA_PADRE)
                    AND (P_CODEMPLEADOASIGNADO = 0 OR CODEMPLEADOASIGNADO = P_CODEMPLEADOASIGNADO)
                    AND T.FECHAINIESTIMADA BETWEEN  TRUNC(SYSDATE, 'MM') -- Inicio del mes actual
                    AND TRUNC(ADD_MONTHS(SYSDATE, 1), 'MM') - 1/24/60/60;

        V_S_RESPUESTA := 'Retorno de datos completo';
        V_S_CODIGO := 200;
    EXCEPTION
        WHEN OTHERS THEN
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'Error: ' || SQLERRM;
    END;
END SP_GET_TAREAS_HIJAS_FORMS;
