PROCEDURE SP_GET_TAREA_ACTIVIDAD_TIPO (
    P_IS_PADRE IN NUMBER,
    P_CURSOR_GET_TAREA_ACTIVIDAD OUT SYS_REFCURSOR,
    V_S_RESPUESTA OUT VARCHAR2,
    V_S_CODIGO OUT NUMBER
)
AS
BEGIN
    BEGIN
        IF P_IS_PADRE = 1 THEN
            OPEN P_CURSOR_GET_TAREA_ACTIVIDAD FOR
            SELECT TP.ID AS ID, TP.NOMBRE AS NOMBRE, 
                   CASE 
                       WHEN TP.ID = 24 THEN 'Desarrollo Modificacion Microservicio'
                       WHEN TP.ID = 25 THEN 'Desarrollo Modificacion API'
                       WHEN TP.ID = 65 THEN 'Desarrollo API'
                       WHEN TP.ID = 66 THEN 'Desarrollo Microservicio'
                       ELSE 'Otro'
                   END AS DESCRIPCION
            FROM DF_TAREATIPO TP
            WHERE TP.ID IN (24, 25, 65, 66);
        
        ELSIF P_IS_PADRE = 2 THEN
            OPEN P_CURSOR_GET_TAREA_ACTIVIDAD FOR
            SELECT TP.ID, TP.NOMBRE, 'Accesos/Infraestructura' AS DESCRIPCION
            FROM DF_TAREATIPO TP WHERE TP.ID = 104
            UNION ALL
            SELECT TP.ID, TP.NOMBRE, 'Configuracion' FROM DF_TAREATIPO TP WHERE TP.ID = 6
            UNION ALL
            SELECT TP.ID, TP.NOMBRE, 'Contextualizacion' FROM DF_TAREATIPO TP WHERE TP.ID = 44
            UNION ALL
            SELECT TP.ID, TP.NOMBRE, 'Cronograma' FROM DF_TAREATIPO TP WHERE TP.ID = 44
            UNION ALL
            SELECT TP.ID, TP.NOMBRE, 'Daily' FROM DF_TAREATIPO TP WHERE TP.ID = 84
            UNION ALL
            SELECT TP.ID, TP.NOMBRE, 'Diseño' FROM DF_TAREATIPO TP WHERE TP.ID = 44
            UNION ALL
            SELECT TP.ID, TP.NOMBRE, 'Documentacion Otros' FROM DF_TAREATIPO TP WHERE TP.ID = 44
            UNION ALL
            SELECT TP.ID, TP.NOMBRE, 'Documentacion SSD' FROM DF_TAREATIPO TP WHERE TP.ID = 6
            UNION ALL
            SELECT TP.ID, TP.NOMBRE, 'Entrega Operacion' FROM DF_TAREATIPO TP WHERE TP.ID = 84
            UNION ALL
            SELECT TP.ID, TP.NOMBRE, 'Guia Integracion' FROM DF_TAREATIPO TP WHERE TP.ID = 44
            UNION ALL
            SELECT TP.ID, TP.NOMBRE, 'PAP' FROM DF_TAREATIPO TP WHERE TP.ID = 84
            UNION ALL
            SELECT TP.ID, TP.NOMBRE, 'Pruebas' FROM DF_TAREATIPO TP WHERE TP.ID = 6;
        
        ELSE
            OPEN P_CURSOR_GET_TAREA_ACTIVIDAD FOR
            SELECT TP.ID AS ID, TP.NOMBRE AS NOMBRE, 
                   CASE 
                       WHEN TP.ID = 3 THEN 'Soporte servicio'
                       WHEN TP.ID = 4 THEN 'Documentación'
                       WHEN TP.ID = 5 THEN 'Pruebas unitarias'
                       WHEN TP.ID = 6 THEN 'Pruebas end2end'
                       WHEN TP.ID = 24 THEN 'Nuevo microservicio'
                       WHEN TP.ID = 25 THEN 'Modificar middleware'
                       WHEN TP.ID = 44 THEN 'Propuesta de diseño'
                       WHEN TP.ID = 64 THEN 'Modificar microservicio'
                       WHEN TP.ID = 65 THEN 'Nuevo middleware'
                       WHEN TP.ID = 66 THEN 'Nuevo microservicio'
                       WHEN TP.ID = 84 THEN 'Contextualización diseño'
                       WHEN TP.ID = 104 THEN 'Configuración infraestructura'
                       ELSE 'Otro'
                   END AS DESCRIPCION
            FROM DF_TAREATIPO TP
            WHERE TP.ID > 2;
        END IF;

        V_S_RESPUESTA := 'Retorno de datos completo';
        V_S_CODIGO := 200;

    EXCEPTION
        WHEN OTHERS THEN
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'Error en el procedimiento: ' || SQLERRM;
    END;
END;
