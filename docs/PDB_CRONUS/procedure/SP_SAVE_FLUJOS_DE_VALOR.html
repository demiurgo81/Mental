PROCEDURE SP_SAVE_FLUJOS_DE_VALOR (
    P_CODEMPLEADO IN NUMBER,
    P_PROYECTO IN NUMBER,
    P_ENTREGABLE IN VARCHAR2,
    P_FECHA_INI IN DATE, 
    P_FECHA_FIN IN DATE,
    P_PEP_INICIATIVA IN VARCHAR2,
    P_LIDER_INICIATIVA IN VARCHAR2,
    P_ID_INICIATIVA IN VARCHAR2,
    P_NOMBRE_INICIATIVA IN VARCHAR2,
    P_HORAS IN NUMBER,
    P_TIPO_TAREA IN NUMBER,
    P_DESCRIP_TAREA IN VARCHAR2,
    V_ID_CREADO OUT NUMBER,
    V_S_RESPUESTA OUT VARCHAR2,
    V_S_CODIGO OUT NUMBER
)
AS
    V_COUNT NUMBER;  -- Variable para verificar si el registro existe
    V_CELULA NUMBER;
    V_RESPUESTA VARCHAR2(4000);
    V_CODIGO NUMBER;
    V_MENSAJE VARCHAR2(4000);
    V_CODEMPLEADOLIDER NUMBER;
    V_CANT_TAREA_PADRE NUMBER;
    V_COD_TAREAESTADO NUMBER := 3;
    V_NIVEL_TAREA_PADRE NUMBER := 1;
    V_ID_TAREA_PADRE NUMBER;
    V_ESLOGRO NUMBER := 1;
    V_ESTADO VARCHAR(1) := 'A';
    V_ID_TAREA NUMBER := 0;
    V_ID_EMPLEADOCONTROL NUMBER;
    V_RESPUESTA_EMPLEADO VARCHAR2(4000);
    V_HORAS_INIT NUMBER:= 0.01;
BEGIN
    BEGIN

        IF P_HORAS > 1 THEN 
            V_HORAS_INIT := P_HORAS;
        END IF;
        
        -- CONSULTA SI TAREA PADRE EXISTE, SINO LA CREA
        BEGIN
            SELECT ID INTO V_ID_TAREA_PADRE FROM DF_TAREA WHERE NOMBRE = P_ENTREGABLE 
                AND CODPROYECTO = P_PROYECTO AND CODEMPLEADOASIGNADO = P_CODEMPLEADO ;
            DBMS_OUTPUT.PUT_LINE('ENCUENTRA EL V_ID_TAREA_PADRE ' || V_ID_TAREA_PADRE);
        EXCEPTION
            WHEN NO_DATA_FOUND THEN
                V_ID_TAREA_PADRE := 0;
                V_CANT_TAREA_PADRE := 0;
                DBMS_OUTPUT.PUT_LINE('NO ENCUENTRA EL V_ID_TAREA_PADRE ' || V_ID_TAREA_PADRE);
        END;
        
        IF V_ID_TAREA_PADRE = 0  THEN
            --- CREA TAREA PADRE
            DBMS_OUTPUT.PUT_LINE('LLAMADO A CREAR TAREA_PADRE ' || V_ID_TAREA_PADRE);
            SP_SAVE_DF_TAREA(V_ID_TAREA_PADRE, V_ID_TAREA_PADRE, 
                P_TIPO_TAREA, V_COD_TAREAESTADO, P_CODEMPLEADO, 
                0, P_PROYECTO, '', 
                P_ENTREGABLE, V_NIVEL_TAREA_PADRE, P_DESCRIP_TAREA, 
                P_FECHA_INI, P_FECHA_FIN, 
                P_FECHA_INI, P_FECHA_FIN, V_HORAS_INIT, V_HORAS_INIT, V_ESLOGRO, V_ESTADO,
                CURRENT_TIMESTAMP, CURRENT_TIMESTAMP, '', 0, V_RESPUESTA, V_CODIGO, V_MENSAJE );
            --RETORNAR EL ID DE LA TAREA PADRE CREADA
            V_ID_TAREA_PADRE := TO_NUMBER(V_RESPUESTA);
            DBMS_OUTPUT.PUT_LINE('CREA tareapadre ' || V_ID_TAREA_PADRE);
        END IF;

        V_S_RESPUESTA := 'Retorno de datos completo ';
        V_S_CODIGO := 200;


    EXCEPTION
        WHEN OTHERS THEN
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'Error en el procedimiento: ' || SQLERRM;
            ROLLBACK;
    END;
END;
