PROCEDURE SP_GET_HORAS_CELULA (
    P_CODEMPLEADO IN NUMBER,                -- Codempleado para realizar consulta por celula
    P_FECHA_INI IN DATE,                -- Fecha de inicio de la tarea
    P_FECHA_FIN IN DATE,                -- Fecha de fin de la tarea
    P_CURSOR_GET_HORAS_CELULA OUT SYS_REFCURSOR, -- Cursor de salida
    V_S_RESPUESTA OUT VARCHAR2,         -- Respuesta de salida
    V_S_CODIGO OUT NUMBER               -- Código de respuesta
)
AS
    V_COUNT NUMBER;  -- Variable para verificar si el registro existe
    V_CELULA NUMBER;
BEGIN
    BEGIN

        SELECT CODESTRUCTURAORGANIZACIONAL INTO V_CELULA FROM DF_EMPLEADO WHERE ID = P_CODEMPLEADO;

        OPEN P_CURSOR_GET_HORAS_CELULA FOR
            SELECT  EC.CODEMPLEADO AS CODEMPLEADO, P.NOMCOMPLETO AS NOMBRE, SUM(EC.HORAS) AS HORAS FROM DF_EMPLEADOCONTROL EC
                LEFT JOIN DF_TAREA T ON EC.CODTAREA = T.ID
                LEFT JOIN DF_EMPLEADO E ON EC.CODEMPLEADO = E.ID
                LEFT JOIN DF_PERSONA P ON P.ID = E.CODPERSONA
                    WHERE E.CODESTRUCTURAORGANIZACIONAL = V_CELULA
                        AND TO_DATE(EC.FECHAHORAINICIO) BETWEEN P_FECHA_INI AND P_FECHA_FIN  -- Filtra por rango de fechas de inicio
                        AND TO_DATE(EC.FECHAHORAFIN) BETWEEN P_FECHA_INI AND P_FECHA_FIN
                    GROUP BY EC.CODEMPLEADO, P.NOMCOMPLETO;
        V_S_RESPUESTA := 'Retorno de datos completo ';
        V_S_CODIGO := 200;


    EXCEPTION
        WHEN OTHERS THEN
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'Error en el procedimiento: ' || SQLERRM;
            ROLLBACK;
    END;
END;
