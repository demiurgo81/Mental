PROCEDURE SP_SAVE_PROYECTO_ESTIMACION (
    P_ID                      IN DF_PROYECTO_ESTIMACION.ID%TYPE,
    P_ID_PROYECTO             IN DF_PROYECTO_ESTIMACION.ID_PROYECTO%TYPE,
    P_NOMBRE_ESTIMACION       IN DF_PROYECTO_ESTIMACION.NOMBRE_ESTIMACION%TYPE,
    P_URL_SHAREPOINT          IN DF_PROYECTO_ESTIMACION.URL_SHAREPOINT%TYPE,
    P_ID_ESTADO               IN DF_PROYECTO_ESTIMACION.ID_ESTADO%TYPE,
    P_HUT                     IN DF_PROYECTO_ESTIMACION.HUT%TYPE,
    P_OBSERVACIONES           IN DF_PROYECTO_ESTIMACION.OBSERVACIONES%TYPE,
    P_CODTAREA                IN DF_PROYECTO_ESTIMACION.CODTAREA%TYPE,
    P_ID_SOLPED               IN DF_PROYECTO_ESTIMACION.ID_SOLPED%TYPE,
    P_CODEMPLEADO             IN DF_PROYECTO_ESTIMACION.CODEMPLEADO_CREA%TYPE,
    V_ID_CREADO               OUT DF_PROYECTO_ESTIMACION.ID%TYPE,
    V_S_RESPUESTA             OUT VARCHAR2,
    V_S_CODIGO                OUT NUMBER
) AS
    V_COUNT NUMBER;
    V_ESTADO_PROY_ESTIMACION NUMBER;
    V_ID_PROYECTO_ESTIMACION NUMBER;
BEGIN
    -- Verificar existencia por ID o combinación proyecto/nombre
    SELECT COUNT(*)
      INTO V_COUNT
      FROM DF_PROYECTO_ESTIMACION
     WHERE ID = P_ID
        OR (NOMBRE_ESTIMACION = P_NOMBRE_ESTIMACION);

    IF V_COUNT > 0 THEN
        -- Actualizar registro existente
        
        SELECT ID INTO V_ID_PROYECTO_ESTIMACION FROM DF_PROYECTO_ESTIMACION
            WHERE ID = P_ID OR (NOMBRE_ESTIMACION = P_NOMBRE_ESTIMACION);
        
        --IF P_ID_ESTADO <> 2 THEN 
            V_ESTADO_PROY_ESTIMACION := P_ID_ESTADO;
        --ELSE 
            --V_ESTADO_PROY_ESTIMACION := 1;
        --END IF;
        
        UPDATE DF_PROYECTO_ESTIMACION
           SET ID_PROYECTO         = P_ID_PROYECTO,
               NOMBRE_ESTIMACION   = P_NOMBRE_ESTIMACION,
               URL_SHAREPOINT      = P_URL_SHAREPOINT,
               ID_ESTADO           = V_ESTADO_PROY_ESTIMACION,
               HUT                 = P_HUT,
               OBSERVACIONES       = P_OBSERVACIONES,
               CODTAREA            = P_CODTAREA,
               ID_SOLPED           = P_ID_SOLPED,
               FECHAMODIFICACION   = CURRENT_TIMESTAMP,
               CODEMPLEADO_MODIFCA = P_CODEMPLEADO
         WHERE ID = V_ID_PROYECTO_ESTIMACION;

        V_ID_CREADO   := V_ID_PROYECTO_ESTIMACION;
        V_S_RESPUESTA := 'Registro actualizado exitosamente';
    ELSE
        -- Insertar nuevo registro
        INSERT INTO DF_PROYECTO_ESTIMACION (
            ID_PROYECTO,
            NOMBRE_ESTIMACION,
            URL_SHAREPOINT,
            ID_ESTADO,
            HUT,
            OBSERVACIONES,
            CODTAREA,
            ID_SOLPED,
            FECHACREACION,
            CODEMPLEADO_CREA
        ) VALUES (
            P_ID_PROYECTO,
            P_NOMBRE_ESTIMACION,
            P_URL_SHAREPOINT,
            P_ID_ESTADO,
            P_HUT,
            P_OBSERVACIONES,
            P_CODTAREA,
            P_ID_SOLPED,
            CURRENT_TIMESTAMP,
            P_CODEMPLEADO
        ) RETURNING ID INTO V_ID_CREADO;

        V_S_RESPUESTA := 'Registro insertado exitosamente';
    END IF;

    COMMIT;
    V_S_CODIGO := 200;

EXCEPTION
    WHEN OTHERS THEN
        V_S_CODIGO    := 500;
        V_S_RESPUESTA := 'Error interno: ' || SQLERRM;
        ROLLBACK;
END SP_SAVE_PROYECTO_ESTIMACION;
