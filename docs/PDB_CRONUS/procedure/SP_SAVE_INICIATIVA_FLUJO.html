PROCEDURE SP_SAVE_INICIATIVA_FLUJO (
    P_ID IN NUMBER,
    P_INICIATIVA IN VARCHAR2,
    P_CODEMPLEADO_CREO IN NUMBER,
    P_CODEMPLEADO_MODIFICA IN NUMBER,
    P_ID_LIDER_INICIATIVA IN NUMBER,
    V_ID_CREADO OUT NUMBER,
    V_S_RESPUESTA OUT VARCHAR2,
    V_S_CODIGO OUT NUMBER
)
AS
    V_COUNT NUMBER;
BEGIN
    BEGIN
        -- Verificar si el registro ya existe
        SELECT COUNT(*)
        INTO V_COUNT
        FROM DF_INICIATIVA_FLUJO
        WHERE ID = P_ID OR UPPER(TRIM(INICIATIVA)) = UPPER(TRIM(P_INICIATIVA));

        -- Si el registro existe, actualizarlo; de lo contrario, insertar uno nuevo
        IF V_COUNT > 0 THEN
            UPDATE DF_INICIATIVA_FLUJO
            SET INICIATIVA = P_INICIATIVA,
                --CODEMPLEADO_MODIFICA = P_CODEMPLEADO_MODIFICA,
                FECHAMODIFICACION = CURRENT_TIMESTAMP,
                ID_LIDER_INICIATIVA = P_ID_LIDER_INICIATIVA
            WHERE ID = P_ID;

            V_ID_CREADO := P_ID;
            V_S_RESPUESTA := 'Registro actualizado exitosamente';
        ELSE
            INSERT INTO DF_INICIATIVA_FLUJO (ID, INICIATIVA, CODEMPLEADO_CREO, FECHACREACION, ID_LIDER_INICIATIVA)
            VALUES (P_ID, P_INICIATIVA, P_CODEMPLEADO_CREO, CURRENT_TIMESTAMP, P_ID_LIDER_INICIATIVA)
            RETURNING ID INTO V_ID_CREADO;

            V_S_RESPUESTA := 'Registro insertado exitosamente';
        END IF;

        -- Confirmar transacción
        COMMIT;
        V_S_CODIGO := 200;

    EXCEPTION
        WHEN OTHERS THEN
            -- Manejo de errores
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'Error: ' || SQLERRM;
            ROLLBACK;
    END;
END SP_SAVE_INICIATIVA_FLUJO;
