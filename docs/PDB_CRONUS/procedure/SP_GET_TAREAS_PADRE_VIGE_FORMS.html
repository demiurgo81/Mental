PROCEDURE "SP_GET_TAREAS_PADRE_VIGE_FORMS" (
    P_ID_TAREA IN NUMBER,
    P_COD_EMPLEADO IN NUMBER, 
    P_CURSOR_TAREAS_PADRE_VIG_FORMS OUT SYS_REFCURSOR,
    V_S_RESPUESTA OUT VARCHAR2,
    V_S_CODIGO OUT NUMBER
)
AS
BEGIN
    BEGIN
        OPEN P_CURSOR_TAREAS_PADRE_VIG_FORMS FOR
            SELECT DISTINCT
    tp.id, 
    tp.nombre, 
    tp.CODEMPLEADOCREO as CODEMPLEADOASIGNADO,
    tp.CODPROYECTO,
    i.EVIDENCIA AS URL_ENTREGABLE,
    tp.FECHAINIESTIMADA, 
    tp.FECHAFINESTIMADA,
    TP.DESCRIPCION,
    PE.ID ID_PEP,
    PE.PEP,
    I.ID AS ID_ENTREGABLE,
    i.NOMBRE AS NOMBRE_ENTREGABLE,
    TP.CODTAREATIPO AS TIPO_TAREA,
    I.ID_LIDER_INICIATIVA,
    tp.TIEMPOESTIMADO,
    I.ID_INI_FLUJO
FROM 
    DF_TAREA tp 
INNER JOIN DF_INICIATIVA i ON i.NOMBRE = tp.NOMBRE OR I.CODTAREA = TP.ID
INNER JOIN DF_PEP PE ON PE.ID = I.ID_PEP OR PE.PEP = I.PEP
WHERE  
    tp.nivel = 1 
    AND (
        (EXTRACT(DAY FROM SYSDATE) <= 5 AND  -- Si estamos dentro de los primeros 5 días del mes
         tp.FECHAINIESTIMADA >= TRUNC(ADD_MONTHS(SYSDATE, -1), 'MM') AND  -- Incluir desde el mes pasado
         tp.FECHAINIESTIMADA <= TRUNC(ADD_MONTHS(SYSDATE, 2), 'MM'))  -- Hasta el siguiente mes (excluyendo el primer día)

        OR  -- Si ya pasaron los primeros 5 días del mes

        (EXTRACT(DAY FROM SYSDATE) > 5 AND
         tp.FECHAINIESTIMADA >= TRUNC(SYSDATE, 'MM') AND  -- Solo desde el mes actual
         tp.FECHAINIESTIMADA <= TRUNC(ADD_MONTHS(SYSDATE, 2), 'MM'))
    ) 
    AND (
        (EXTRACT(DAY FROM SYSDATE) <= 5 AND  
         tp.FECHAFINESTIMADA >= TRUNC(ADD_MONTHS(SYSDATE, -1), 'MM') AND  
         tp.FECHAFINESTIMADA <= TRUNC(ADD_MONTHS(SYSDATE, 2), 'MM'))  
        
        OR 
        
        (EXTRACT(DAY FROM SYSDATE) > 5 AND
         tp.FECHAFINESTIMADA >= TRUNC(SYSDATE, 'MM') AND  
         tp.FECHAFINESTIMADA <= TRUNC(ADD_MONTHS(SYSDATE, 2), 'MM'))
    )
    AND (P_ID_TAREA = 0 OR tp.ID = P_ID_TAREA)
    AND (P_COD_EMPLEADO = 0 OR tp.CODEMPLEADOCREO = P_COD_EMPLEADO);


        V_S_CODIGO := 200;
        V_S_RESPUESTA := 'Retorno Correcto';
    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            ROLLBACK;
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'No Fue posible traer la data';
            RAISE;
    END;
END SP_GET_TAREAS_PADRE_VIGE_FORMS;
