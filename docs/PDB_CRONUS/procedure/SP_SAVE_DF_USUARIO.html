PROCEDURE SP_SAVE_DF_USUARIO (
    P_ID              IN NUMBER,
    P_CODUSUARIOROL   IN NUMBER,
    P_CODPERSONA      IN NUMBER,
    P_USUARIO         IN VARCHAR2,
    P_NOMBRE          IN VARCHAR2,
    P_CLAVE           IN VARCHAR2,
    P_ESTADO          IN VARCHAR2,
    P_TOKENAZURE      IN VARCHAR2,
    P_AZUREEMAIL      IN VARCHAR2,
    V_S_RESPUESTA     OUT VARCHAR2,
    V_ID_CREADO       OUT NUMBER,
    V_S_CODIGO        OUT NUMBER
)
AS
    V_COUNT NUMBER;
    V_ID_USER NUMBER;
    V_ID_ROL_USER NUMBER;
BEGIN
    BEGIN
        -- Verificar si el registro ya existe
        SELECT COUNT(*) INTO V_COUNT FROM DF_USUARIO WHERE (ID = P_ID OR USUARIO = P_USUARIO OR CODPERSONA = P_CODPERSONA) ;
        DBMS_OUTPUT.PUT_LINE('LLEGA AQUI V_COUNT: ' || V_COUNT);
        IF V_COUNT > 0 THEN
            -- Actualizar registro
            
            SELECT ID INTO V_ID_USER FROM DF_USUARIO WHERE ID = P_ID OR CODPERSONA = P_CODPERSONA;
            
            UPDATE DF_USUARIO
            SET 
                CODUSUARIOROL   = P_CODUSUARIOROL,
                CODPERSONA      = P_CODPERSONA,
                USUARIO         = P_USUARIO,
                NOMBRE          = P_NOMBRE,
                --CLAVE           = P_CLAVE,
                ESTADO          = P_ESTADO,
                TOKENAZURE      = P_TOKENAZURE,
                AZUREEMAIL      = P_AZUREEMAIL,
                FECHAMODIFICACION = CURRENT_TIMESTAMP
            WHERE ID = V_ID_USER;
            
            BEGIN 
                SELECT ID INTO V_ID_ROL_USER FROM DF_USUARIO_ROLES WHERE CODUSUARIO = V_ID_USER;
            EXCEPTION
            WHEN NO_DATA_FOUND THEN
                DBMS_OUTPUT.PUT_LINE('LLEGA AQUI MAL V_ID_ROL_USER: ' || V_ID_ROL_USER);
                V_ID_ROL_USER := 0;
            END;
    
            IF V_ID_ROL_USER = 0 THEN 
                INSERT INTO DF_USUARIO_ROLES (CODUSUARIO, CODUSUARIOROL) VALUES (V_ID_CREADO, P_CODUSUARIOROL);    
            ELSE
                UPDATE DF_USUARIO_ROLES SET CODUSUARIO = V_ID_USER, CODUSUARIOROL = P_CODUSUARIOROL WHERE ID = V_ID_ROL_USER;
            END IF;
    
            V_ID_CREADO := V_ID_USER;
            V_S_RESPUESTA := 'Registro actualizado exitosamente';
        ELSE
            -- Insertar nuevo registro
            INSERT INTO DF_USUARIO (
                CODUSUARIOROL, CODPERSONA, USUARIO, NOMBRE, CLAVE,
                ESTADO, TOKENAZURE, AZUREEMAIL, FECHACREACION, FECHAMODIFICACION
            ) VALUES (
                P_CODUSUARIOROL, P_CODPERSONA, P_USUARIO, P_NOMBRE, P_CLAVE,
                P_ESTADO, P_TOKENAZURE, P_AZUREEMAIL, CURRENT_TIMESTAMP, CURRENT_TIMESTAMP
            )
            RETURNING ID INTO V_ID_CREADO;
            
            INSERT INTO DF_USUARIO_ROLES (CODUSUARIO, CODUSUARIOROL) VALUES (V_ID_CREADO, P_CODUSUARIOROL);

            V_S_RESPUESTA := 'Registro insertado exitosamente';
        END IF;

        COMMIT;
        V_S_CODIGO := 200;
    EXCEPTION
        WHEN OTHERS THEN
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'Error: ' || SQLERRM;
            ROLLBACK;
    END;
END SP_SAVE_DF_USUARIO;
