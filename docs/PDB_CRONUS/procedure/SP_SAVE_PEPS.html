PROCEDURE SP_SAVE_PEPS (
    P_ID_PEP IN NUMBER,
    P_PEP IN VARCHAR2,
    P_CODEMPLEADO_CREA IN NUMBER,
    P_CODEMPLEADO_MODIFICA IN NUMBER,
    P_CODGERENTE IN NUMBER,
    P_ESTADO IN NUMBER,
    P_URL_SOPORTE IN VARCHAR2,
    P_PRESUPUESTO NUMBER,
    P_OBSERVACIONES VARCHAR2,
    V_ID_CREADO OUT NUMBER,
    V_S_RESPUESTA OUT VARCHAR2,
    V_S_CODIGO OUT NUMBER
)
AS
    V_COUNT NUMBER;
    V_ID_PEP NUMBER;
    V_ESTADO NUMBER;
BEGIN
    BEGIN
        SELECT COUNT(*)
        INTO V_COUNT
        FROM DF_PEP
        WHERE ID = P_ID_PEP OR PEP = P_PEP;

        -- Si el registro existe, actualizarlo; de lo contrario, insertar uno nuevo
        IF V_COUNT > 0 THEN

            SELECT ID, ESTADO
                INTO V_ID_PEP, V_ESTADO
            FROM DF_PEP
                WHERE ID = P_ID_PEP OR PEP = P_PEP
                AND ROWNUM = 1;

            IF P_ESTADO <> 1 AND P_ESTADO <> 4 THEN
                    UPDATE DF_PEP
                SET PEP = P_PEP,
                    FECHAMODIFICACION = CURRENT_TIMESTAMP,
                    CODGERENTE = P_CODGERENTE,
                    ESTADO = P_ESTADO,
                    URL_SOPORTE = P_URL_SOPORTE,
                    CODEMPLEADOMODIFICA = P_CODEMPLEADO_MODIFICA,
                    PRESUPUESTO = ROUND(P_PRESUPUESTO,2),
                    OBSERVACIONES = P_OBSERVACIONES
                WHERE ID = V_ID_PEP;
            END IF;
            
            V_ID_CREADO := V_ID_PEP;
            V_S_RESPUESTA := 'Registro actualizado exitosamente '|| P_PEP;
        ELSE
            INSERT INTO DF_PEP (PEP, FECHACREACION,CODEMPLEADOCREO, CODGERENTE, ESTADO, URL_SOPORTE, PRESUPUESTO, OBSERVACIONES)
            VALUES (P_PEP, CURRENT_TIMESTAMP, P_CODEMPLEADO_CREA, P_CODGERENTE, P_ESTADO, P_URL_SOPORTE, ROUND(P_PRESUPUESTO), TRIM(P_OBSERVACIONES))
            RETURNING ID INTO V_ID_CREADO;
            V_S_RESPUESTA := 'Registro insertado exitosamente '|| P_PEP;
        END IF;

        COMMIT;
        V_S_CODIGO := 200;
    EXCEPTION
        WHEN OTHERS THEN
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'Error: ' || SQLERRM;
    END;
END SP_SAVE_PEPS;
