PROCEDURE SP_RPT_INIT_HORAS_CEL_COR (
    P_CODCOR IN NUMBER,                -- Codempleado para realizar consulta por celula
    P_CODCELULA IN NUMBER,
    P_FECHA_INI IN DATE,                -- Fecha de inicio de la tarea
    P_FECHA_FIN IN DATE,                -- Fecha de fin de la tarea
    P_CURSOR_RPT_INIT_HORAS_CEL_COR OUT SYS_REFCURSOR, -- Cursor de salida
    V_S_RESPUESTA OUT VARCHAR2,         -- Respuesta de salida
    V_S_CODIGO OUT NUMBER               -- Código de respuesta
)
AS
    V_COUNT NUMBER;  -- Variable para verificar si el registro existe
    V_CELULA NUMBER;
BEGIN
    BEGIN


        OPEN P_CURSOR_RPT_INIT_HORAS_CEL_COR FOR
             SELECT E.ID, P.NOMCOMPLETO, COALESCE(SUM(EC.HORAS), 0) AS HORAS 
                FROM DF_EMPLEADO E
                    INNER JOIN DF_PERSONA P ON E.CODPERSONA = P.ID
                    INNER JOIN DF_ESTRUCTURAORGANIZACIONAL EO ON EO.ID = E.CODESTRUCTURAORGANIZACIONAL
                    LEFT JOIN DF_EMPLEADOCONTROL EC ON EC.CODEMPLEADO = E.ID 
                        AND EC.FECHAHORAINICIO BETWEEN P_FECHA_INI AND P_FECHA_FIN
                        AND EC.FECHAHORAFIN BETWEEN P_FECHA_INI AND P_FECHA_FIN
                    LEFT JOIN DF_TAREA T ON T.ID = EC.CODTAREA AND T.ESTADO = 'A'
                WHERE E.VIGENTE = 1
                    AND (P_CODCELULA = 0 OR E.CODESTRUCTURAORGANIZACIONAL = P_CODCELULA)
                    AND (P_CODCOR = 0 OR EO.CODPADRE = P_CODCOR)
                GROUP BY 
                    E.ID, 
                    P.NOMCOMPLETO;
        V_S_RESPUESTA := 'Retorno de datos completo ';
        V_S_CODIGO := 200;

    EXCEPTION
        WHEN OTHERS THEN
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'Error en el procedimiento: ' || SQLERRM;
            ROLLBACK;
    END;
END;
