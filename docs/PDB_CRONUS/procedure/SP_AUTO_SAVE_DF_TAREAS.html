PROCEDURE "SP_AUTO_SAVE_DF_TAREAS" (
    P_ID_PROYECTO_ESTIMACION IN NUMBER, 
    P_ID_TAREA_PADRE IN NUMBER,
    P_ID_EMPLEADO_ASIGNADO IN NUMBER,
    V_S_RESPUESTA OUT VARCHAR2,
    V_S_CODIGO OUT NUMBER
) AS
    V_ID_PROYECTO NUMBER;
    V_NOMBRE_ESTIMACION VARCHAR2(350 BYTE);
    V_NOMBRE_ENTREGABLE VARCHAR2(350 BYTE);
    V_CANTIDAD_HORAS NUMBER;
    V_CANTIDAD_HORAS_TOTAL NUMBER;
    V_FECHA_INICIO DATE;
    V_FECHA_FIN DATE;
    V_FECHA_INICIO_TAREA_PADRE DATE;
    V_FECHA_FIN_TAREA_PADRE DATE;
    V_ID_ESTIMACION NUMBER;
    V_CANT_DIGITOS_POR_NIVEL NUMBER := 10;
    V_JERARQUIA VARCHAR2(1000);
    V_ESTADO_TAREA VARCHAR2(1) := 'A';
    V_CODTAREATIPO NUMBER;
    V_CODTAREAESTADO NUMBER;
    V_CODEMPLEADOCREO NUMBER;
    V_CODEMPLEADOASIGNADO NUMBER;
    V_NIVEL NUMBER;
    V_ID_TAREAPADRE_CREADA NUMBER;
    V_ID_TAREAHIJA_CREADA NUMBER;
    V_ID_TAREAPADRE_UPDATE NUMBER := 0;
    V_ID_TAREAHIJA_UPDATE NUMBER := 0;
    V_FECHAINIREAL DATE;
    V_FECHAFINREAL DATE;
    V_TIEMPOREAL NUMBER;
    V_VALIDACION_UPDATE BOOLEAN := TRUE;
    V_NOMBRE_TAREA_HIJA VARCHAR2(4000);
BEGIN

    BEGIN
    
    
    
    --Consulta el proyecto estimacion por el id que llega
    SELECT ID_PROYECTO, NOMBRE_ESTIMACION
    INTO V_ID_PROYECTO, V_NOMBRE_ESTIMACION 
    FROM DF_PROYECTO_ESTIMACION 
    WHERE ID = P_ID_PROYECTO_ESTIMACION;
    
    
    -- valida si existe las tareas, para modificar el empleado asignado, siempre y cuando no tenga registros de tiempos
    BEGIN
        SELECT ID INTO V_ID_TAREAPADRE_UPDATE FROM DF_TAREA WHERE NOMBRE = V_NOMBRE_ESTIMACION AND CODEMPLEADOASIGNADO = P_ID_EMPLEADO_ASIGNADO ORDER BY ID DESC;
        DBMS_OUTPUT.PUT_LINE('LLEGA AQU BIEN: ' || V_ID_TAREAPADRE_UPDATE);

    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            DBMS_OUTPUT.PUT_LINE('LLEGA AQUI MAL: ' || V_ID_TAREAPADRE_UPDATE);
            V_ID_TAREAPADRE_UPDATE := NULL;
    END;
    
    DBMS_OUTPUT.PUT_LINE('V_ID_TAREAPADRE_UPDATE: ' || V_ID_TAREAPADRE_UPDATE);
    
    /*IF V_ID_TAREAPADRE_UPDATE IS NOT NULL THEN
        FOR CURSOR_TAREAS_UPD IN (
            SELECT ID, FECHAINIREAL, FECHAFINREAL, TIEMPOREAL FROM DF_TAREA 
                WHERE CODTAREAPADRE = V_ID_TAREAPADRE_UPDATE
        ) LOOP
            V_ID_TAREAHIJA_UPDATE := CURSOR_TAREAS_UPD.ID;
            V_FECHAINIREAL := CURSOR_TAREAS_UPD.FECHAINIREAL;
            V_FECHAFINREAL := CURSOR_TAREAS_UPD.FECHAFINREAL;
            V_TIEMPOREAL := CURSOR_TAREAS_UPD.TIEMPOREAL;
            IF V_FECHAINIREAL IS NOT NULL AND V_FECHAFINREAL IS NOT NULL AND V_TIEMPOREAL IS NOT NULL THEN
                V_S_RESPUESTA := 'No se puede asignar usuario a las tareas, porque ya tienen registro de horas';
                V_VALIDACION_UPDATE := FALSE;
                EXIT;
            END IF; 
        END LOOP;
        
        IF V_VALIDACION_UPDATE  THEN 
            UPDATE DF_TAREA SET CODEMPLEADOASIGNADO = P_ID_EMPLEADO_ASIGNADO WHERE ID = V_ID_TAREAPADRE_UPDATE;
            UPDATE DF_TAREA SET CODEMPLEADOASIGNADO = P_ID_EMPLEADO_ASIGNADO WHERE CODTAREAPADRE = V_ID_TAREAPADRE_UPDATE;
        END IF;
        
    ELSE   */ 
    --Consulta la cantidad total de horas, la minima fecha de inicio y la maxima fechafin, de la tabla estimacion
        SELECT SUM(CANTIDAD) AS CANTIDAD_HORAS,  MIN(FECHAINICIO) AS FECHAINICIO, MAX(FECHAFIN) AS FECHAFIN 
        INTO  V_CANTIDAD_HORAS_TOTAL,  V_FECHA_INICIO_TAREA_PADRE,  V_FECHA_FIN_TAREA_PADRE        
        FROM DF_ESTIMACION
        WHERE ID_PROYECTO_ESTIMACION = P_ID_PROYECTO_ESTIMACION;
    
    
        UPDATE DF_PROYECTO_ESTIMACION SET CODTAREA = P_ID_TAREA_PADRE 
            WHERE ID = P_ID_PROYECTO_ESTIMACION;
    
        INSERT INTO DF_ESTIMACION_TAREA (CODTAREA, CODESTIMACION) VALUES (P_ID_TAREA_PADRE,P_ID_PROYECTO_ESTIMACION);
        --Consulta datos de la tarea padre a la cual se le van a asignar las nuevas tareas hijas
        SELECT CODTAREATIPO, CODTAREAESTADO, CODEMPLEADOCREO, NIVEL, JERARQUIA
        INTO V_CODTAREATIPO, V_CODTAREAESTADO, V_CODEMPLEADOCREO,  V_NIVEL, V_JERARQUIA
        FROM DF_TAREA t WHERE t.ID = P_ID_TAREA_PADRE;
    
    
        --SELECT MAX(ID)+1 INTO V_ID_TAREAPADRE_CREADA FROM DF_TAREA;
        DBMS_OUTPUT.PUT_LINE('V_ID_TAREAPADRE_CREADA: ' || V_ID_TAREAPADRE_CREADA);
        --Se realiza el primer insert, de la tarea padre nivel 3, asociado a la tarea padre nivel 2 que llega por parametro y nos retorna el valor creado
        /*INSERT INTO DF_TAREA (CODTAREAPADRE, CODTAREATIPO, CODTAREAESTADO, CODEMPLEADOCREO, CODEMPLEADOASIGNADO, CODPROYECTO,
                NOMBRE, NIVEL, DESCRIPCION, FECHAINIESTIMADA, FECHAFINESTIMADA, TIEMPOESTIMADO, 
                ESLOGRO, ESTADO, JERARQUIA) 
        VALUES (P_ID_TAREA_PADRE, V_CODTAREATIPO, V_CODTAREAESTADO, V_CODEMPLEADOCREO, P_ID_EMPLEADO_ASIGNADO, V_ID_PROYECTO,
                V_NOMBRE_ESTIMACION, V_NIVEL+1, 'Tarea ' || V_NOMBRE_ESTIMACION, V_FECHA_INICIO_TAREA_PADRE, V_FECHA_FIN_TAREA_PADRE,
                V_CANTIDAD_HORAS_TOTAL, 1, V_ESTADO_TAREA, V_JERARQUIA)
        RETURNING ID INTO V_ID_TAREAPADRE_CREADA;
        DBMS_OUTPUT.PUT_LINE('V_ID_TAREAPADRE_CREADA: ' || V_ID_TAREAPADRE_CREADA);
    
        --Modifica la jearquia de la tarea creada
        UPDATE DF_TAREA 
            SET JERARQUIA = JERARQUIA || LPAD(V_ID_TAREAPADRE_CREADA, LENGTH(TO_CHAR(V_ID_TAREAPADRE_CREADA))+5, '0') 
            WHERE ID=V_ID_TAREAPADRE_CREADA;
    
        SELECT CODTAREATIPO, CODTAREAESTADO, CODEMPLEADOCREO, NIVEL, JERARQUIA
            INTO V_CODTAREATIPO, V_CODTAREAESTADO, V_CODEMPLEADOCREO,  V_NIVEL, V_JERARQUIA
            FROM DF_TAREA t WHERE t.ID = V_ID_TAREAPADRE_CREADA;*/
    
        -- Cursor que busca las estimaciones creadas, las cuales se volveran tareas hijas
        FOR CURSOR_PROYECTO_ESTIMACION IN (
            SELECT ID, ENTREGABLE, CANTIDAD, FECHAINICIO, FECHAFIN
            FROM DF_ESTIMACION
            WHERE ID_PROYECTO_ESTIMACION = P_ID_PROYECTO_ESTIMACION
        )
        LOOP
            V_ID_ESTIMACION := CURSOR_PROYECTO_ESTIMACION.ID;
            V_NOMBRE_ENTREGABLE := CURSOR_PROYECTO_ESTIMACION.ENTREGABLE;
            V_CANTIDAD_HORAS := CURSOR_PROYECTO_ESTIMACION.CANTIDAD;
            V_FECHA_INICIO := CURSOR_PROYECTO_ESTIMACION.FECHAINICIO;
            V_FECHA_FIN := CURSOR_PROYECTO_ESTIMACION.FECHAFIN;
            
            V_NOMBRE_TAREA_HIJA := V_NOMBRE_ESTIMACION || '/' || V_NOMBRE_ENTREGABLE;
            --SELECT MAX(ID)+1 INTO V_ID_TAREAHIJA_CREADA FROM DF_TAREA;
            
            BEGIN
                SELECT ID INTO V_ID_TAREAHIJA_CREADA FROM DF_TAREA WHERE NOMBRE = V_NOMBRE_TAREA_HIJA ORDER BY ID DESC FETCH FIRST 1 ROW ONLY;
                DBMS_OUTPUT.PUT_LINE('LLEGA AQU BIEN: ' || V_ID_TAREAPADRE_UPDATE);

            EXCEPTION
                WHEN NO_DATA_FOUND THEN
                    DBMS_OUTPUT.PUT_LINE('LLEGA AQUI MAL: ' || V_ID_TAREAPADRE_UPDATE);
                    V_ID_TAREAHIJA_CREADA := NULL;
            END;
            
            IF V_ID_TAREAHIJA_CREADA IS NULL THEN 
            -- INSERT DE LA TAREA HIJA-PADRE
                INSERT INTO DF_TAREA (CODTAREAPADRE, CODTAREATIPO, CODTAREAESTADO, CODEMPLEADOCREO, CODEMPLEADOASIGNADO, CODPROYECTO,
                    NOMBRE, NIVEL, DESCRIPCION, FECHAINIESTIMADA, FECHAFINESTIMADA, TIEMPOESTIMADO, 
                    ESLOGRO, ESTADO, JERARQUIA, CODLINEAPRODUCTO, DESCRIPCION_SOLICITUD) 
                VALUES (P_ID_TAREA_PADRE, V_CODTAREATIPO, V_CODTAREAESTADO, V_CODEMPLEADOCREO, NULL, V_ID_PROYECTO,
                    V_NOMBRE_TAREA_HIJA, V_NIVEL+1, 'Tarea ' || V_NOMBRE_TAREA_HIJA, 
                    V_FECHA_INICIO, V_FECHA_FIN, V_CANTIDAD_HORAS, 1, V_ESTADO_TAREA, V_JERARQUIA, 1, 'CREADA POR ESTIMACIÓN DE MANERA AUTOMATICA')
                RETURNING ID INTO V_ID_TAREAHIJA_CREADA;
        
                -- UPDATE DE LA TAREA HIJA CAMPO JERARQUIA
                UPDATE DF_TAREA 
                SET JERARQUIA = JERARQUIA || LPAD(V_ID_TAREAHIJA_CREADA, LENGTH(TO_CHAR(V_ID_TAREAHIJA_CREADA))+5, '0') 
                WHERE ID=V_ID_TAREAHIJA_CREADA;    
            
            ELSE 
                UPDATE DF_TAREA SET
                    FECHAINIESTIMADA = V_FECHA_INICIO,
                    FECHAFINESTIMADA = V_FECHA_FIN,
                    TIEMPOESTIMADO = V_CANTIDAD_HORAS
                    WHERE ID = V_ID_TAREAHIJA_CREADA;
            END IF;
            -- Realizar operaciones con los datos
            DBMS_OUTPUT.PUT_LINE('V_ID_ESTIMACION: ' || V_ID_ESTIMACION);
            DBMS_OUTPUT.PUT_LINE('V_NOMBRE_ENTREGABLE: ' || V_NOMBRE_ENTREGABLE);
            DBMS_OUTPUT.PUT_LINE('V_NOMBRE_ESTIMACION: ' || V_NOMBRE_ESTIMACION);
        END LOOP;
        COMMIT;
        V_S_CODIGO    := 200;
		V_S_RESPUESTA := 'creados Exitosamente';
    EXCEPTION
            WHEN NO_DATA_FOUND THEN
            V_S_CODIGO    := 500;
            V_S_RESPUESTA := 'No existe estructura';
            rollback;
    END;
END SP_AUTO_SAVE_DF_TAREAS;
