PROCEDURE SP_SAVE_DF_COSTO_EMPLEADO (
    P_COSTO_PUNTO IN NUMBER,
    P_COSTO_MES IN NUMBER,
    P_PUNTOS_MES IN NUMBER,
    P_FACTOR_PUNTO IN NUMBER,
    P_MODALIDAD IN VARCHAR2,
    P_CODEMPLEADO IN NUMBER,
    P_DESDE IN DATE,
    P_HASTA IN DATE,
    V_S_RESPUESTA OUT VARCHAR2,
    V_S_CODIGO OUT NUMBER
)
AS
    V_EXISTE NUMBER;
BEGIN
    BEGIN
        IF P_CODEMPLEADO IS NULL THEN
            V_S_RESPUESTA := 'El campo CODEMPLEADO no puede ser nulo.';
            V_S_CODIGO := 400;
            RETURN;
        END IF;

        -- Validar si ya existe un registro para el mismo CODEMPLEADO y rango de fechas
        SELECT COUNT(*) INTO V_EXISTE FROM DF_COSTO_EMPLEADO WHERE CODEMPLEADO = P_CODEMPLEADO AND VIGENTE = 1;

        IF V_EXISTE > 0 THEN
            -- Actualizar
            UPDATE DF_COSTO_EMPLEADO
            SET HASTA = P_DESDE
            WHERE CODEMPLEADO = P_CODEMPLEADO AND VIGENTE = 1;

            INSERT INTO DF_COSTO_EMPLEADO (COSTO_PUNTO, COSTO_MES, PUNTOS_MES, FACTOR_PUNTO, MODALIDAD, CODEMPLEADO, DESDE)
                VALUES (P_COSTO_PUNTO, P_COSTO_MES, P_PUNTOS_MES, P_FACTOR_PUNTO, P_MODALIDAD, P_CODEMPLEADO, P_DESDE);

            V_S_RESPUESTA := 'Registro actualizado exitosamente.';
        ELSE
            -- Insertar
            INSERT INTO DF_COSTO_EMPLEADO (COSTO_PUNTO, COSTO_MES, PUNTOS_MES, FACTOR_PUNTO, MODALIDAD, CODEMPLEADO, DESDE)
                VALUES (P_COSTO_PUNTO, P_COSTO_MES, P_PUNTOS_MES, P_FACTOR_PUNTO, P_MODALIDAD, P_CODEMPLEADO, P_DESDE);

            V_S_RESPUESTA := 'Registro insertado exitosamente.';
        END IF;

        V_S_CODIGO := 200;
        COMMIT;
    EXCEPTION
        WHEN OTHERS THEN
            V_S_RESPUESTA := 'Error: ' || SQLERRM;
            V_S_CODIGO := 500;
            ROLLBACK;
    END;
END SP_SAVE_DF_COSTO_EMPLEADO;
