PROCEDURE SP_SAVE_DF_PERSONA (
    P_ID IN NUMBER,
    P_CODTIPODOCUMENTO IN NUMBER,
    P_NUMERODOCUMENTO_OLD IN NUMBER,
    P_NOMBRE1 IN VARCHAR2,
    P_NOMBRE2 IN VARCHAR2,
    P_APELLIDO1 IN VARCHAR2,
    P_APELLIDO2 IN VARCHAR2,
    P_TELEFONO IN VARCHAR2,
    P_CORREO IN VARCHAR2,
    P_DIRECCIONRESIDENCIA IN VARCHAR2,
    P_GENERO IN VARCHAR2,
    P_FECHANACIMIENTO IN DATE,
    P_CODTERCEROEPS IN NUMBER,
    P_UBICACION IN VARCHAR2,
    P_CELULAR IN VARCHAR2,
    P_GRUPOSANGUINEO IN VARCHAR2,
    P_CONTACTOEMERGENCIANOMBRE IN VARCHAR2,
    P_CONTACTOEMERGENCIATELEFONO IN VARCHAR2,
    P_NUMERODOCUMENTO IN VARCHAR2,
    V_ID_CREADO OUT NUMBER,
    V_S_CODIGO OUT NUMBER,
    V_S_MENSAJE OUT VARCHAR2
)
AS
    V_COUNT NUMBER;
    V_ID NUMBER;
BEGIN
    BEGIN
        SELECT COUNT(*) INTO V_COUNT FROM DF_PERSONA 
            WHERE (NUMERODOCUMENTO = P_NUMERODOCUMENTO AND CODTIPODOCUMENTO = P_CODTIPODOCUMENTO) OR ID = P_ID;

        IF V_COUNT > 0 THEN
        
            SELECT ID INTO V_ID FROM DF_PERSONA 
            WHERE (NUMERODOCUMENTO = P_NUMERODOCUMENTO AND CODTIPODOCUMENTO = P_CODTIPODOCUMENTO) OR ID = P_ID;
        
            UPDATE DF_PERSONA
            SET CODTIPODOCUMENTO = P_CODTIPODOCUMENTO,
                NUMERODOCUMENTO_OLD = P_NUMERODOCUMENTO_OLD,
                NOMBRE1 = P_NOMBRE1,
                NOMBRE2 = P_NOMBRE2,
                APELLIDO1 = P_APELLIDO1,
                APELLIDO2 = P_APELLIDO2,
                TELEFONO = P_TELEFONO,
                CORREO = P_CORREO,
                DIRECCIONRESIDENCIA = P_DIRECCIONRESIDENCIA,
                GENERO = P_GENERO,
                FECHANACIMIENTO = P_FECHANACIMIENTO,
                CODTERCEROEPS = P_CODTERCEROEPS,
                UBICACION = P_UBICACION,
                CELULAR = P_CELULAR,
                GRUPOSANGUINEO = P_GRUPOSANGUINEO,
                CONTACTOEMERGENCIANOMBRE = P_CONTACTOEMERGENCIANOMBRE,
                CONTACTOEMERGENCIATELEFONO = P_CONTACTOEMERGENCIATELEFONO,
                NUMERODOCUMENTO = P_NUMERODOCUMENTO,
                FECHAMODIFICACION = CURRENT_TIMESTAMP
            WHERE ID = V_ID;

            V_ID_CREADO := V_ID;
            V_S_MENSAJE := 'Registro actualizado exitosamente';
        ELSE
            INSERT INTO DF_PERSONA (CODTIPODOCUMENTO, NUMERODOCUMENTO_OLD, NOMBRE1, NOMBRE2,
                APELLIDO1, APELLIDO2, TELEFONO, CORREO, DIRECCIONRESIDENCIA,
                GENERO, FECHANACIMIENTO, FECHACREACION, FECHAMODIFICACION,
                CODTERCEROEPS, UBICACION, CELULAR, GRUPOSANGUINEO,
                CONTACTOEMERGENCIANOMBRE, CONTACTOEMERGENCIATELEFONO, NUMERODOCUMENTO
            ) VALUES (P_CODTIPODOCUMENTO, P_NUMERODOCUMENTO_OLD, P_NOMBRE1, P_NOMBRE2,
                P_APELLIDO1, P_APELLIDO2, P_TELEFONO, P_CORREO, P_DIRECCIONRESIDENCIA,
                P_GENERO, P_FECHANACIMIENTO, CURRENT_TIMESTAMP, CURRENT_TIMESTAMP,
                P_CODTERCEROEPS, P_UBICACION, P_CELULAR, P_GRUPOSANGUINEO,
                P_CONTACTOEMERGENCIANOMBRE, P_CONTACTOEMERGENCIATELEFONO, P_NUMERODOCUMENTO
            ) RETURNING ID INTO V_ID_CREADO;
            V_S_MENSAJE := 'Registro insertado exitosamente';
        END IF;

        COMMIT;
        V_S_CODIGO := 200;

    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            V_S_CODIGO := 500;
            V_S_MENSAJE := 'No se encontró el registro especificado';
            ROLLBACK;
        WHEN OTHERS THEN
            V_S_CODIGO := 500;
            V_S_MENSAJE := 'Error: ' || SQLERRM;
            ROLLBACK;
    END;
END SP_SAVE_DF_PERSONA;
