PROCEDURE "SP_CUENTAS_RECHAZADAS" AS

v_idopera NUMBER(20);


BEGIN
FOR dta IN
  (SELECT A.ID_BENEFICIARIO,
          A.CONTR_APOR_MINTIC,
          A.CUENTA
   FROM MINTIC_OPERACIONES_ODI A
   WHERE A.ESTADO_FONTIC = 'RECHAZADO'
     AND A.ID_BENEFICIARIO IS NOT NULL ) LOOP 

v_idopera :=0;

--1.Retirar cuentas del proyecto.
UPDATE MINTIC_OPERACIONES_ODI  
    SET
        ID_BENEFICIARIO = NULL,
        FECHA_NOVEDAD = NULL,
        NOVEDAD = NULL,
        FECHA_RETIRO = NULL            
            WHERE
            ID_BENEFICIARIO = dta.ID_BENEFICIARIO
            AND CONTR_APOR_MINTIC = dta.CONTR_APOR_MINTIC
            AND CUENTA = dta.CUENTA;

COMMIT;                  

        EXECUTE IMMEDIATE 
                  'SELECT NVL(A.CUENTA,0) CUENTA
                    FROM MINTIC_OPERACIONES_ODI A
                    INNER JOIN
                    (
                    SELECT 
                    ID_BENEFICIARIO,
                    CONTR_APOR_MINTIC, 
                    MAX(FECHA_NOVEDAD) FECHA_NOVEDAD
                    FROM  MINTIC_OPERACIONES_ODI
                    WHERE
                    ID_BENEFICIARIO = :ID_BENEFICIARIO	
                    AND CONTR_APOR_MINTIC = :CONTR_APOR_MINTIC	
                    GROUP BY 
                    ID_BENEFICIARIO, CONTR_APOR_MINTIC) B ON  
                    A.ID_BENEFICIARIO = B.ID_BENEFICIARIO AND A.CONTR_APOR_MINTIC = B.CONTR_APOR_MINTIC AND A.FECHA_NOVEDAD = B.FECHA_NOVEDAD
                  ' INTO v_idopera USING dta.ID_BENEFICIARIO,
                                         dta.CONTR_APOR_MINTIC;


      UPDATE MINTIC_OPERACIONES_ODI
                    SET REMPLAZO = 'S'
                    WHERE ID_BENEFICIARIO = dta.ID_BENEFICIARIO
                      AND CONTR_APOR_MINTIC = dta.CONTR_APOR_MINTIC
                      AND CUENTA = v_idopera;
COMMIT;


END LOOP;


COMMIT;

END;
