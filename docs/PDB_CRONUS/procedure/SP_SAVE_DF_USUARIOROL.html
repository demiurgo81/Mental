PROCEDURE "SP_SAVE_DF_USUARIOROL" (
    P_ID IN NUMBER,
    P_NOMBRE IN VARCHAR2,
    V_S_RESPUESTA OUT VARCHAR2,
    V_ID_CREADO OUT NUMBER,
    V_S_CODIGO OUT NUMBER
)
AS
    V_COUNT NUMBER;
    V_ID NUMBER;
BEGIN
    -- Verificar si el registro ya existe en la tabla
    SELECT COUNT(*)
    INTO V_COUNT
    FROM DF_USUARIOROL
    WHERE ID = P_ID;

    -- Si el registro existe, actualizarlo; de lo contrario, insertar uno nuevo
    IF V_COUNT > 0 THEN
        UPDATE DF_USUARIOROL
        SET NOMBRE = P_NOMBRE,
            FECHAMODIFICACION = CURRENT_TIMESTAMP
        WHERE ID = P_ID;
        V_ID_CREADO := P_ID;
        V_S_RESPUESTA := 'Registro actualizado exitosamente';
    ELSE
        INSERT INTO DF_USUARIOROL (NOMBRE, FECHACREACION, FECHAMODIFICACION)
        VALUES ( P_NOMBRE, CURRENT_TIMESTAMP, CURRENT_TIMESTAMP)
        RETURNING ID INTO V_ID_CREADO;
        V_S_RESPUESTA := 'Registro insertado exitosamente';
    END IF;

    COMMIT;
    V_S_CODIGO := 200;

EXCEPTION
    WHEN OTHERS THEN
        ROLLBACK;
        V_S_CODIGO := 500;
        V_S_RESPUESTA := 'Error: ' || SQLERRM;
END;
