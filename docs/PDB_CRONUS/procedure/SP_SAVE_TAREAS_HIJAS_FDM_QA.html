PROCEDURE SP_SAVE_TAREAS_HIJAS_FDM_QA(
    P_ID_TAREA_HIJA IN NUMBER,
    P_ID_TAREA_PADRE IN NUMBER,
    P_CODEMPLEADO IN NUMBER,
    P_NOMBRE_TAREA IN VARCHAR2,
    P_FECHA_INI IN DATE, 
    P_FECHA_FIN IN DATE,
    P_HORAS IN NUMBER,
    P_TIPO_TAREA IN NUMBER,
    P_DESCRIP_TAREA IN VARCHAR2,
    P_ID_EMPLEADOCONTROL IN NUMBER,
    V_ID_CREADO OUT NUMBER,
    V_S_RESPUESTA OUT VARCHAR2,
    V_S_CODIGO OUT NUMBER
)
AS
    V_CELULA NUMBER;
    V_RESPUESTA VARCHAR2(4000);
    V_CODIGO NUMBER;
    V_PROYECTO NUMBER;
    V_MENSAJE VARCHAR2(4000);
    V_CODEMPLEADOLIDER NUMBER;
    V_CANT_TAREA_PADRE NUMBER;
    V_COD_TAREAESTADO NUMBER := 3;
    V_NIVEL_TAREA_PADRE NUMBER := 1;
    V_ID_TAREA_PADRE NUMBER;
    V_ESLOGRO NUMBER := 1;
    V_ESTADO VARCHAR(1) := 'A';
    V_ID_TAREA NUMBER := 0;
    V_ID_TAREA_HIJA NUMBER;
    V_ID_EMPLEADOCONTROL NUMBER;
    V_RESPUESTA_EMPLEADO VARCHAR2(4000);
    V_HORAS_INIT NUMBER:= 0.01;
    V_ID_INICIATIVA NUMBER;
    V_ID_INI_RETORNO NUMBER;
    V_NOMBRE_TAREA VARCHAR2(400);
    V_HORAS_EMPLEADO NUMBER(12,5);
    V_HORASEMPCON_SUM NUMBER(12,5);
    V_HORAS_DF_TAREA NUMBER(12,2);
    V_FECHA_FIN_DF_TAREA DATE;
BEGIN
    BEGIN
        -- VALIDACION DE HORAS EN EL DÍA
        BEGIN
        SELECT SUM(HORAS) INTO V_HORASEMPCON_SUM  FROM DF_EMPLEADOCONTROL
            WHERE CODEMPLEADO = P_CODEMPLEADO AND FECHAHORAINICIO BETWEEN P_FECHA_INI AND P_FECHA_FIN 
                AND FECHAHORAFIN BETWEEN P_FECHA_INI AND P_FECHA_FIN;
            DBMS_OUTPUT.PUT_LINE('HORAS DEL EMPLEADO EN EL DÍA V_HORASEMPCON_SUM ' || V_HORASEMPCON_SUM);
            V_HORAS_EMPLEADO := P_HORAS + V_HORASEMPCON_SUM;
        EXCEPTION
            WHEN NO_DATA_FOUND THEN
                V_HORASEMPCON_SUM := 0;
                DBMS_OUTPUT.PUT_LINE('NO ENCUENTRA HORAS EN EL DÍA' || P_CODEMPLEADO);
        END;
        
        IF V_HORASEMPCON_SUM >= 9 THEN 
            V_ID_CREADO := P_ID_TAREA_HIJA;
            V_S_RESPUESTA := 'No puede tener más de 9 horas en el día ' || P_FECHA_INI;
            V_S_CODIGO := 400;
            return;
        END IF;
        
        IF V_HORAS_EMPLEADO >= 9 THEN
            V_ID_CREADO := P_ID_TAREA_HIJA;
            V_S_RESPUESTA := 'No puede tener más de 9 horas en el día ' || P_FECHA_INI || ', actualmente tiene ' || V_HORASEMPCON_SUM;
            V_S_CODIGO := 400;
            return;
        END IF;
        
        IF P_HORAS > 0.9 THEN 
            V_HORAS_INIT := P_HORAS;
        END IF;
        
        BEGIN
            SELECT ID, CODEMPLEADOCREO, CODPROYECTO, NOMBRE INTO V_ID_TAREA_PADRE, V_CODEMPLEADOLIDER, V_PROYECTO, V_NOMBRE_TAREA FROM DF_TAREA WHERE ID = P_ID_TAREA_PADRE ;
            V_NOMBRE_TAREA := V_NOMBRE_TAREA || '/' || P_NOMBRE_TAREA;
            DBMS_OUTPUT.PUT_LINE('ENCUENTRA EL V_ID_TAREA_PADRE ' || V_ID_TAREA_PADRE);
        EXCEPTION
            WHEN NO_DATA_FOUND THEN
                V_ID_TAREA_PADRE := 0;
                V_CANT_TAREA_PADRE := 0;
                DBMS_OUTPUT.PUT_LINE('NO ENCUENTRA EL V_ID_TAREA_PADRE ' || V_ID_TAREA_PADRE);
        END;
        
        IF P_ID_TAREA_HIJA = 0 THEN 
        -- CONSULTA SI TAREA PADRE EXISTE, SINO LA CREA
            --- CREA TAREA HIJA
            DBMS_OUTPUT.PUT_LINE('LLAMADO A CREAR TAREA_PADRE ' || V_ID_TAREA_PADRE);
            SP_SAVE_DF_TAREA(P_ID_TAREA_HIJA, V_ID_TAREA_PADRE, 
                P_TIPO_TAREA, V_COD_TAREAESTADO, V_CODEMPLEADOLIDER, 
                P_CODEMPLEADO, V_PROYECTO, '', 
                V_NOMBRE_TAREA, V_NIVEL_TAREA_PADRE, P_DESCRIP_TAREA, 
                P_FECHA_INI, P_FECHA_FIN,
                P_FECHA_INI, P_FECHA_FIN, V_HORAS_INIT, V_HORAS_INIT, V_ESLOGRO, V_ESTADO,
                CURRENT_TIMESTAMP, CURRENT_TIMESTAMP, '', 0, V_RESPUESTA, V_CODIGO, V_MENSAJE );
            --RETORNAR EL ID DE LA TAREA PADRE CREADA
            V_ID_TAREA_HIJA := TO_NUMBER(V_RESPUESTA);
            DBMS_OUTPUT.PUT_LINE('CREA TAREAHIJA ' || V_ID_TAREA_HIJA);
        --END IF;
            -- CREA REGISTRO NUEVO DE LA TAREA EN EMPLEADOCONTROL
            
            V_S_RESPUESTA := 'Creación de tarea hija por formulario desarrollo FDM ';    
            V_ID_CREADO := V_ID_TAREA_HIJA;
            
            SP_SAVE_DF_EMPLEADOCONTROL2(0, P_DESCRIP_TAREA, V_HORAS_INIT, P_CODEMPLEADO, V_ID_TAREA_HIJA, 
            P_FECHA_INI, P_FECHA_FIN, V_RESPUESTA_EMPLEADO, V_CODIGO);

        ELSE
            -- MODIFICAR TAREA
            DBMS_OUTPUT.PUT_LINE('ENTRA A MODIFICAT');
            
            
            SELECT TIEMPOREAL, FECHAFINREAL INTO V_HORAS_DF_TAREA, V_FECHA_FIN_DF_TAREA FROM DF_TAREA WHERE ID = P_ID_TAREA_HIJA;

            UPDATE DF_TAREA 
                SET CODTAREAPADRE =  V_ID_TAREA_PADRE,
                    CODPROYECTO = V_PROYECTO,
                    CODTAREATIPO = P_TIPO_TAREA,
                    --NOMBRE = V_NOMBRE_TAREA,
                    DESCRIPCION = P_DESCRIP_TAREA,
                    FECHAINIESTIMADA = P_FECHA_INI, 
                    --FECHAFINESTIMADA = P_FECHA_FIN, 
                    FECHAINIREAL = P_FECHA_INI, 
                    FECHAFINREAL = V_FECHA_FIN_DF_TAREA, 
                    --TIEMPOESTIMADO = V_HORAS_INIT, 
                    TIEMPOREAL = V_HORAS_INIT+V_HORAS_DF_TAREA,
                    FECHAMODIFICACION = CURRENT_TIMESTAMP,
                    DESCRIPCION_SOLICITUD = 'REPORTADA POR CALENDARIO O TAREAS HIJAS'
                    WHERE ID = P_ID_TAREA_HIJA;
                   
            BEGIN 
                SELECT ID, HORAS INTO V_ID_EMPLEADOCONTROL, V_HORAS_EMPLEADO FROM DF_EMPLEADOCONTROL WHERE ID = P_ID_EMPLEADOCONTROL;
            EXCEPTION
            WHEN NO_DATA_FOUND THEN
                V_ID_EMPLEADOCONTROL := 0;
            END;     
            
            SP_SAVE_DF_EMPLEADOCONTROL2(V_ID_EMPLEADOCONTROL, P_DESCRIP_TAREA, V_HORAS_INIT, P_CODEMPLEADO, P_ID_TAREA_HIJA, 
            P_FECHA_INI, P_FECHA_FIN, V_RESPUESTA_EMPLEADO, V_CODIGO);
            
            V_ID_CREADO := P_ID_TAREA_HIJA;
            V_S_RESPUESTA := 'Modificación de tarea hija por formulario desarrollo FDM ';    
        END IF;
        
        V_S_CODIGO := 200;
        
    EXCEPTION
        WHEN OTHERS THEN
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'Error en el procedimiento: ' || SQLERRM;
            ROLLBACK;
    END;
END;
