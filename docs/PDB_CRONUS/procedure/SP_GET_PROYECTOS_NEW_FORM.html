PROCEDURE SP_GET_PROYECTOS_NEW_FORM (
    P_CURSOR_GET_PROYECTOS_NEW_FORM OUT SYS_REFCURSOR, -- Cursor de salida
    V_S_RESPUESTA OUT VARCHAR2,         -- Respuesta de salida
    V_S_CODIGO OUT NUMBER               -- Código de respuesta
)
AS
BEGIN
    BEGIN

        OPEN P_CURSOR_GET_PROYECTOS_NEW_FORM FOR
            SELECT P.ID AS ID, P.CODIGOPROYECTO AS CODIGOPROYECTO, 
                SUBSTR(p.NOMBRE, 1, 100) AS NOMBRE, PR.CODEMPLEADOGERENTE AS CODEMPLEADOGERENTE, 
                PR.ID AS CODPRESUPUESTO, PER.NOMCOMPLETO  AS NOMCOMPLETO
                FROM DF_PROYECTO P 
                LEFT JOIN DF_PRESUPUESTO PR ON P.CODPRESUPUESTO=PR.ID
                LEFT JOIN DF_EMPLEADO E ON E.ID = PR.CODEMPLEADOGERENTE
                LEFT JOIN DF_PERSONA PER ON E.CODPERSONA = PER.ID
                WHERE PR.VIGENCIA IN (TO_CHAR(SYSDATE, 'YYYY'), TO_CHAR(SYSDATE - INTERVAL '1' YEAR, 'YYYY'))
                ORDER BY ID DESC;
        V_S_RESPUESTA := 'Retorno de datos completo ';
        V_S_CODIGO := 200;


    EXCEPTION
        WHEN OTHERS THEN
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'Error en el procedimiento: ' || SQLERRM;
            ROLLBACK;
    END;
END;
