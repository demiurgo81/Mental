PROCEDURE SP_UPLOAD_DATA_FILE_PEP (
    P_PEP IN VARCHAR2,
    P_ESTADO IN VARCHAR2,
    P_GERENTE IN VARCHAR2,
    P_PRESUPUESTO IN NUMBER,
    P_OBSERVACIONES IN VARCHAR2,
    P_URL_SEGUIMIENTO IN VARCHAR2,
    P_CODEMPLEADO IN NUMBER,
    V_ID_CREADO OUT NUMBER,
    V_S_RESPUESTA OUT VARCHAR2,
    V_S_CODIGO OUT NUMBER
)
AS
    V_COUNT NUMBER;
    V_ESTADO NUMBER;
    V_CODEMPLEADO_GERENTE NUMBER;
    V_ID_PEP NUMBER;
    V_RESPUESTA VARCHAR2(4000);
    V_CODIGO NUMBER;
BEGIN
    BEGIN
        -- Verificar si el registro ya existe

        IF P_PEP IS NULL OR TRIM(P_PEP) = '' THEN
            V_S_RESPUESTA := 'El campo PEP no puede estar vacío';
            V_S_CODIGO := 400;
            RETURN;
        END IF;    

        IF P_ESTADO IS NULL OR TRIM(P_ESTADO) = '' THEN
            V_S_RESPUESTA := 'El campo ESTADO no puede estar vacío';
            V_S_CODIGO := 400;
            RETURN;
        END IF;

        IF P_GERENTE IS NULL OR TRIM(P_GERENTE) = '' THEN
            V_S_RESPUESTA := 'El campo GERENTE no puede estar vacío';
            V_S_CODIGO := 400;
            RETURN;
        END IF;

        IF P_PRESUPUESTO IS NULL THEN
            V_S_RESPUESTA := 'El campo PRESUPUESTO no puede estar vacío';
            V_S_CODIGO := 400;
            RETURN;
        END IF;

        IF P_OBSERVACIONES IS NULL OR TRIM(P_OBSERVACIONES) = '' THEN
            V_S_RESPUESTA := 'El campo OBSERVACIONES no puede estar vacío';
            V_S_CODIGO := 400;
            RETURN;
        END IF;

        IF P_URL_SEGUIMIENTO IS NULL OR TRIM(P_URL_SEGUIMIENTO) = '' THEN
            V_S_RESPUESTA := 'El campo URL_SEGUIMIENTO no puede estar vacío';
            V_S_CODIGO := 400;
            RETURN;
        END IF;

        IF P_CODEMPLEADO IS NULL THEN
            V_S_RESPUESTA := 'El campo CODEMPLEADO no puede estar vacío';
            V_S_CODIGO := 400;
            RETURN;
        END IF;
        
        CASE P_ESTADO
            WHEN 'APROBADO' THEN V_ESTADO := 2;
            WHEN 'NO APROBADO' THEN V_ESTADO := 3;
            ELSE V_ESTADO := 1;
        END CASE;
            
        BEGIN
            SELECT E.ID INTO V_CODEMPLEADO_GERENTE 
                FROM DF_EMPLEADO E WHERE E.CODPERSONA IN 
                    (SELECT PER.ID FROM DF_PERSONA PER WHERE PER.NOMCOMPLETO = P_GERENTE);
        EXCEPTION
        WHEN NO_DATA_FOUND THEN
            DBMS_OUTPUT.PUT_LINE('LLEGA AQUI MAL: ' || P_GERENTE);
            V_CODEMPLEADO_GERENTE := 0;
        END;
        -- Confirmar transacción
        
        SP_SAVE_PEPS(0, P_PEP, P_CODEMPLEADO, P_CODEMPLEADO, V_CODEMPLEADO_GERENTE, V_ESTADO, P_URL_SEGUIMIENTO, P_PRESUPUESTO, 
            P_OBSERVACIONES, V_ID_PEP, V_RESPUESTA, V_CODIGO);
        
        V_ID_CREADO := V_ID_PEP;
        V_S_CODIGO := 200;
        V_S_RESPUESTA := V_RESPUESTA;
        COMMIT;
    EXCEPTION
        WHEN OTHERS THEN
            -- Manejo de errores
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'Error: ' || SQLERRM;
            ROLLBACK;
    END;
END SP_UPLOAD_DATA_FILE_PEP;
