PROCEDURE              "TFS_UPDATE_COMPLEMENTO_SP" ( IDLP_IN IN NUMBER, DATE_IN IN DATE)
IS
  V_IDLP NUMBER;
  V_FPD NUMBER;
  V_FECHA_PLAN_DESPLIEGUE DATE;
  V_F1 NUMBER;
  V_F2 NUMBER;
  V_NC NUMBER;
BEGIN
    V_IDLP := 0;
    V_FPD := 0;
    V_NC := 0;
    DBMS_OUTPUT.PUT_LINE('INICIO:'||IDLP_IN);
    DBMS_OUTPUT.PUT_LINE('FECHA:'||DATE_IN);
    SELECT COUNT(ID_COMPLEMENTO) INTO V_IDLP FROM TFS_INDICADORES_COMPLEMENTO  WHERE ID_WI_LP = IDLP_IN;
    DBMS_OUTPUT.PUT_LINE(V_IDLP);
    IF V_IDLP>0 THEN
        DBMS_OUTPUT.PUT_LINE('UPDATE');
        SELECT FECHA_PLAN_DESPLIEGUE INTO V_FECHA_PLAN_DESPLIEGUE FROM TFS_INDICADORES_COMPLEMENTO  WHERE ID_WI_LP = IDLP_IN;
        DBMS_OUTPUT.PUT_LINE('FPD BD: '||V_FECHA_PLAN_DESPLIEGUE);
        
        V_F1 := 0;
        IF V_FECHA_PLAN_DESPLIEGUE IS NOT NULL THEN
            V_F1 := TO_CHAR(V_FECHA_PLAN_DESPLIEGUE,'YYYYMMDD');    
        END IF;
        
        V_F2 := 0;
        IF DATE_IN IS NOT NULL THEN
            V_F2 := TO_CHAR(DATE_IN,'YYYYMMDD');
        END IF;
        
        DBMS_OUTPUT.PUT_LINE('COMP: &'||V_F1||'&=&'||V_F2||'&');
        IF V_F1=V_F2 THEN 
            DBMS_OUTPUT.PUT_LINE('IGUAL');
            V_FPD := 0;
        ELSE
            DBMS_OUTPUT.PUT_LINE('DIFERENTE');
            V_FPD := 1;
        END IF;
         
        DBMS_OUTPUT.PUT_LINE('FPD RES: '||V_FPD);
        UPDATE TFS_INDICADORES_COMPLEMENTO 
            SET ID_USUARIO=8888, 
                FECHA_PLAN_DESPLIEGUE=DATE_IN, 
                ALERTA_FPD = (CASE WHEN ALERTA_FPD=1 THEN 1 ELSE V_FPD END) 
                WHERE ID_WI_LP=IDLP_IN;
    ELSE
        DBMS_OUTPUT.PUT_LINE('INSERT');
        IF DATE_IN IS NULL THEN
            DBMS_OUTPUT.PUT_LINE('VACIA');
            V_FPD := 0;
        ELSE
            DBMS_OUTPUT.PUT_LINE('CON FECHA');
            V_FPD := 1;
        END IF;
        INSERT INTO TFS_INDICADORES_COMPLEMENTO (ID_WI_LP,ID_USUARIO,FECHA_PLAN_DESPLIEGUE,ALERTA_FPD)VALUES(IDLP_IN,9999,DATE_IN,V_FPD);        
    END IF;
    DBMS_OUTPUT.PUT_LINE('INSERTAR CHECKLIST');
    
    SELECT COUNT(1) INTO V_NC FROM TFS_ACTIVIDADES_CHECKLIST WHERE ID_LP = IDLP_IN;
    DBMS_OUTPUT.PUT_LINE(V_NC);
    IF V_NC>0 THEN
        DBMS_OUTPUT.PUT_LINE('NO GENERA REGISTROS!');
    ELSE 
        DBMS_OUTPUT.PUT_LINE('TRAE ACTIVIDADES');
        INSERT INTO TFS_ACTIVIDADES_CHECKLIST (ID_ACTIVIDAD_CHECKLIST, FECHA_GRABACION, ID_TAREA_CHECKLIST,OBSERVACION,ESTADO,ID_LP,PESO,ID_COORDINACION)SELECT '' ID_ACTIVIDAD_CHECKLIST, SYSDATE, ID_TAREA_CHECKLIST,'' OBSERVACION, 'E' ESTADO, IDLP_IN ID_LP, 0 PESO, 1 ID_COORDINACION  FROM TFS_TAREAS_CHECKLISTS WHERE ESTADO=1 ORDER BY 1;
    END IF;
    
    DBMS_OUTPUT.PUT_LINE('PROCESO EXITOSO');
  EXCEPTION
  WHEN OTHERS THEN
    DBMS_OUTPUT.PUT_LINE('ERROR PROCESO: '||SQLCODE||' -ERROR- '||SQLERRM);
END;
