PROCEDURE SP_AUTO_SAVE_NOTIFICACION (
    V_S_RESPUESTA OUT VARCHAR2,
    V_S_CODIGO OUT NUMBER    
)
AS
    -- Declaración de variables
    P_CURSOR_EMPLEADOS SYS_REFCURSOR;   -- Declaración del cursor
    V_RESPUESTA VARCHAR2(100);          -- Variable para la respuesta de salida
    V_CODIGO NUMBER;
    V_CODEMPLEADO NUMBER;
    V_NOMBRE_EMPLEADO VARCHAR2(250);    -- Cambiado de VARCHAR a VARCHAR2
    V_NUMERODOCUMENTO VARCHAR2(100);     -- Cambiado de VARCHAR a VARCHAR2
    V_ID_TAREA NUMBER;
    V_COUNT_TAREA_CREA NUMBER;          -- Declaración de la variable de conteo
    V_ID_CREADO NUMBER;                 -- Declaración de la variable para el ID creado
    V_S_MENSAJE VARCHAR2(255);          -- Declaración de la variable para el mensaje
    V_CODEMPLEADOLIDER NUMBER;          -- Nueva variable para almacenar el CODEMPLEADOLIDER
BEGIN
    -- Llamada al procedimiento para obtener todos los empleados activos
    SP_GET_ALL_EMPLEADO_ACT_COMBO(  
        P_CURSOR_ALL_EMPLEADO_ACT_COMBO => P_CURSOR_EMPLEADOS,
        V_S_RESPUESTA => V_RESPUESTA,
        V_S_CODIGO => V_CODIGO
    );

    -- Procesar el cursor con los empleados
    LOOP
        FETCH P_CURSOR_EMPLEADOS INTO V_CODEMPLEADO, V_NOMBRE_EMPLEADO, V_NUMERODOCUMENTO;  
        EXIT WHEN P_CURSOR_EMPLEADOS%NOTFOUND;
        
        -- Verificar si hay tareas creadas para el empleado
        BEGIN
            SELECT MAX(ID) INTO V_COUNT_TAREA_CREA  
            FROM DF_TAREA 
            WHERE CODEMPLEADOASIGNADO = V_CODEMPLEADO 
              AND TRUNC(FECHAINIESTIMADA) BETWEEN TRUNC(CURRENT_DATE) AND TRUNC(CURRENT_DATE + 1)
              AND TRUNC(FECHAFINESTIMADA) BETWEEN TRUNC(CURRENT_DATE) AND TRUNC(CURRENT_DATE + 1)
              AND FECHAINIREAL IS NULL AND FECHAFINREAL IS NULL;
        EXCEPTION
            WHEN NO_DATA_FOUND THEN
            V_COUNT_TAREA_CREA := NULL;
        END;

        -- Si hay tareas, registrar la notificación
        IF V_COUNT_TAREA_CREA > 0 THEN
        
            DBMS_OUTPUT.PUT_LINE('INSERT EMPLEADO: ' || V_CODEMPLEADO);

            SP_SAVE_DF_NOTIFICACION(
                P_ID_NOTIFICACION => 0, 
                P_DESCRIP => 'Por favor, recuerde registrar horas en sus tareas del día',  
                P_URL => '/Empleado/Tareas',     
                P_ESTADO => 'NO LEIDO',         
                P_CODEMPLEADO => V_CODEMPLEADO,  
                V_S_RESPUESTA => V_S_RESPUESTA,  
                V_ID_CREADO => V_ID_CREADO,
                V_S_CODIGO => V_S_CODIGO,
                V_S_MENSAJE => V_S_MENSAJE
            );
        END IF;

        -- Si no hay tareas, enviar notificación al líder
        IF V_COUNT_TAREA_CREA IS NULL THEN
            -- Obtener el CODEMPLEADOLIDER para el empleado actual
            SELECT CODEMPLEADOLIDER INTO V_CODEMPLEADOLIDER
            FROM DF_EMPLEADO_VU 
            WHERE CODEMPLEADO = V_CODEMPLEADO
            AND ROWNUM = 1;
            
            -- Registrar la notificación para el líder
            /*SP_SAVE_DF_NOTIFICACION(
                P_ID_NOTIFICACION => 0, 
                P_DESCRIP => 'Por favor, recuerde crear tareas para el día: ' || CURRENT_DATE || ' a el empleado ' || V_NOMBRE_EMPLEADO,  
                P_URL => '/Seguimiento/Gantt',     
                P_ESTADO => 'NO LEIDO',         
                P_CODEMPLEADO => V_CODEMPLEADOLIDER,  -- Se notifica al líder
                V_S_RESPUESTA => V_S_RESPUESTA,  
                V_ID_CREADO => V_ID_CREADO,
                V_S_CODIGO => V_S_CODIGO,
                V_S_MENSAJE => V_S_MENSAJE
            );*/
        END IF;
        
    END LOOP;
    
    DBMS_OUTPUT.PUT_LINE('FINAL DEL LOOP: ' || V_CODEMPLEADO);
    
    -- Cerrar el cursor después de procesarlo
    CLOSE P_CURSOR_EMPLEADOS;

    -- Asignar respuestas de éxito
    V_S_RESPUESTA := 'Proceso completado correctamente';
    V_S_CODIGO := 200;

    -- Realizar un COMMIT si es necesario
    COMMIT;

EXCEPTION
    WHEN OTHERS THEN
        -- Manejo de errores
        V_S_CODIGO    := 500;
        V_S_RESPUESTA := 'Error en el procedimiento: ' || SQLERRM;
        ROLLBACK;
END;
