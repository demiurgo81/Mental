TEXT WITH  -- Costo total por estimación (Sin cambios) ESTIMACION_COSTO AS (     SELECT CODESTIMACION, SUM(COSTO) AS ESCOSTO     FROM DF_ESTIMACIONES_ALL_V     GROUP BY CODESTIMACION ), -- Cantidad de tareas por estimación (Sin cambios) ESTIMACION_CANTIDAD AS (     SELECT CODESTIMACION, COUNT(*) AS CNT     FROM DF_ESTIMACION_TAREA     GROUP BY CODESTIMACION ), -- Cantidad de estimaciones a las que pertenece una tarea (Sin cambios) ENTREGABLE_CANTIDAD AS (     SELECT CODTAREA, COUNT(*) AS CNT     FROM DF_ESTIMACION_TAREA     GROUP BY CODTAREA ), -- Costo real total por tarea (Modificado para obtener el costo total de la tarea) TAREA_COSTO_TOTAL AS (     SELECT CODTAREAPADRE, SUM(COSTO_REAL) AS ENCOSTO_TOTAL     FROM DF_ENTREGABLES_YEAR_V     GROUP BY CODTAREAPADRE ), -- Nueva CTE para obtener los proveedores por tarea y contar cuántos son TAREA_PROVEEDORES AS (     SELECT DISTINCT         V.CODTAREAPADRE,         E.CODPROVEEDOR,         COUNT(DISTINCT E.CODPROVEEDOR) OVER (PARTITION BY V.CODTAREAPADRE) AS NUM_PROVEEDORES     FROM         DF_ENTREGABLES_YEAR_V V     JOIN         DF_EMPLEADO E ON V.CODEMPLEADO = E.ID ) -- Unión final con la nueva lógica de cálculo SELECT      DET.CODTAREA,     DET.CODESTIMACION,     pe.nombre_estimacion,     df_replace_clasifica_fn(df_cleartextfn(pe.nombre_estimacion)) est_fab,     p.nombreproyecto proyecto,     p.gerente,     TP.CODPROVEEDOR, -- Campo adicionado     df_take_description_fn('df_proveedor', TP.CODPROVEEDOR) proveedor,     CASE pe.id_estado WHEN 1 THEN 'NO APROBADO' WHEN 2 THEN 'APROBADO' WHEN 3 THEN 'FINALIZADO' ELSE 'INVALIDO' END est_estado,     EC.ESCOSTO,     -- El costo estimado parcial ahora se divide por el número de proveedores de la tarea     ROUND((EC.ESCOSTO / ECNT.CNT) / TP.NUM_PROVEEDORES) AS ESCOSTOPART,     df_take_description_fn('df_tarea', DET.CODTAREA) AS entregable,     df_replace_clasifica_fn(df_cleartextfn(df_take_description_fn('df_tarea', DET.CODTAREA))) AS ent_fab,     TCT.ENCOSTO_TOTAL AS ENCOSTO,     -- El costo ejecutado parcial ahora se divide por el número de proveedores de la tarea     ROUND((TCT.ENCOSTO_TOTAL / ENCT.CNT) / TP.NUM_PROVEEDORES) AS ENCOSTOPART,     -- El ejecutado se calcula con los nuevos valores parciales     (ROUND((EC.ESCOSTO / ECNT.CNT) / TP.NUM_PROVEEDORES)) - (ROUND((TCT.ENCOSTO_TOTAL / ENCT.CNT) / TP.NUM_PROVEEDORES)) AS EJECUTADO FROM      DF_ESTIMACION_TAREA DET JOIN      ESTIMACION_COSTO EC ON DET.CODESTIMACION = EC.CODESTIMACION JOIN      ESTIMACION_CANTIDAD ECNT ON DET.CODESTIMACION = ECNT.CODESTIMACION JOIN      ENTREGABLE_CANTIDAD ENCT ON DET.CODTAREA = ENCT.CODTAREA JOIN      TAREA_COSTO_TOTAL TCT ON DET.CODTAREA = TCT.CODTAREAPADRE JOIN      TAREA_PROVEEDORES TP ON DET.CODTAREA = TP.CODTAREAPADRE -- Este JOIN expande las filas por proveedor LEFT JOIN      df_proyecto_estimacion pe ON pe.id = det.codestimacion LEFT JOIN      df_proyecto_presupuesto_vu p ON p.idproyecto = pe.id_proyecto
