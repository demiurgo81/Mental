FUNCTION df_replace_clasifica_fn (
    ORIGINAL IN VARCHAR2
) RETURN VARCHAR2 DETERMINISTIC
AS
    -- Variable para almacenar la cadena mientras se procesa.
    -- Se inicializa con el tamaño máximo de VARCHAR2 en PL/SQL por si las
    -- operaciones de reemplazo expanden significativamente la cadena.
    v_processed_string VARCHAR2(32767);
BEGIN
    -- Si la cadena original es NULL, no hay nada que procesar.
    IF ORIGINAL IS NULL THEN
        RETURN NULL;
    END IF;

    -- Inicializar la cadena de trabajo con el valor original.
    v_processed_string := ORIGINAL;

    -- Recorrer los registros de df_clasifica que cumplen la condición,
    -- ordenados por prioridad, para aplicar los reemplazos secuencialmente.
    FOR rec IN (
        SELECT BUSCA, REEMPLAZA
        FROM df_clasifica
        WHERE OPERACION = 'REGEXP'
        ORDER BY PRIORIDAD ASC
    ) LOOP
        -- Asegurarse de que el patrón de búsqueda (BUSCA) no sea NULL.
        -- Un patrón NULL en REGEXP_REPLACE puede causar errores o comportamientos no deseados.
        IF rec.BUSCA IS NOT NULL THEN
            -- Aplicar el reemplazo.
            -- rec.REEMPLAZA puede ser NULL, lo que significa que la subcadena que coincida
            -- con rec.BUSCA será eliminada.
            v_processed_string := REGEXP_REPLACE(
                v_processed_string, -- La cadena actual, resultado de iteraciones previas
                rec.BUSCA,          -- El patrón de expresión regular a buscar
                rec.REEMPLAZA       -- La cadena de reemplazo
                -- Se utiliza la versión básica de REGEXP_REPLACE. Si se necesitaran
                -- modificadores como 'i' (insensible a mayúsculas/minúsculas), 'c' (sensible),
                -- 'n' (permitir que '.' coincida con nueva línea), 'm' (modo multilínea),
                -- estos deberían incluirse en el propio patrón en rec.BUSCA (ej. '(?i)patron')
                -- o usando la versión de 6 argumentos de REGEXP_REPLACE si se requiere
                -- un control más granular a través de match_parameter.
            );
        END IF;
    END LOOP;

    -- Devolver la cadena después de todos los reemplazos.
    RETURN TRIM(v_processed_string);

EXCEPTION
    WHEN OTHERS THEN
        -- En caso de un error (ej. una expresión regular inválida en rec.BUSCA),
        -- se retorna NULL. Considera un logging del error para depuración.
        -- DBMS_OUTPUT.PUT_LINE('Error en df_replace_clasifica_fn: ' || SQLERRM);
        RETURN NULL;
END df_replace_clasifica_fn;
