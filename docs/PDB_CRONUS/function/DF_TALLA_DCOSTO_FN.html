FUNCTION df_talla_dcosto_fn (
    p_costo IN NUMBER,
    p_tolerancia IN NUMBER DEFAULT 1,
    p_tecnologia IN VARCHAR2 DEFAULT 'MICROSERVICIOS'
) RETURN VARCHAR2
IS
    v_talla VARCHAR2(3);
    v_costo_min NUMBER;
    v_costo_max NUMBER;

BEGIN
    -- Validar que el costo sea un valor válido
    IF p_costo IS NULL OR p_costo <= 0 THEN
        RETURN NULL;
    END IF;

    -- Lógica según el valor del parámetro tolerancia
    IF p_tolerancia = 1 THEN
        -- Usar campo TOLERANCIA de la tabla
        SELECT t.TALLA
        INTO v_talla
        FROM (
            SELECT TALLA, COSTO,
                   ROW_NUMBER() OVER (ORDER BY COSTO ASC) as rn
            FROM DF_TALLAS
            WHERE TECNOLOGIA = p_tecnologia
              AND (
                  -- Condición 1: (costo*(1-TOLERANCIA)) BETWEEN RANGO_MIN AND RANGO_MAX
                  (p_costo * (1 - TOLERANCIA)) BETWEEN RANGO_MIN AND RANGO_MAX
                  OR
                  -- Condición 2: (costo*(1+TOLERANCIA)) BETWEEN RANGO_MIN AND RANGO_MAX
                  (p_costo * (1 + TOLERANCIA)) BETWEEN RANGO_MIN AND RANGO_MAX
              )
        ) t
        WHERE t.rn = 1;

    ELSE
        -- Omitir tolerancia (tolerancia = 0)
        SELECT t.TALLA
        INTO v_talla
        FROM (
            SELECT TALLA, COSTO,
                   ROW_NUMBER() OVER (ORDER BY COSTO ASC) as rn
            FROM DF_TALLAS
            WHERE TECNOLOGIA = p_tecnologia
              AND p_costo BETWEEN RANGO_MIN AND RANGO_MAX
        ) t
        WHERE t.rn = 1;

    END IF;

    RETURN v_talla;

EXCEPTION
    WHEN NO_DATA_FOUND THEN
        -- No se encontró ningún registro que cumpla las condiciones
        RETURN NULL;
    WHEN TOO_MANY_ROWS THEN
        -- Esto no debería ocurrir por el ROW_NUMBER, pero por seguridad
        RETURN NULL;
    WHEN OTHERS THEN
        -- Cualquier otro error
        RETURN NULL;

END df_talla_dcosto_fn;
