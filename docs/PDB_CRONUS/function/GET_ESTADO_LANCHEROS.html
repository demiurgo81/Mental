FUNCTION "GET_ESTADO_LANCHEROS" (
  ESTADO_BRIEF IN VARCHAR2 
, ESTADO_LINEA IN VARCHAR2 
, CLASIFICACION IN VARCHAR2 
, COSTO_DESARROLLO IN VARCHAR2 
, TIEMPO_DIAS IN VARCHAR2 
, HORAS_DESARROLLO IN VARCHAR2 
, FECHA_PLANEADA_DESPLIEGUE IN VARCHAR2 
, FECHA_DESPLIEGUE IN VARCHAR2 
, BLOQUEADO IN VARCHAR2 
) RETURN VARCHAR2 IS
    ESTADO_LANCHEROS VARCHAR(200) :='';
BEGIN
    ESTADO_LANCHEROS :=ESTADO_BRIEF||'|'||COSTO_DESARROLLO||'|'||TIEMPO_DIAS||'|'||HORAS_DESARROLLO;
    IF ( UPPER(ESTADO_BRIEF)=UPPER('Cerrado') OR UPPER(ESTADO_BRIEF)=UPPER('Cancelado') ) THEN 
        ESTADO_LANCHEROS:='Cancelado';   
    ELSIF ( UPPER(ESTADO_LINEA)=UPPER('Produccion') ) THEN 
        ESTADO_LANCHEROS:='Produccion';      
    ELSIF ( UPPER(ESTADO_BRIEF)=UPPER('Creado') OR UPPER(ESTADO_BRIEF)=UPPER('Radicado') OR UPPER(CLASIFICACION)=UPPER('Iniciativa') ) THEN 
        ESTADO_LANCHEROS:='Iniciativa';
    ELSIF ( UPPER(ESTADO_BRIEF)=UPPER('CO-Creacion Funcional') OR UPPER(ESTADO_BRIEF)=UPPER('Factible')  ) THEN
        ESTADO_LANCHEROS:='Co-creacion de alcance';
    ELSIF ( UPPER(ESTADO_BRIEF)=UPPER('CO-Creacion Tecnica')  
        --AND COSTO_DESARROLLO IS NULL  AND TIEMPO_DIAS IS NULL  AND HORAS_DESARROLLO IS NULL  
        ) THEN
        ESTADO_LANCHEROS:='Sin estimar'; 
    ELSIF ( UPPER(ESTADO_BRIEF)=UPPER('Estimado') AND  ( COSTO_DESARROLLO IS NULL OR TIEMPO_DIAS IS NULL OR HORAS_DESARROLLO IS NULL )  )THEN
        ESTADO_LANCHEROS:='Sin estimar';
    ELSIF ( ( UPPER(ESTADO_BRIEF)=UPPER('Estimado') OR UPPER(ESTADO_BRIEF)=UPPER('FSP Cerrado') )  AND COSTO_DESARROLLO IS NOT NULL AND TIEMPO_DIAS IS NOT NULL AND HORAS_DESARROLLO IS NOT NULL  ) THEN 
        ESTADO_LANCHEROS:='Estimado';
    ELSIF ( UPPER(ESTADO_BRIEF)=UPPER('Aprobado Junta') AND FECHA_PLANEADA_DESPLIEGUE IS NULL AND FECHA_DESPLIEGUE IS NULL) THEN 
        ESTADO_LANCHEROS:='Backlog';
    ELSIF ( UPPER(ESTADO_BRIEF)=UPPER('Aprobado Junta') AND UPPER(ESTADO_LINEA)=UPPER('Creado')  )THEN
        ESTADO_LANCHEROS:='Vacio';
    ELSIF ( UPPER(ESTADO_BRIEF)=UPPER('Aprobado Junta') AND UPPER(ESTADO_LINEA)=UPPER('Aprobado') ) THEN 
        ESTADO_LANCHEROS:='Alarma A/D';
    ELSIF ( UPPER(ESTADO_BRIEF)=UPPER('Aprobado Junta') AND UPPER(ESTADO_LINEA)=UPPER('Analisis') 
    AND UPPER(BLOQUEADO)='SI' AND FECHA_PLANEADA_DESPLIEGUE IS NOT NULL)THEN
        ESTADO_LANCHEROS:='Alarma A/D';
    ELSIF ( UPPER(ESTADO_BRIEF)=UPPER('Aprobado Junta') AND UPPER(ESTADO_LINEA)=UPPER('Analisis') AND UPPER(BLOQUEADO)='NO' AND FECHA_PLANEADA_DESPLIEGUE IS NOT NULL) THEN 
        ESTADO_LANCHEROS:='A/D';
    ELSIF ( UPPER(ESTADO_BRIEF)=UPPER('Aprobado Junta') AND UPPER(ESTADO_LINEA)=UPPER('Construccion') AND UPPER(BLOQUEADO)='SI' AND FECHA_PLANEADA_DESPLIEGUE IS NOT NULL) THEN
        ESTADO_LANCHEROS:='Alarma Desarrollo';
    ELSIF ( UPPER(ESTADO_BRIEF)=UPPER('Aprobado Junta') AND UPPER(ESTADO_LINEA)=UPPER('Construccion') AND UPPER(BLOQUEADO)='NO' AND FECHA_PLANEADA_DESPLIEGUE IS NOT NULL)THEN
        ESTADO_LANCHEROS:='Desarrollo';
    ELSIF ( UPPER(ESTADO_BRIEF)=UPPER('Aprobado Junta') AND UPPER(ESTADO_LINEA)=UPPER('Pruebas') AND UPPER(BLOQUEADO)='SI' AND FECHA_PLANEADA_DESPLIEGUE IS NOT NULL ) THEN
        ESTADO_LANCHEROS:='Alarma Pruebas';
    ELSIF ( UPPER(ESTADO_BRIEF)=UPPER('Aprobado Junta') AND UPPER(ESTADO_LINEA)=UPPER('Pruebas') AND UPPER(BLOQUEADO)='NO' AND FECHA_PLANEADA_DESPLIEGUE IS NOT NULL ) THEN
        ESTADO_LANCHEROS:='Pruebas';
    ELSIF ( UPPER(ESTADO_BRIEF)=UPPER('Aprobado Junta') AND UPPER(ESTADO_LINEA)=UPPER('Despliegue') ) THEN
        ESTADO_LANCHEROS:='Despliegue';
    ELSIF ( UPPER(ESTADO_LINEA)=UPPER('Produccion') ) THEN 
        ESTADO_LANCHEROS:='Produccion';
    END IF;

    RETURN ESTADO_LANCHEROS;

END GET_ESTADO_LANCHEROS;
