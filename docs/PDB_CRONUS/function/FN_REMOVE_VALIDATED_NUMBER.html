FUNCTION "FN_REMOVE_VALIDATED_NUMBER" (
    p_input_string IN VARCHAR2,
    p_check_number IN NUMBER
) RETURN VARCHAR2 IS
    v_temp_string VARCHAR2(100) := p_input_string;
    v_remaining_numbers VARCHAR2(100);
    v_result_string VARCHAR2(100);
BEGIN
    -- Inicializar la cadena de números restantes
    v_remaining_numbers := v_temp_string;

    -- Verificar si el número está presente en la cadena
    IF INSTR(',' || v_temp_string || ',', ',' || TO_CHAR(p_check_number) || ',') > 0 THEN
        -- Eliminar el número validado de la cadena
        v_remaining_numbers := REPLACE(',' || v_temp_string || ',', ',' || TO_CHAR(p_check_number) || ',', ',');
        v_remaining_numbers := TRIM(BOTH ',' FROM v_remaining_numbers);

        -- Si se eliminó el número validado y la cadena resultante es vacía, devolver una cadena vacía
        IF LENGTH(v_remaining_numbers) = 0 THEN
            v_result_string := '';
        ELSE
            v_result_string := v_remaining_numbers;
        END IF;
    ELSE
        -- Si el número no está presente en la cadena, devolver la cadena original
        v_result_string := v_temp_string;
    END IF;

    RETURN v_result_string;
END FN_REMOVE_VALIDATED_NUMBER;
