TEXT SELECT   au.id  as ID,  em.nombre AS NOMBRE_EMPLEADO ,CASE WHEN AU.STATE = 'BAD_REQUEST' THEN  'FALLO' ELSE 'OK'  END AS ESTADO,          au.ID_AZURE AS ID_TAREA_AZURE, AU.OBSERVATIONS AS OBSERVACIONES ,to_char( au.UPDATE_DATE, 'dd/mm/yyyy HH24:MI:SS') as fecha_actualizacion FROM  DF_AUDIT_TASKVALUESTREAM au  JOIN  DF_TASKVALUESTREAM_FIELDS_V ta on au.ID = ta.ID JOIN DF_EMPLEADO_VU em ON au.id_empleadoassigned = em.codempleado JOIN DF_RANGO r ON 1=1 WHERE ta.FECHAINICIO  between to_char(r.FINI, 'dd/mm/yyyy') and to_char(r.FFIN, 'dd/mm/yyyy')    AND SUBSTR(em.JERARQUIA, 0, 20) = LPAD(r.GER,10,0)||LPAD(r.COORD,10,0) ORDER BY  em.NOMBRE, au.UPDATE_DATE  desc
