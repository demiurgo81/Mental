PROCEDURE SP_REPORTE_CHART (
    P_JERARQUIA IN VARCHAR2,
    P_PROYECTO IN NUMBER,
    P_VIGENCIA IN VARCHAR2,
    P_CURSOR_COSTO_FABRICA_X_COORD OUT SYS_REFCURSOR,
    P_CURSOR_PRESUPUESTO_DISPONIBLE OUT SYS_REFCURSOR,
    P_CURSOR_COSTO_TOTAL_X_ESTIMA OUT SYS_REFCURSOR,
    P_CURSOR_HORAS_X_ENTREGABLE OUT SYS_REFCURSOR,
    P_CURSOR_COSTO_TOTAL_PROY OUT SYS_REFCURSOR,
    V_S_RESPUESTA OUT VARCHAR2,
    V_S_CODIGO OUT NUMBER
)
AS
BEGIN
    BEGIN
    
    OPEN P_CURSOR_COSTO_TOTAL_X_ESTIMA FOR
        SELECT P.NOMBRE AS LABEL,  NVL(SUM(E.COSTO),0) AS VALOR
            FROM DF_ESTIMACION E
            INNER JOIN DF_PROYECTO P ON P.ID = E.CODPROYECTO
            WHERE E.CODPROYECTO = P_PROYECTO
                AND E.ID_PROYECTO_ESTIMACION = (
                    SELECT PE.ID 
                    FROM DF_PROYECTO_ESTIMACION PE 
                    WHERE PE.ID_ESTADO = 2 
                      AND PE.ID_PROYECTO = P_PROYECTO
                    FETCH FIRST 1 ROWS ONLY
                )GROUP BY P.NOMBRE;
    

    OPEN P_CURSOR_COSTO_FABRICA_X_COORD FOR
        SELECT 
            P.NOMBRE AS LABEL, 
            NVL(SUM(CE.COSTO_PUNTO * T.TIEMPOREAL),0) AS VALOR 
        FROM DF_TAREA T
        INNER JOIN DF_COSTO_EMPLEADO CE ON T.CODEMPLEADOASIGNADO = CE.CODEMPLEADO
        INNER JOIN DF_PROYECTO P ON P.ID = T.CODPROYECTO
        WHERE 
            T.CODPROYECTO = P_PROYECTO 
            AND T.FECHAINIREAL >= 
                CASE 
                    -- Si es enero y dentro de los primeros 10 días, incluir también los últimos 10 días del año pasado
                    WHEN EXTRACT(MONTH FROM SYSDATE) = 1 AND EXTRACT(DAY FROM SYSDATE) <= 10 
                    THEN TO_DATE('22/12/' || (EXTRACT(YEAR FROM SYSDATE) - 1), 'DD/MM/YYYY')
                    ELSE TO_DATE('01/01/' || EXTRACT(YEAR FROM SYSDATE), 'DD/MM/YYYY')
                END
            AND T.FECHAINIREAL <= TO_DATE('31/12/' || EXTRACT(YEAR FROM SYSDATE), 'DD/MM/YYYY') 
            AND T.ESTADO = 'A' 
            AND CE.HASTA IS NULL
        GROUP BY P.NOMBRE;

    OPEN P_CURSOR_PRESUPUESTO_DISPONIBLE FOR
        SELECT 
            P.NOMBRE AS LABEL,
            (
                SELECT NVL(SUM(COSTO),0) AS PRESUPUESTO 
                FROM DF_ESTIMACION
                WHERE CODPROYECTO = P_PROYECTO
                  AND ID_PROYECTO_ESTIMACION IN (
                      SELECT PE.ID 
                      FROM DF_PROYECTO_ESTIMACION PE 
                )
            ) - 
            (
                SELECT NVL(SUM(CE.COSTO_PUNTO * T.TIEMPOREAL),0) AS VALOR 
                FROM DF_TAREA T
                INNER JOIN DF_COSTO_EMPLEADO CE 
                    ON T.CODEMPLEADOASIGNADO = CE.CODEMPLEADO
                LEFT JOIN DF_PROYECTO P 
                    ON P.ID = T.CODPROYECTO
                WHERE T.CODPROYECTO = P_PROYECTO 
                  AND T.FECHAINIREAL BETWEEN 
                      CASE 
                          -- Si estamos en los primeros 10 días de enero, tomar TODO el año anterior
                          WHEN EXTRACT(MONTH FROM SYSDATE) = 1 AND EXTRACT(DAY FROM SYSDATE) <= 10 
                          THEN TO_DATE('01/01/' || (EXTRACT(YEAR FROM SYSDATE) - 1), 'DD/MM/YYYY')
                          -- Si es después del 10 de enero, tomar TODO el año actual
                          ELSE TO_DATE('01/01/' || EXTRACT(YEAR FROM SYSDATE), 'DD/MM/YYYY')
                      END 
                  AND 
                      CASE 
                          -- Si estamos en los primeros 10 días de enero, tomar TODO el año anterior
                          WHEN EXTRACT(MONTH FROM SYSDATE) = 1 AND EXTRACT(DAY FROM SYSDATE) <= 10 
                          THEN TO_DATE('31/12/' || (EXTRACT(YEAR FROM SYSDATE) - 1), 'DD/MM/YYYY')
                          -- Si es después del 10 de enero, tomar TODO el año actual
                          ELSE TO_DATE('31/12/' || EXTRACT(YEAR FROM SYSDATE), 'DD/MM/YYYY')
                      END
                  AND T.FECHAFINREAL BETWEEN 
                      CASE 
                          -- Si estamos en los primeros 10 días de enero, tomar TODO el año anterior
                          WHEN EXTRACT(MONTH FROM SYSDATE) = 1 AND EXTRACT(DAY FROM SYSDATE) <= 10 
                          THEN TO_DATE('01/01/' || (EXTRACT(YEAR FROM SYSDATE) - 1), 'DD/MM/YYYY')
                          -- Si es después del 10 de enero, tomar TODO el año actual
                          ELSE TO_DATE('01/01/' || EXTRACT(YEAR FROM SYSDATE), 'DD/MM/YYYY')
                      END 
                  AND 
                      CASE 
                          -- Si estamos en los primeros 10 días de enero, tomar TODO el año anterior
                          WHEN EXTRACT(MONTH FROM SYSDATE) = 1 AND EXTRACT(DAY FROM SYSDATE) <= 10 
                          THEN TO_DATE('31/12/' || (EXTRACT(YEAR FROM SYSDATE) - 1), 'DD/MM/YYYY')
                          -- Si es después del 10 de enero, tomar TODO el año actual
                          ELSE TO_DATE('31/12/' || EXTRACT(YEAR FROM SYSDATE), 'DD/MM/YYYY')
                      END
                  AND T.ESTADO = 'A' 
                  AND CE.HASTA IS NULL
            ) AS VALOR
            FROM DF_PROYECTO P
        WHERE P.ID = P_PROYECTO;


        OPEN P_CURSOR_HORAS_X_ENTREGABLE FOR
            ----HORAS X ENTREGABLE SE CAMBIO POR PRESUPUESTO TOTAL
            SELECT NVL(SUM(TIEMPOREAL),0) AS VALOR, T.NOMBRE AS LABEL FROM DF_TAREA T
                INNER JOIN DF_COSTO_EMPLEADO CE ON T.CODEMPLEADOASIGNADO = CE.CODEMPLEADO
                    WHERE T.CODPROYECTO = P_PROYECTO and T.FECHAINIREAL BETWEEN TO_DATE ('01/01/25 01:00', 'dd/mm/yy HH12:MI am')
                    AND TO_DATE ('31/12/25 11:59', 'dd/mm/yy HH12:MI pm') and T.ESTADO = 'A' AND CE.HASTA IS NULL
                    GROUP BY T.NOMBRE;
                    
        OPEN P_CURSOR_COSTO_TOTAL_PROY FOR
        ---DETALLE COSTO TOTAL PROYECTO
            SELECT NVL(SUM(T.TIEMPOREAL * CE.COSTO_PUNTO),0) AS VALOR, P.NOMCOMPLETO AS LABEL FROM DF_TAREA T
                INNER JOIN DF_COSTO_EMPLEADO CE ON T.CODEMPLEADOASIGNADO = CE.CODEMPLEADO
                INNER JOIN DF_EMPLEADO E ON CE.CODEMPLEADO = E.ID
                INNER JOIN DF_PERSONA P ON E.CODPERSONA = P.ID
                    WHERE T.CODPROYECTO = P_PROYECTO and T.FECHAINIREAL BETWEEN TO_DATE ('01/01/25 01:00', 'dd/mm/yy HH12:MI am')
                AND TO_DATE ('31/12/25 11:59', 'dd/mm/yy HH12:MI pm') and T.ESTADO = 'A' AND CE.HASTA IS NULL
                GROUP BY P.NOMCOMPLETO;
        

      V_S_CODIGO := 200;
      V_S_RESPUESTA := 'Retorno Correcto';
    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            ROLLBACK;
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'No Fue posible traer la data';
            RAISE;
    END;
END SP_REPORTE_CHART;
