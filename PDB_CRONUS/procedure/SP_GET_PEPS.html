PROCEDURE SP_GET_PEPS (
    P_ID_PEP IN NUMBER,
    P_ESTADO IN NUMBER,
    P_CURSOR_GET_PEPS OUT SYS_REFCURSOR,
    V_S_RESPUESTA OUT VARCHAR2,
    V_S_CODIGO OUT NUMBER
)
AS
BEGIN
    BEGIN
        -- Abrir cursor para obtener los líderes según el ID o todos si el ID no está definido
        OPEN P_CURSOR_GET_PEPS FOR
                SELECT PE.ID, PE.PEP, PE.FECHACREACION, PE.FECHAMODIFICACION, P.NOMCOMPLETO AS GERENTE, 
                    PE.CODGERENTE, PE.ESTADO, PE.URL_SOPORTE, PE.CODEMPLEADOCREO, PE.CODEMPLEADOMODIFICA, PE.PRESUPUESTO
                FROM DF_PEP PE
                LEFT JOIN DF_EMPLEADO E ON E.ID = PE.CODGERENTE
                LEFT JOIN DF_PERSONA P ON P.ID = E.CODPERSONA
                WHERE (P_ID_PEP = 0 OR PE.ID = P_ID_PEP)
                    AND (P_ESTADO = 0 OR PE.ESTADO = P_ESTADO)
                    AND PE.ESTADO IS NOT NULL AND PE.ESTADO <> 4
                ORDER BY PE.ID, PE.FECHAMODIFICACION, PE.FECHACREACION  DESC;

        -- Respuesta de éxito
        V_S_RESPUESTA := 'Datos retornados exitosamente';
        V_S_CODIGO := 200;

    EXCEPTION
        WHEN OTHERS THEN
            -- Manejo de errores
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'Error: ' || SQLERRM;
    END;
END SP_GET_PEPS;
