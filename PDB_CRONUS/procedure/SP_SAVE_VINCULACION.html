PROCEDURE SP_SAVE_VINCULACION (
    P_ID IN NUMBER,
    P_CODPERSONA IN NUMBER,
    P_CODPERFIL IN NUMBER,
    P_CODPERFILTIPO IN NUMBER,
    P_CODLINEAPRODUCTO IN NUMBER,
    P_CODPROVEEDOR IN NUMBER,
    P_CODPERFILNIVEL IN NUMBER,
    P_USUARIOCLARO IN VARCHAR2,
    P_FECHAINICIO IN DATE,
    P_FECHAFIN IN DATE DEFAULT NULL,
    P_CODESTRUCTURAORGANIZACIONAL IN NUMBER,
    P_CODMODALIDAD IN NUMBER,
    P_CELULA IN VARCHAR2,
    P_CHORA IN NUMBER,
    P_CMES IN NUMBER,
    P_PASS IN VARCHAR2,
    P_ROLUSER IN NUMBER,
    V_S_RESPUESTA OUT VARCHAR2,
    V_ID_CREADO OUT NUMBER,
    V_S_CODIGO OUT NUMBER
)
AS
    V_COUNT NUMBER;
    V_USUARIO VARCHAR2(20);
    V_RESPUESTA VARCHAR2(4000);
    V_CODIGO NUMBER;
    V_ID_EMPLEADO_CREADO NUMBER;
    V_NOMBRE_EMPLEADO VARCHAR(100);
    V_ID_USUARIO_CREADO NUMBER;
    V_ESTADO_USUARIO VARCHAR(1) := 'A';
BEGIN
    BEGIN
        -- Verificar si el registro ya existe en la tabla
        --SELECT COUNT(*) INTO V_COUNT FROM DF_EMPLEADO WHERE ID = P_ID;
        
        IF P_CODPERSONA IS NULL OR P_CODPERSONA = 0 THEN
            V_S_RESPUESTA := 'El campo P_CODPERSONA claro no puede estar vacío o ser 0';
            V_S_CODIGO := 400;
            RETURN;
        END IF;
        
        IF P_USUARIOCLARO IS NULL OR TRIM(P_USUARIOCLARO) = '' THEN
            V_S_RESPUESTA := 'El campo Usuario claro no puede estar vacío';
            V_S_CODIGO := 400;
            RETURN;
        END IF;
        
        SP_SAVE_DF_EMPLEADO(P_ID, P_CODPERSONA, P_CODPERFIL, P_CODPERFILTIPO, P_CODLINEAPRODUCTO, P_CODPROVEEDOR, P_CODPERFILNIVEL,
            P_USUARIOCLARO, P_FECHAINICIO, P_FECHAFIN, P_CODESTRUCTURAORGANIZACIONAL, P_CODMODALIDAD, P_CELULA, P_CHORA, P_CMES,
            V_RESPUESTA, V_ID_EMPLEADO_CREADO, V_CODIGO);
        
        IF V_CODIGO = 400 THEN 
            V_S_RESPUESTA := V_RESPUESTA;
            V_S_CODIGO := V_CODIGO;
            ROLLBACK;
            RETURN;
        END IF;
        
        SELECT NOMBRE1 || ' ' || APELLIDO1 INTO V_NOMBRE_EMPLEADO FROM DF_PERSONA WHERE ID=P_CODPERSONA;
        
        IF P_FECHAFIN IS NOT NULL THEN 
            V_ESTADO_USUARIO := 'I';
        END IF;
        
        SP_SAVE_DF_USUARIO(0, P_ROLUSER, P_CODPERSONA, P_USUARIOCLARO, V_NOMBRE_EMPLEADO, P_PASS, V_ESTADO_USUARIO, '', '', 
            V_RESPUESTA, V_ID_USUARIO_CREADO, V_CODIGO);
        
        IF V_CODIGO = 400 THEN 
            V_S_RESPUESTA := V_RESPUESTA;
            V_S_CODIGO := V_CODIGO;
            ROLLBACK;
            RETURN;
        ELSE
            V_S_RESPUESTA := 'Empleado Vinculado Exitosamente';
            V_S_CODIGO := 200;
            V_ID_CREADO := V_ID_EMPLEADO_CREADO;
        END IF;
        
        COMMIT;
    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'No existe estructura';
            ROLLBACK;
        WHEN OTHERS THEN
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'Error: ' || SQLERRM;
            ROLLBACK;
    END;
END SP_SAVE_VINCULACION;
