PROCEDURE SP_REPORT_EXCEL_ESTIMA (
    P_ID_PROY_ESTIMA IN NUMBER,
    P_CURSOR_REPORT_EXCEL_ESTIMA OUT SYS_REFCURSOR,
    V_S_RESPUESTA OUT VARCHAR2,
    V_S_CODIGO OUT NUMBER
)
AS
BEGIN
    BEGIN
        OPEN P_CURSOR_REPORT_EXCEL_ESTIMA FOR
        SELECT ID, ENTREGABLE, PERFIL, TIPO_PREREQUISITO, PREREQUISITO, DIAS_HABILES, COSTO, TIPO
        FROM (
            -- 1. ENCABEZADO
            SELECT 
                '00' AS ORDEN,
                'No' AS ID, 
                'Entregable Servicio ' || NOMBRE_ESTIMACION AS ENTREGABLE, 
                'Perfil' AS PERFIL, 
                'Tipo de Pre-Requisito' AS TIPO_PREREQUISITO, 
                'Pre-Requisito' AS PREREQUISITO, 
                'Dias Habiles' AS DIAS_HABILES, 
                'Costo $' AS COSTO,
                'Tipo' AS TIPO
            FROM DF_PROYECTO_ESTIMACION 
            WHERE ID = P_ID_PROY_ESTIMA
        
            UNION ALL
        
            -- 2. DETALLE NORMAL
            SELECT 
                LPAD(ROWNUM, 2, '0') AS ORDEN,
                TO_CHAR(ROWNUM) AS ID,
                ENTREGABLE,
                PERFIL,
                CASE 
                    WHEN ENTREGABLE LIKE 'Desarrollo%' THEN 'Documento' 
                    ELSE TIPO_PREREQUISITO 
                END AS TIPO_PREREQUISITO,
                PREREQUISITO,
                CASE 
                    WHEN ENTREGABLE = 'Total' THEN TO_CHAR(DIAS_HABILES, '999') 
                    ELSE TO_CHAR(DIAS_HABILES, '90.99') 
                END AS DIAS_HABILES,
                TO_CHAR(COSTO) AS COSTO,
                CASE 
                    WHEN ENTREGABLE IN ('Análisis y Diseño', 'Validación de Estandares') 
                      OR ENTREGABLE LIKE 'Desarrollo%' THEN 'DESARROLLO'
                    ELSE 'GESTIÓN'
                END AS TIPO
            FROM (
                SELECT 
                    CASE 
                        WHEN ENTREGABLE = 'Planeación y Gestión' THEN 1
                        WHEN ENTREGABLE = 'Análisis y Diseño' THEN 2
                        WHEN ENTREGABLE LIKE 'Desarrollo%' THEN 3
                        WHEN ENTREGABLE = 'Documentación' THEN 4
                        WHEN ENTREGABLE = 'Validación de Estandares' THEN 5
                        WHEN ENTREGABLE = 'Certificación de servicio' THEN 6
                        WHEN ENTREGABLE = 'Proceso de Cambios' THEN 7
                        WHEN ENTREGABLE = 'Producción' THEN 8
                        WHEN ENTREGABLE = 'Pos-Producción' THEN 9
                        ELSE 99
                    END AS ORDEN_REAL,
                    ENTREGABLE,
                    CE.PERFIL,
                    (SELECT COALESCE(TIPO_PREREQUISITO, '') 
                     FROM DF_PORCEN_ESTIMACION PO 
                     WHERE PO.ENTREGABLE = E.ENTREGABLE AND ROWNUM = 1) AS TIPO_PREREQUISITO,
                    COALESCE(E.PREREQUISITO, '') AS PREREQUISITO,
                    ROUND((E.COSTO/CE.COSTO) / 9, 2) AS DIAS_HABILES,
                    CE.COSTO * (E.COSTO/CE.COSTO) AS COSTO
                FROM DF_ESTIMACION E
                LEFT JOIN DF_COSTO_ESTIMACION CE ON E.PERFIL = CE.ID
                WHERE E.ID_PROYECTO_ESTIMACION = P_ID_PROY_ESTIMA
                ORDER BY ORDEN_REAL
            )
        
            UNION ALL
        
            -- 3. TOTAL DESARROLLO
            SELECT 
                '98' AS ORDEN,
                'T1' AS ID,
                'TOTAL DESARROLLO' AS ENTREGABLE,
                ' ' AS PERFIL,
                ' ' AS TIPO_PREREQUISITO,
                ' ' AS PREREQUISITO,
                TO_CHAR(ROUND(SUM(DIAS_HABILES), 2)) AS DIAS_HABILES,
                TO_CHAR(SUM(COSTO)) AS COSTO,
                'DESARROLLO' AS TIPO
            FROM (
                SELECT 
                    ROUND((E.COSTO/CE.COSTO) / 9, 2) AS DIAS_HABILES,
                    CE.COSTO * (E.COSTO/CE.COSTO) AS COSTO
                FROM DF_ESTIMACION E
                LEFT JOIN DF_COSTO_ESTIMACION CE ON E.PERFIL = CE.ID
                WHERE E.ID_PROYECTO_ESTIMACION = P_ID_PROY_ESTIMA
                  AND (
                    E.ENTREGABLE = 'Análisis y Diseño'
                    OR E.ENTREGABLE LIKE 'Desarrollo%'
                    OR E.ENTREGABLE = 'Validación de Estandares'
                  )
            )
        
            UNION ALL
        
            -- 4. TOTAL GESTION
            SELECT 
                '99' AS ORDEN,
                'T2' AS ID,
                'TOTAL GESTION' AS ENTREGABLE,
                ' ' AS PERFIL,
                ' ' AS TIPO_PREREQUISITO,
                ' ' AS PREREQUISITO,
                TO_CHAR(ROUND(SUM(DIAS_HABILES), 2)) AS DIAS_HABILES,
                TO_CHAR(SUM(COSTO)) AS COSTO,
                'GESTIÓN' AS TIPO
            FROM (
                SELECT 
                    ROUND((E.COSTO/CE.COSTO) / 9, 2) AS DIAS_HABILES,
                    CE.COSTO * (E.COSTO/CE.COSTO) AS COSTO
                FROM DF_ESTIMACION E
                LEFT JOIN DF_COSTO_ESTIMACION CE ON E.PERFIL = CE.ID
                WHERE E.ID_PROYECTO_ESTIMACION = P_ID_PROY_ESTIMA
                  AND NOT (
                    E.ENTREGABLE = 'Análisis y Diseño'
                    OR E.ENTREGABLE LIKE 'Desarrollo%'
                    OR E.ENTREGABLE = 'Validación de Estandares'
                  )
            )

            UNION ALL

            -- 5. TOTAL FINAL
            SELECT 
                '99.5' AS ORDEN,
                'TF' AS ID,
                'TOTAL FINAL' AS ENTREGABLE,
                ' ' AS PERFIL,
                ' ' AS TIPO_PREREQUISITO,
                ' ' AS PREREQUISITO,
                TO_CHAR(ROUND(SUM(DIAS_HABILES), 2)) AS DIAS_HABILES,
                TO_CHAR(SUM(COSTO)) AS COSTO,
                'FINAL' AS TIPO
            FROM (
                SELECT 
                    ROUND((E.COSTO/CE.COSTO) / 9, 2) AS DIAS_HABILES,
                    CE.COSTO * (E.COSTO/CE.COSTO) AS COSTO
                FROM DF_ESTIMACION E
                LEFT JOIN DF_COSTO_ESTIMACION CE ON E.PERFIL = CE.ID
                WHERE E.ID_PROYECTO_ESTIMACION = P_ID_PROY_ESTIMA
            )
        )
        ORDER BY ORDEN;

        V_S_CODIGO := 200;
        V_S_RESPUESTA := 'Retorno de datos completo';

    EXCEPTION
        WHEN OTHERS THEN
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'Error: ' || SQLERRM;
    END;
END SP_REPORT_EXCEL_ESTIMA;
