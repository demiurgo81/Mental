PROCEDURE SP_SAVE_REGISTRO_EMPLEADO (
    P_NUMERODOCUMENTO IN VARCHAR2,
    P_CODTIPODOCUMENTO IN NUMBER,
    P_NOMBRE1 IN VARCHAR2,
    P_NOMBRE2 IN VARCHAR2,
    P_APELLIDO1 IN VARCHAR2,
    P_APELLIDO2 IN VARCHAR2,
    P_CORREO IN VARCHAR2,
    P_FECHANACIMIENTO IN DATE,
    P_CODTERCEROEPS IN NUMBER,
    P_GENERO IN VARCHAR2,
    P_USUARIOCLARO IN VARCHAR2,
    P_CODPROVEEDOR IN NUMBER,
    V_ID_CREADO OUT NUMBER,
    V_S_RESPUESTA OUT VARCHAR2,
    V_S_CODIGO OUT NUMBER
)
AS
    V_COUNT NUMBER;
    V_COUNT_USER NUMBER;
    V_RESPUESTA VARCHAR2(4000);
    V_CODIGO NUMBER;
    V_ID_PERSONA NUMBER;
    V_MENSAJE VARCHAR2(4000);
    V_COUNT_PERSONA NUMBER;
    V_ID_EMPLEADO NUMBER;
    V_ID_USUARIO NUMBER;
    V_NOMBRECOMPLETO VARCHAR2(200);
    V_NOMBRE2 VARCHAR2(50);
    V_APELLIDO2 VARCHAR2(50);
BEGIN
    BEGIN
        -- Verificar si el registro ya existe
        
        SELECT COUNT(*) INTO V_COUNT_PERSONA FROM DF_PERSONA WHERE TO_CHAR(NUMERODOCUMENTO)= P_NUMERODOCUMENTO;
        
        SELECT COUNT(*) INTO V_COUNT_USER FROM DF_USUARIO WHERE USUARIO = P_USUARIOCLARO;
        
        DBMS_OUTPUT.PUT_LINE('VALIDA QUE EXISTA LA PERSONA POR CEDULA ' || V_COUNT_PERSONA);
    
        V_NOMBRECOMPLETO := P_NOMBRE1 || ' ' || P_NOMBRE2 || ' ' || P_APELLIDO1 || ' ' || P_APELLIDO2;
        
        IF P_NOMBRE2 IS NULL THEN
            V_NOMBRE2 := '';
        END IF;

        IF P_APELLIDO2 IS NULL THEN
            V_APELLIDO2 := '';
        END IF;
                
        IF V_COUNT_PERSONA > 0 THEN 
            V_S_RESPUESTA := 'Esta persona ya existe en la herramienta';
            V_MENSAJE := 'Esta persona ya existe en la herramienta';
            V_S_CODIGO := 409; -- Código HTTP de "conflicto"
        ELSIF V_COUNT_USER > 0 THEN
            V_S_RESPUESTA := 'El usuario ya existe en la herramienta';
            V_MENSAJE := 'El usuario ya existe en la herramienta';
            V_S_CODIGO := 409;
        ELSE
            -- INSERTA EN DF_PERSONA
            
            SP_SAVE_DF_PERSONA(0, P_CODTIPODOCUMENTO, 0, P_NOMBRE1, V_NOMBRE2, P_APELLIDO1, V_APELLIDO2,
                '', P_CORREO, '', P_GENERO, P_FECHANACIMIENTO, P_CODTERCEROEPS, '', '', '', '', '', P_NUMERODOCUMENTO,
                V_ID_PERSONA, V_CODIGO, V_MENSAJE
            );
            
            DBMS_OUTPUT.PUT_LINE('INSERTA EN DF_PERSONA ' || V_ID_PERSONA);
            
            -- INSERTA EN DF_EMPLEADO
            
            IF V_ID_PERSONA IS NOT NULL AND V_ID_PERSONA > 0 THEN
            
            SP_SAVE_DF_EMPLEADO(0, V_ID_PERSONA, 4, 12, 95, P_CODPROVEEDOR,
                1, P_USUARIOCLARO, SYSDATE, NULL, NULL, 0, NULL, NULL, NULL,
                V_RESPUESTA, V_ID_EMPLEADO, V_CODIGO);
            
            DBMS_OUTPUT.PUT_LINE('INSERTA EN DF_EMPLEADO ' || V_ID_EMPLEADO);
            
            -- INSERTA DF_USUARIO
            
            SP_SAVE_DF_USUARIO(0, 2, V_ID_PERSONA, P_USUARIOCLARO, V_NOMBRECOMPLETO,
                '', 'I', NULL, NULL, V_RESPUESTA, V_ID_USUARIO, V_CODIGO );
            
            DBMS_OUTPUT.PUT_LINE('INSERTA EN DF_USUARIO ' || V_ID_USUARIO);
            
            ELSE
                V_S_RESPUESTA := 'Error al crear la persona. No se insertaron registros adicionales.';
                V_MENSAJE := 'Error en la inserción de persona, no se generaron otros registros.';
                V_S_CODIGO := 500;
            END IF;
        -- Confirmar transacción
        COMMIT;
        
        V_ID_CREADO := V_ID_PERSONA;
        IF V_ID_PERSONA IS NOT NULL AND V_ID_PERSONA > 0 THEN
            V_S_RESPUESTA := 'Creado exitoso';
            V_MENSAJE := 'Creado exitosamente, por favor comuníquese con su líder para que sea activado en la herramienta';
            V_S_CODIGO := 200;
        END IF;
        
        END IF;
        V_S_CODIGO := 200;

    EXCEPTION
        WHEN OTHERS THEN
            -- Manejo de errores
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'Error: ' || SQLERRM;
            ROLLBACK;
    END;
END SP_SAVE_REGISTRO_EMPLEADO;
