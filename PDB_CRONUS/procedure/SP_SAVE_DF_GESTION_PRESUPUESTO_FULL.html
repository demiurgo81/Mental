PROCEDURE SP_SAVE_DF_GESTION_PRESUPUESTO_FULL (
    P_ID_GESTION IN NUMBER,
    P_GESTION IN VARCHAR2,
    P_ID_PROYECTO IN NUMBER,
    P_ID_PROVEEDOR IN NUMBER,
    P_ID_CONTRATOMARCO IN NUMBER,
    P_CONTRATOMARCO IN VARCHAR2,
    P_ID_PEP IN NUMBER,
    P_PEP IN VARCHAR2,
    P_ID_SOLPED IN NUMBER,
    P_SOLPED IN VARCHAR2,
    P_ID_TEMA IN NUMBER,
    P_TEMA IN VARCHAR2,
    P_ID_ORDEN_COMPRA IN NUMBER,
    P_ORDEN_COMPRA IN VARCHAR2,
    P_ID_FACTURA_COMPRA IN NUMBER,
    P_FACTURA_COMPRA IN VARCHAR2,
    P_RESPONSABLE IN VARCHAR2,
    P_ESTADO IN NUMBER,
    P_TRAMITA_CON IN VARCHAR2,
    P_CODEMPLEADO_CREO IN NUMBER,
    P_CODEMPLEADO_MODIFICA IN NUMBER,
    V_ID_CREADO OUT NUMBER,
    V_S_RESPUESTA OUT VARCHAR2,
    V_S_CODIGO OUT NUMBER
)
AS
    V_COUNT NUMBER;
    V_RESPUESTA VARCHAR2(4000);
    V_CODIGO NUMBER;
    V_CODIGO_OUT NUMBER;
    V_MENSAJE VARCHAR2(4000);
    V_MENSAJE_OUT VARCHAR2(4000);
    V_ID_CONTRATOMARCO NUMBER;
    V_ID_SOLPED NUMBER;
    V_ID_ORDEN_COMPRA NUMBER;
    V_ID_TEMA NUMBER;
    V_ID_FACTURA_COMPRA NUMBER;
    V_ID_GESTION_SAVE NUMBER;
    V_GESTION VARCHAR2(100);
BEGIN
    BEGIN
           
        SP_SAVE_CONTRATO_MARCO(P_ID_CONTRATOMARCO, P_CONTRATOMARCO, P_ID_PROVEEDOR, P_CODEMPLEADO_CREO, P_CODEMPLEADO_MODIFICA, 
            V_ID_CONTRATOMARCO, V_MENSAJE_OUT, V_CODIGO_OUT); 
        DBMS_OUTPUT.PUT_LINE('LLEGA AQUI V_ID_CONTRATOMARCO: ' || V_ID_CONTRATOMARCO);    
        IF V_CODIGO_OUT <> 200 THEN 
            V_S_RESPUESTA := 'No se puede crear o encontrar el contrato marco: ' || V_MENSAJE_OUT;
            V_S_CODIGO := V_CODIGO_OUT;
            rollback;
            return;
        END IF;
        
        SP_SAVE_SOLPED(P_ID_SOLPED, P_SOLPED, CURRENT_TIMESTAMP, 1, '', '', P_CODEMPLEADO_CREO, P_CODEMPLEADO_MODIFICA, 0,
            V_ID_SOLPED, V_MENSAJE_OUT, V_CODIGO_OUT);
        DBMS_OUTPUT.PUT_LINE('LLEGA AQUI V_ID_SOLPED: ' || V_ID_SOLPED);    
        IF V_CODIGO_OUT <> 200 THEN 
            V_S_RESPUESTA := 'No se puede crear o encontrar el solped: ' || V_MENSAJE_OUT;
            V_S_CODIGO := V_CODIGO_OUT;
            rollback;
            return;
        END IF;    
           
        SP_SAVE_DF_ORDEN_COMPRA(P_ID_ORDEN_COMPRA, P_ORDEN_COMPRA, CURRENT_TIMESTAMP, P_CODEMPLEADO_CREO, P_CODEMPLEADO_MODIFICA, 
            V_ID_ORDEN_COMPRA, V_MENSAJE_OUT, V_CODIGO_OUT);
        DBMS_OUTPUT.PUT_LINE('LLEGA AQUI V_ID_ORDEN_COMPRA: ' || V_ID_ORDEN_COMPRA);      
        IF V_CODIGO_OUT <> 200 THEN 
            V_S_RESPUESTA := 'No se puede crear o encontrar la orden de compra: ' || V_MENSAJE_OUT;
            V_S_CODIGO := V_CODIGO_OUT;
            rollback;
            return;
        END IF;       
        
        SP_SAVE_TEMA_BITACORA(P_ID_TEMA, P_TEMA, P_CODEMPLEADO_CREO, P_CODEMPLEADO_MODIFICA, 
            V_ID_TEMA, V_MENSAJE_OUT, V_CODIGO_OUT);
        DBMS_OUTPUT.PUT_LINE('LLEGA AQUI V_ID_TEMA: ' || V_ID_TEMA);         
        IF V_CODIGO_OUT <> 200 THEN 
            V_S_RESPUESTA := 'No se puede crear o encontrar el tema de la bitacora: ' || V_MENSAJE_OUT;
            V_S_CODIGO := V_CODIGO_OUT;
            rollback;
            return;
        END IF;       
        
        SP_SAVE_DF_FACTURA_COMPRA(P_ID_FACTURA_COMPRA, P_FACTURA_COMPRA, CURRENT_TIMESTAMP, P_CODEMPLEADO_CREO, P_CODEMPLEADO_MODIFICA, 
            V_ID_FACTURA_COMPRA, V_MENSAJE_OUT, V_CODIGO_OUT);
        DBMS_OUTPUT.PUT_LINE('LLEGA AQUI V_ID_FACTURA_COMPRA: ' || V_ID_FACTURA_COMPRA);             
        IF V_CODIGO_OUT <> 200 THEN 
            V_S_RESPUESTA := 'No se puede crear o encontrar la factura de compra: ' || V_MENSAJE_OUT;
            V_S_CODIGO := V_CODIGO_OUT;
            rollback;
            return;
        END IF; 
        
        IF  P_GESTION IS NULL OR TRIM(P_GESTION) = '' THEN
            V_GESTION := 'GESTION PAGO ' || TO_CHAR(SYSDATE, 'DD/MM/YYYY HH24:MI:SS');
        ELSE
            V_GESTION:= P_GESTION;
        END IF;
        SP_SAVE_DF_GESTION_PRESUPUESTO(P_ID_GESTION, V_GESTION, P_ID_PROYECTO, P_ID_PROVEEDOR, V_ID_CONTRATOMARCO, P_ID_PEP,
            V_ID_SOLPED, V_ID_TEMA, V_ID_ORDEN_COMPRA, V_ID_FACTURA_COMPRA, P_RESPONSABLE, P_ESTADO, P_TRAMITA_CON, 
            P_CODEMPLEADO_CREO, P_CODEMPLEADO_MODIFICA, V_ID_GESTION_SAVE, V_MENSAJE_OUT, V_CODIGO_OUT);
        DBMS_OUTPUT.PUT_LINE('LLEGA AQUI V_ID_GESTION_SAVE: ' || V_ID_GESTION_SAVE);             
        IF V_CODIGO_OUT = 200 THEN 
            V_S_RESPUESTA :=  V_MENSAJE_OUT;
            V_S_CODIGO := 200;
            V_ID_CREADO := V_ID_GESTION_SAVE;
            COMMIT;
        ELSE 
            V_S_RESPUESTA :=  'Hubo un error, por favor contacte al administrador '|| V_MENSAJE_OUT;
            V_S_CODIGO := 400;
        END IF; 
        
    EXCEPTION
        WHEN OTHERS THEN
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'Error: ' || SQLERRM;
            ROLLBACK;
    END;
END SP_SAVE_DF_GESTION_PRESUPUESTO_FULL;
