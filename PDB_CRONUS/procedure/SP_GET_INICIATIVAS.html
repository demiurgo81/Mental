PROCEDURE SP_GET_INICIATIVAS (
    P_ID_INICIATIVA IN NUMBER,
    P_CODTAREA IN NUMBER,
    P_CURSOR_GET_INICIATIVAS OUT SYS_REFCURSOR,
    V_S_RESPUESTA OUT VARCHAR2,
    V_S_CODIGO OUT NUMBER
)
AS
BEGIN
    BEGIN
        -- Primer cursor para obtener las iniciativas según el ID
        OPEN P_CURSOR_GET_INICIATIVAS FOR
            SELECT DISTINCT I.ID, I.NOMBRE, PE.PEP, PE.ID AS ID_PEP, I.CODPROYECTO, PE.CODGERENTE, I.FECHACREACION, I.FECHAMODIFICACION, 
                P.NOMCOMPLETO AS NOMBREGERENTE, PR.NOMBRE AS NOMBREPROYECTO, I.EVIDENCIA, L.LIDER_INICIATIVA, I.ID_LIDER_INICIATIVA, I.CODTAREA
            FROM DF_INICIATIVA I
                LEFT JOIN DF_PEP PE ON PE.ID = I.ID_PEP
                LEFT JOIN DF_EMPLEADO E ON E.ID = PE.CODGERENTE
                LEFT JOIN DF_PERSONA P ON P.ID = E.CODPERSONA
                LEFT JOIN DF_PROYECTO PR ON PR.ID = I.CODPROYECTO
                LEFT JOIN DF_TAREA TP ON TP.NOMBRE = I.NOMBRE
                LEFT JOIN DF_LIDER_INICIATIVA L ON L.ID = I.ID_LIDER_INICIATIVA
                WHERE (P_ID_INICIATIVA = 0 OR I.ID = P_ID_INICIATIVA)
                AND tp.nivel = 1 AND ( P_CODTAREA = 0 OR I.CODTAREA = P_CODTAREA )
                /*AND tp.FECHAINIESTIMADA BETWEEN TRUNC(SYSDATE, 'MM') 
                            AND TRUNC(ADD_MONTHS(SYSDATE, 1), 'MM') - 1/24/60/60 
    AND tp.FECHAFINESTIMADA BETWEEN TRUNC(SYSDATE, 'MM') 
                              AND TRUNC(ADD_MONTHS(SYSDATE, 1), 'MM') - 1/24/60/60*/;

        V_S_RESPUESTA := 'Retorno de datos completo';
        V_S_CODIGO := 200;
    EXCEPTION
        WHEN OTHERS THEN
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'Error: ' || SQLERRM;
    END;
END SP_GET_INICIATIVAS;
