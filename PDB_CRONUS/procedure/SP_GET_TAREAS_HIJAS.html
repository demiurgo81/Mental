PROCEDURE "SP_GET_TAREAS_HIJAS" (
    P_CODTAREA_PADRE IN NUMBER,
    P_FECHA_INI IN DATE, 
    P_FECHA_FIN IN DATE,
    P_CURSOR_GET_TAREAS_HIJAS OUT SYS_REFCURSOR,
    V_S_RESPUESTA OUT VARCHAR2,
    V_S_CODIGO OUT NUMBER
)
AS
BEGIN
    BEGIN
        OPEN P_CURSOR_GET_TAREAS_HIJAS FOR
     /*   SELECT th.ID, th.NOMBRE, th.CODTAREAESTADO, th.FECHAHORAINICIO, th.FECHAHORAFIN, th.HORAS 
    FROM 
        (
            SELECT tp.id, tp.nombre, tp.codtareaestado, MIN(ec.FECHAHORAINICIO) as FECHAHORAINICIO, MAX(ec.FECHAHORAFIN) as FECHAHORAFIN, SUM(ec.horas) AS HORAS
            FROM 
                DF_TAREA tp
            LEFT JOIN 
                df_empleadocontrol ec ON ec.codtarea = tp.id
            WHERE 
                tp.codtareapadre IN (P_CODTAREA_PADRE)
            GROUP BY 
                tp.id, tp.nombre, tp.codtareaestado ) th ;
                */
                --- ACTUALIZAR LA CONSULTA DESPUES DE REVISAR LA ACTUALIZACIÓN DE LAS ESTIMACIONES

        SELECT th.ID, th.NOMBRE, th.CODTAREAESTADO, th.FECHAHORAINICIO, th.FECHAHORAFIN, th.HORAS 
       FROM 
        (
            SELECT tp.id, tp.nombre, tp.codtareaestado, MIN(ec.FECHAHORAINICIO) as FECHAHORAINICIO, 
                MAX(ec.FECHAHORAFIN) as FECHAHORAFIN, SUM(ec.horas) AS HORAS
                FROM DF_TAREA tp 
                LEFT JOIN df_empleadocontrol ec ON ec.codtarea = tp.id
                LEFT JOIN DF_PROYECTO_ESTIMACION PE ON PE.CODTAREA = tp.CODTAREAPADRE 
                LEFT JOIN DF_INICIATIVA I ON I.CODTAREA = tp.CODTAREAPADRE 
                LEFT JOIN DF_PEP PEP ON PEP.ID = I.ID_PEP
                WHERE tp.codtareapadre IN (P_CODTAREA_PADRE)
               -- AND (PE.ID_ESTADO = 2 OR PE.ID_ESTADO IS NULL) AND PEP.ESTADO = 2
                    GROUP BY tp.id, tp.nombre, tp.codtareaestado
        ) th ; 


        V_S_CODIGO := 200;
        V_S_RESPUESTA := 'Retorno Correcto';
    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            ROLLBACK;
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'No Fue posible traer la data';
            RAISE;
    END;
END SP_GET_TAREAS_HIJAS;
