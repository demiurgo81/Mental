PROCEDURE SP_SAVE_DF_EMPLEADO (
    P_ID IN NUMBER,
    P_CODPERSONA IN NUMBER,
    P_CODPERFIL IN NUMBER,
    P_CODPERFILTIPO IN NUMBER,
    P_CODLINEAPRODUCTO IN NUMBER,
    P_CODPROVEEDOR IN NUMBER,
    P_CODPERFILNIVEL IN NUMBER,
    P_USUARIOCLARO IN VARCHAR2,
    P_FECHAINICIO IN DATE,
    P_FECHAFIN IN DATE DEFAULT NULL,
    P_CODESTRUCTURAORGANIZACIONAL IN NUMBER,
    P_CODMODALIDAD IN NUMBER,
    P_CELULA IN VARCHAR2,
    P_CHORA IN NUMBER,
    P_CMES IN NUMBER,
    V_S_RESPUESTA OUT VARCHAR2,
    V_ID_CREADO OUT NUMBER,
    V_S_CODIGO OUT NUMBER
)
AS
    V_COUNT NUMBER;
    V_USUARIO VARCHAR2(20);
BEGIN
    BEGIN
        -- Verificar si el registro ya existe en la tabla
        SELECT COUNT(*) INTO V_COUNT FROM DF_EMPLEADO WHERE ID = P_ID; --OR ;

        -- Validar si el usuario está vacío
        IF P_USUARIOCLARO IS NULL OR P_USUARIOCLARO = '' THEN
            V_USUARIO := 'SIN USUARIO';
        ELSE
            V_USUARIO := P_USUARIOCLARO;
        END IF;

        -- Si el registro existe, actualizarlo; de lo contrario, insertar uno nuevo
        IF V_COUNT > 0 THEN
            UPDATE DF_EMPLEADO
            SET CODPERSONA = P_CODPERSONA,
                CODPERFIL = P_CODPERFIL,
                CODPERFILTIPO = P_CODPERFILTIPO,
                CODLINEAPRODUCTO = P_CODLINEAPRODUCTO,
                CODPROVEEDOR = P_CODPROVEEDOR,
                CODPERFILNIVEL = P_CODPERFILNIVEL,
                USUARIOCLARO = V_USUARIO,
                FECHAINICIO = P_FECHAINICIO,
                FECHAFIN = P_FECHAFIN,
                CODESTRUCTURAORGANIZACIONAL = P_CODESTRUCTURAORGANIZACIONAL,
                CODMODALIDAD = P_CODMODALIDAD,
                CELULA = P_CELULA,
                CHORA = P_CHORA,
                CMES = P_CMES,
                FECHAMODIFICACION = CURRENT_TIMESTAMP
            WHERE ID = P_ID;

            V_ID_CREADO := P_ID;
            V_S_RESPUESTA := 'Registro actualizado exitosamente';
        ELSE
            INSERT INTO DF_EMPLEADO (
                CODPERSONA, CODPERFIL, CODPERFILTIPO, CODLINEAPRODUCTO, CODPROVEEDOR,
                CODPERFILNIVEL, USUARIOCLARO, FECHAINICIO, CODESTRUCTURAORGANIZACIONAL, CODMODALIDAD, CELULA, CHORA, CMES,
                FECHACREACION, FECHAMODIFICACION
            ) VALUES (
                P_CODPERSONA, P_CODPERFIL, P_CODPERFILTIPO, P_CODLINEAPRODUCTO, P_CODPROVEEDOR,
                P_CODPERFILNIVEL, V_USUARIO, P_FECHAINICIO, P_CODESTRUCTURAORGANIZACIONAL, P_CODMODALIDAD, P_CELULA, P_CHORA, P_CMES,
                CURRENT_TIMESTAMP, CURRENT_TIMESTAMP
            )
            RETURNING ID INTO V_ID_CREADO;

            V_S_RESPUESTA := 'Registro insertado exitosamente';
        END IF;

        COMMIT;
        V_S_CODIGO := 200;

    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'No existe estructura';
            ROLLBACK;
        WHEN OTHERS THEN
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'Error: ' || SQLERRM;
            ROLLBACK;
    END;
END SP_SAVE_DF_EMPLEADO;
