PROCEDURE SP_GET_DF_GESTION_PRESUPUESTO_FULL (
    P_ID_GESTION IN VARCHAR2,
    P_CURSOR_GET_DF_GESTION_PRESUPUESTO OUT SYS_REFCURSOR,
    V_S_RESPUESTA OUT VARCHAR2,
    V_S_CODIGO OUT NUMBER
)
AS
    V_COUNT NUMBER := 0; -- Inicializar la variable
BEGIN
    BEGIN
        -- Validar si el documento ya existe
        OPEN P_CURSOR_GET_DF_GESTION_PRESUPUESTO FOR
            SELECT G.ID, G.GESTION, G.ID_PROYECTO, P.NOMBRE AS NOMBRE_PROYECTO, 
                G.ID_PROVEEDOR, PRO.NOMBRE AS NOMBRE_PROVEEDOR,
                G.ID_CONTRATOMARCO, CM.CONTRATOMARCO,
                G.ID_SOLPED, S.SOLPED,
                G.ID_TEMA, T.TEMA,
                G.ID_ORDEN_COMPRA, OC.CODIGO_ORDEN_COMPRA,
                G.ID_FACTURA_COMPRA, FC.CODIGO_FACTURA_COMPRA,
                G.CODEMPLEADO_CREO, G.CODEMPLEADO_MODIFICA, G.FECHACREACION, G.FECHAMODIFICACION, G.ESTADO
                FROM DF_GESTION_PRESUPUESTO G
                LEFT JOIN DF_PROYECTO P ON P.ID = G.ID_PROYECTO
                LEFT JOIN DF_PROVEEDOR PRO ON PRO.ID = G.ID_PROVEEDOR
                LEFT JOIN DF_CONTRATO_MARCO CM ON CM.ID = G.ID_CONTRATOMARCO
                LEFT JOIN DF_SOLPED S ON S.ID = G.ID_SOLPED
                LEFT JOIN DF_TEMA_BITACORA T ON T.ID = G.ID_TEMA
                LEFT JOIN DF_ORDEN_COMPRA OC ON OC.ID = G.ID_ORDEN_COMPRA
                LEFT JOIN DF_FACTURA_COMPRA FC ON FC.ID = G.ID_FACTURA_COMPRA
            WHERE (P_ID_GESTION = 0 OR G.ID = P_ID_GESTION);

        V_S_CODIGO := 200;
        V_S_RESPUESTA := 'Datos retornados exitosamente';

    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            V_S_CODIGO := 404;
            V_S_RESPUESTA := 'No se encontró información';
        WHEN OTHERS THEN
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'Error: ' || SQLERRM;
    END;
END SP_GET_DF_GESTION_PRESUPUESTO_FULL;
