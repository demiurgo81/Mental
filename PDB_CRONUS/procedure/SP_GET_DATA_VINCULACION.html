PROCEDURE "SP_GET_DATA_VINCULACION" (
    P_CODEMPLEADO IN NUMBER,
    P_CURSOR_DATA_VINCULACION OUT SYS_REFCURSOR,
    V_S_RESPUESTA OUT VARCHAR2,
    V_S_CODIGO OUT NUMBER
)
AS
    V_IDROL NUMBER := 0;
    V_ID_ESTRUCTURAORGANIZACIONAL NUMBER;
    V_ID_COORDINACION NUMBER;
BEGIN
    BEGIN
        
        DBMS_OUTPUT.PUT_LINE('INICIA PROCEDURE P_CODEMPLEADO ' || P_CODEMPLEADO);

        
        IF P_CODEMPLEADO IS NULL OR P_CODEMPLEADO = 0 THEN 
            -- TRAE SOLO LOS ACTIVOS
            DBMS_OUTPUT.PUT_LINE('LLEGA AL SI ' );
            OPEN P_CURSOR_DATA_VINCULACION FOR 
                SELECT CODEMPLEADO, CODPERSONA, NOMBRE, TIPODOCUMENTO, NUMERODOCUMENTO, CORREO, PERFIL, PROVEEDOR, USUARIOCLARO, 
                    ESTRUCTURAORGANIZACIONAL, TIPOESTRUCTURA, LEAD, VALORMES, NOMBRE_GERENTE, NOMBRE_COORDINADOR, GERENCIA, COORDINACION 
                        FROM DF_VINCULACION_DATA 
                        WHERE VIGENTE = 1;
        ELSE
            
              SELECT U.CODUSUARIOROL, E.CODESTRUCTURAORGANIZACIONAL, EO.CODPADRE
                INTO V_IDROL, V_ID_ESTRUCTURAORGANIZACIONAL, V_ID_COORDINACION FROM DF_USUARIO U
                LEFT JOIN DF_PERSONA P ON P.ID = U.CODPERSONA
                LEFT JOIN DF_EMPLEADO E ON E.CODPERSONA = P.ID
                LEFT JOIN DF_ESTRUCTURAORGANIZACIONAL EO ON EO.ID = E.CODESTRUCTURAORGANIZACIONAL
                WHERE E.ID = P_CODEMPLEADO;
                
                DBMS_OUTPUT.PUT_LINE('LLEGA AL ELSE ' || V_IDROL || ' ' || V_ID_ESTRUCTURAORGANIZACIONAL);
                    
                IF V_IDROL = 7 THEN
                    OPEN P_CURSOR_DATA_VINCULACION FOR 
                    SELECT CODEMPLEADO, CODPERSONA, NOMBRE, TIPODOCUMENTO, NUMERODOCUMENTO, CORREO, PERFIL, PROVEEDOR, USUARIOCLARO, 
                        ESTRUCTURAORGANIZACIONAL, TIPOESTRUCTURA, LEAD, VALORMES, NOMBRE_GERENTE, NOMBRE_COORDINADOR, GERENCIA, COORDINACION 
                        FROM DF_VINCULACION_DATA;
                        
                ELSIF  V_IDROL = 5 THEN
                    OPEN P_CURSOR_DATA_VINCULACION FOR 
                    SELECT CODEMPLEADO, CODPERSONA, NOMBRE, TIPODOCUMENTO, NUMERODOCUMENTO, CORREO, PERFIL, PROVEEDOR, USUARIOCLARO, 
                        ESTRUCTURAORGANIZACIONAL, TIPOESTRUCTURA, LEAD, VALORMES, NOMBRE_GERENTE, NOMBRE_COORDINADOR, GERENCIA, COORDINACION 
                            FROM DF_VINCULACION_DATA
                            WHERE ID_COORDINACION = V_ID_COORDINACION;
                ELSE 
                    OPEN P_CURSOR_DATA_VINCULACION FOR 
                    SELECT CODEMPLEADO, CODPERSONA, NOMBRE, TIPODOCUMENTO, NUMERODOCUMENTO, CORREO, PERFIL, PROVEEDOR, USUARIOCLARO, 
                        ESTRUCTURAORGANIZACIONAL, TIPOESTRUCTURA, LEAD, VALORMES, NOMBRE_GERENTE, NOMBRE_COORDINADOR, GERENCIA, COORDINACION 
                            FROM DF_VINCULACION_DATA
                            WHERE ID_CELULA = V_ID_ESTRUCTURAORGANIZACIONAL;
                END IF;
                
        END IF;
        V_S_CODIGO := 200;
        V_S_RESPUESTA := 'Retorno de datos completo ';
    EXCEPTION
        WHEN OTHERS THEN
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'Error: ' || SQLERRM;
    END;
END SP_GET_DATA_VINCULACION;
