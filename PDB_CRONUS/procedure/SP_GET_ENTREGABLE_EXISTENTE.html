PROCEDURE SP_GET_ENTREGABLE_EXISTENTE (
    P_ENTREGABLE IN VARCHAR2,
    P_ID_PEP IN NUMBER,
    --P_PEP IN VARCHAR2,
    V_S_RESPUESTA OUT VARCHAR2,
    V_S_CODIGO OUT NUMBER
)
AS
    V_COUNT NUMBER := 0; -- Inicializar la variable
    V_PEP VARCHAR2(200);
BEGIN
    BEGIN
        -- Validar si el documento ya existe PEP = P_PEP OR
        SELECT COUNT(*) INTO V_COUNT
            FROM DF_INICIATIVA WHERE (UPPER(TRIM(NOMBRE)) = UPPER(TRIM(P_ENTREGABLE))) AND (  ID_PEP = P_ID_PEP );
        
        DBMS_OUTPUT.PUT_LINE('NO ENCUENTRA CON NOMBRE ENTREGABLE NI IDPEP EL V_COUNT ' || V_COUNT);
        
        -- Responder en función de los resultados
        IF V_COUNT > 0 THEN
            SELECT PEP INTO V_PEP FROM DF_PEP WHERE ID = P_ID_PEP;
            V_S_RESPUESTA := 'Este Entregable, ya existe con este mismo nombre ' || P_ENTREGABLE || ' y mismo PEP ' || V_PEP ||', desea continuar? ';
            V_S_CODIGO := 200;
        ELSE
            SELECT COUNT(*) INTO V_COUNT FROM DF_INICIATIVA WHERE 
                ID_PEP <> P_ID_PEP AND
                UPPER(TRIM(NOMBRE)) = UPPER(TRIM(P_ENTREGABLE));
            
                IF V_COUNT > 0 THEN
                    V_S_RESPUESTA := 'Existe otro entregable con este mismo nombre, pero con PEP diferente, debe tener otro nombre el entregable. Verifique los datos.';
                    V_S_CODIGO := 401; -- Código diferente para esta validación
                ELSE
                    V_S_RESPUESTA := 'No existen entregables con este nombre y este PEP, puede proceder.';
                    V_S_CODIGO := 100;
                END IF;
        END IF;

    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            V_S_CODIGO := 404;
            V_S_RESPUESTA := 'No se encontró información';
        WHEN OTHERS THEN
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'Error: ' || SQLERRM;
    END;
END SP_GET_ENTREGABLE_EXISTENTE;
