PROCEDURE SP_SAVE_DF_GESTION_PRESUPUESTO (
    P_ID_GESTION IN NUMBER,
    P_GESTION IN VARCHAR2,
    P_ID_PROYECTO IN NUMBER,
    P_ID_PROVEEDOR IN NUMBER,
    P_ID_CONTRATOMARCO IN NUMBER,
    P_ID_PEP IN NUMBER,
    P_ID_SOLPED IN NUMBER,
    P_ID_TEMA IN NUMBER,
    P_ID_ORDEN_COMPRA IN NUMBER,
    P_ID_FACTURA_COMPRA IN NUMBER,
    P_RESPONSABLE IN VARCHAR2,
    P_ESTADO IN NUMBER,
    P_TRAMITA_CON IN VARCHAR2,
    P_CODEMPLEADO_CREO IN NUMBER,
    P_CODEMPLEADO_MODIFICA IN NUMBER,
    V_ID_CREADO OUT NUMBER,
    V_S_RESPUESTA OUT VARCHAR2,
    V_S_CODIGO OUT NUMBER
)
AS
    V_COUNT NUMBER;
BEGIN
    BEGIN
        -- Verificar si el registro ya existe
        SELECT COUNT(*)
        INTO V_COUNT
        FROM DF_GESTION_PRESUPUESTO
        WHERE ID = P_ID_GESTION;
        DBMS_OUTPUT.PUT_LINE('LLEGA AQUI DF_GESTION_PRESUPUESTO V_COUNT: ' || V_COUNT);
        -- Si el registro existe, actualizarlo; de lo contrario, insertarlo
        IF V_COUNT > 0 THEN
            UPDATE DF_GESTION_PRESUPUESTO
            SET GESTION = P_GESTION,
                ID_PROYECTO = P_ID_PROYECTO,
                ID_PROVEEDOR = P_ID_PROVEEDOR,
                ID_CONTRATOMARCO = P_ID_CONTRATOMARCO,
                ID_PEP = P_ID_PEP,
                ID_SOLPED = P_ID_SOLPED,
                ID_TEMA = P_ID_TEMA,
                ID_ORDEN_COMPRA = P_ID_ORDEN_COMPRA,
                ID_FACTURA_COMPRA = P_ID_FACTURA_COMPRA,
                RESPONSABLE = P_RESPONSABLE,
                ESTADO = P_ESTADO,
                TRAMITA_CON = P_TRAMITA_CON,
                CODEMPLEADO_MODIFICA = P_CODEMPLEADO_MODIFICA,
                FECHAMODIFICACION = CURRENT_TIMESTAMP
            WHERE ID = P_ID_GESTION;
            DBMS_OUTPUT.PUT_LINE('LLEGA AQUI INSERT DF_GESTION_PRESUPUESTO: ' || P_ID_GESTION);
            V_ID_CREADO := P_ID_GESTION;
            V_S_RESPUESTA := 'Registro actualizado exitosamente';
        ELSE
            DBMS_OUTPUT.PUT_LINE('LLEGA ANTES INSERT DF_GESTION_PRESUPUESTO: ' );
            INSERT INTO DF_GESTION_PRESUPUESTO 
                (GESTION, ID_PROYECTO, ID_PROVEEDOR, ID_CONTRATOMARCO, ID_PEP, ID_SOLPED, 
                 ID_TEMA, ID_ORDEN_COMPRA, ID_FACTURA_COMPRA, RESPONSABLE, ESTADO, 
                 TRAMITA_CON, CODEMPLEADO_CREO, FECHACREACION)
            VALUES 
                (P_GESTION, P_ID_PROYECTO, P_ID_PROVEEDOR, P_ID_CONTRATOMARCO, P_ID_PEP, P_ID_SOLPED, 
                 P_ID_TEMA, P_ID_ORDEN_COMPRA, P_ID_FACTURA_COMPRA, P_RESPONSABLE, P_ESTADO, 
                 P_TRAMITA_CON, P_CODEMPLEADO_CREO, CURRENT_TIMESTAMP)
            RETURNING ID INTO V_ID_CREADO;
            DBMS_OUTPUT.PUT_LINE('LLEGA AQUI INSERT DF_GESTION_PRESUPUESTO: ' || V_ID_CREADO);

            V_S_RESPUESTA := 'Registro insertado exitosamente';
        END IF;

        COMMIT;
        V_S_CODIGO := 200;
    EXCEPTION
        WHEN OTHERS THEN
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'Error: ' || SQLERRM;
            ROLLBACK;
    END;
END SP_SAVE_DF_GESTION_PRESUPUESTO;
