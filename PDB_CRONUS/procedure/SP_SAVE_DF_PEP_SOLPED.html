PROCEDURE SP_SAVE_DF_PEP_SOLPED (
    P_ID_PEP_SOLPED IN NUMBER,
    P_ID_SOLPED IN NUMBER,
    P_ID_PEP IN NUMBER,
    P_CODEMPLEADO_CREO IN NUMBER,
    P_CODEMPLEADO_MODIFICA IN NUMBER,
    V_S_RESPUESTA OUT VARCHAR2,
    V_ID_CREADO OUT NUMBER,
    V_S_CODIGO OUT NUMBER,
    V_S_MENSAJE OUT VARCHAR2
)
AS
    V_COUNT NUMBER;
    V_EXISTING_ID NUMBER;
BEGIN
    BEGIN
        -- Verificar si ya existe el registro
        SELECT COUNT(*)
        INTO V_COUNT
        FROM DF_PEP_SOLPED
        WHERE ID_SOLPED = P_ID_SOLPED
        AND ID_PEP = P_ID_PEP;

        IF V_COUNT > 0 THEN
            -- Obtener ID existente para retornar
            SELECT ID INTO V_EXISTING_ID
            FROM DF_PEP_SOLPED
            WHERE ID_SOLPED = P_ID_SOLPED
            AND ID_PEP = P_ID_PEP;

            -- Actualizar
            UPDATE DF_PEP_SOLPED
            SET CODEMPLEADO_MODIFICA = P_CODEMPLEADO_MODIFICA,
                FECHAMODIFICACION = SYSTIMESTAMP
            WHERE ID = V_EXISTING_ID;

            V_ID_CREADO := V_EXISTING_ID;
            V_S_RESPUESTA := 'Registro actualizado exitosamente';
        ELSE
            -- Insertar nuevo registro (ID es autogenerado por trigger)
            INSERT INTO DF_PEP_SOLPED (
                ID_SOLPED, ID_PEP, CODEMPLEADO_CREO, FECHACREACION
            ) VALUES (
                P_ID_SOLPED, P_ID_PEP, P_CODEMPLEADO_CREO, SYSTIMESTAMP
            ) RETURNING ID INTO V_ID_CREADO;

            V_S_RESPUESTA := 'Registro insertado exitosamente';
        END IF;

        COMMIT;
        V_S_CODIGO := 200;
        V_S_MENSAJE := 'OK';

    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            V_S_CODIGO := 404;
            V_S_RESPUESTA := 'No se encontró registro existente';
            V_S_MENSAJE := 'Error: NO_DATA_FOUND';
            ROLLBACK;
        WHEN OTHERS THEN
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'Error al procesar';
            V_S_MENSAJE := SQLERRM;
            ROLLBACK;
    END;
END SP_SAVE_DF_PEP_SOLPED;
