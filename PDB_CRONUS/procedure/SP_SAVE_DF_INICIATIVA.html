PROCEDURE "SP_SAVE_DF_INICIATIVA" (
    P_ID_INICIATIVA IN NUMBER,
    P_NOMBRE_INICIATIVA IN VARCHAR2,
    P_ID_PEP_INICIATIVA IN NUMBER,
	P_PEP IN VARCHAR2,
    P_CODPROYECTO_INICIATIVA IN NUMBER,
    P_CODGERENTE_INICIATIVA IN NUMBER,
    P_EVIDENCIA IN VARCHAR2,
    P_ID_LIDER_INICIATIVA IN VARCHAR2,
    P_LIDER_INICIATIVA IN VARCHAR2,
    P_ID_INI_FLUJO IN NUMBER,
    P_INI_FLUJO IN VARCHAR2,
	P_CODEMPLEADO_CREA IN NUMBER,
    P_CODTAREA IN NUMBER,
    V_S_RESPUESTA OUT VARCHAR2,
    V_ID_CREADO OUT NUMBER,
    V_S_CODIGO OUT NUMBER,
    V_S_MENSAJE OUT VARCHAR2
)
AS
    V_COUNT NUMBER;
    V_PEP VARCHAR2(200) := '';
    V_EXISTING_ID NUMBER;
    V_ID_PEP NUMBER;
    V_RESPUESTA VARCHAR2(4000);
    V_CODIGO NUMBER;
    V_ID_LIDER_INICIATIVA NUMBER;
    V_ID_INICIATIVA_FLUJO NUMBER;
    V_ID_INICIATIVA_ENTREGABLE NUMBER;
BEGIN
    BEGIN
        -- Verificar si el registro ya existe en la tabla
        BEGIN
            SELECT ID
            INTO V_ID_INICIATIVA_ENTREGABLE
            FROM DF_INICIATIVA WHERE NOMBRE = P_NOMBRE_INICIATIVA;
        EXCEPTION
            WHEN NO_DATA_FOUND THEN
                V_ID_INICIATIVA_ENTREGABLE := 0;
                DBMS_OUTPUT.PUT_LINE('NO ENCUENTRA EL V_ID_INICIATIVA ' || V_ID_INICIATIVA_ENTREGABLE);
        END;
        
        SP_SAVE_LIDER_INICIATIVA (P_ID_LIDER_INICIATIVA, P_LIDER_INICIATIVA, V_ID_LIDER_INICIATIVA, V_RESPUESTA, V_CODIGO);
        
        DBMS_OUTPUT.PUT_LINE('NO ENCUENTRA EL V_ID_LIDER_INICIATIVA ' || V_ID_LIDER_INICIATIVA);
        
        SP_SAVE_INICIATIVA_FLUJO(P_ID_INI_FLUJO, P_INI_FLUJO, 0, 0, V_ID_LIDER_INICIATIVA, V_ID_INICIATIVA_FLUJO, V_RESPUESTA, V_CODIGO);
        
        DBMS_OUTPUT.PUT_LINE('NO ENCUENTRA EL V_ID_INICIATIVA_FLUJO ' || V_ID_INICIATIVA_FLUJO);
		
		SP_SAVE_PEPS(P_ID_PEP_INICIATIVA, P_PEP, P_CODEMPLEADO_CREA, 0, P_CODGERENTE_INICIATIVA, 1, '', 0, '', V_ID_PEP, V_RESPUESTA, V_CODIGO);
        
        DBMS_OUTPUT.PUT_LINE('NO ENCUENTRA EL V_ID_PEP ' || V_ID_PEP);

        IF V_CODIGO = 200 THEN 
        
            -- Si el registro existe, actualizarlo; de lo contrario, insertar uno nuevo
            IF V_ID_INICIATIVA_ENTREGABLE > 0 THEN
                UPDATE DF_INICIATIVA
                SET NOMBRE = P_NOMBRE_INICIATIVA,
                    PEP = P_PEP,
                    CODPROYECTO = P_CODPROYECTO_INICIATIVA,
                    CODGERENTE = P_CODGERENTE_INICIATIVA,
                    FECHAMODIFICACION = CURRENT_TIMESTAMP,
                    EVIDENCIA = P_EVIDENCIA,
                    ID_LIDER_INICIATIVA = V_ID_LIDER_INICIATIVA,
                    ID_PEP = V_ID_PEP,
                    ID_INI_FLUJO = V_ID_INICIATIVA_FLUJO,
                    CODTAREA = P_CODTAREA
                WHERE ID = V_ID_INICIATIVA_ENTREGABLE;
    
                V_ID_CREADO := V_ID_INICIATIVA_ENTREGABLE;
                V_S_RESPUESTA := 'Registro actualizado exitosamente';
            ELSE
                INSERT INTO DF_INICIATIVA (NOMBRE,  CODPROYECTO, CODGERENTE, FECHACREACION, EVIDENCIA, 
                                ID_LIDER_INICIATIVA, ID_PEP, ID_INI_FLUJO, PEP, CODTAREA)
                VALUES (P_NOMBRE_INICIATIVA,  P_CODPROYECTO_INICIATIVA, P_CODGERENTE_INICIATIVA, CURRENT_TIMESTAMP,  
                                P_EVIDENCIA, V_ID_LIDER_INICIATIVA, V_ID_PEP, V_ID_INICIATIVA_FLUJO, P_PEP, P_CODTAREA)
                RETURNING ID INTO V_ID_CREADO;
    
                V_S_RESPUESTA := 'Registro insertado exitosamente';
            END IF;
            COMMIT;
            V_S_CODIGO := 200;
            V_S_MENSAJE := 'Operación completada con éxito';
        ELSE 
            V_S_CODIGO := 400;
            V_S_MENSAJE := V_RESPUESTA;
        END IF;
        

    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'No existe estructura';
            V_S_MENSAJE := 'No se encontró el registro especificado';
            ROLLBACK;
        WHEN OTHERS THEN
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'Error: ' || SQLERRM;
            V_S_MENSAJE := 'Ocurrió un error inesperado';
            ROLLBACK;
    END;
END SP_SAVE_DF_INICIATIVA;
