PROCEDURE "SP_GET_PROYECTOS_VIGENTES" (
    P_COD_COR IN NUMBER, 
    P_FECHA_INI IN DATE, 
    P_FECHA_FIN IN DATE, 
    P_CURSOR_GET_PROY_VIGE OUT SYS_REFCURSOR,
    V_S_RESPUESTA OUT VARCHAR2,
    V_S_CODIGO OUT NUMBER
)
AS
BEGIN
    BEGIN
        OPEN P_CURSOR_GET_PROY_VIGE FOR
            SELECT py.ID, py.CODIGOPROYECTO, py.NOMBRE 
            FROM DF_PROYECTO py
            JOIN DF_TAREA t ON t.codproyecto = py.id
            JOIN df_empleadocontrol ec ON ec.codtarea = t.id
            JOIN df_empleado_vu e ON e.codempleado = ec.codempleado AND e.codcor = P_COD_COR 
            WHERE 
                (ec.fechahorainicio >= P_FECHA_INI AND ec.fechahorainicio <= P_FECHA_FIN)
                OR (ec.fechahorafin >= P_FECHA_INI AND ec.fechahorafin <= P_FECHA_FIN)
                AND e.codtipoestructura <> 4
            GROUP BY py.ID, py.CODIGOPROYECTO, py.NOMBRE
            ORDER BY py.ID;

        V_S_CODIGO := 200;
        V_S_RESPUESTA := 'Retorno Correcto';
    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            ROLLBACK;
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'No Fue posible traer la data';
            RAISE;
    END;
END SP_GET_PROYECTOS_VIGENTES;
