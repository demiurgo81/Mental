PROCEDURE "SP_GET_HORASMES_REPOR_EMPLEADO" (
    P_CODEMPLEADOASIGNADO IN NUMBER,
    P_HORAS_OBLIGATORIAS IN NUMBER,
    P_FECHA_INI IN DATE,
    P_FECHA_FIN IN DATE,
    P_CURSOR_TOT_HORAS OUT SYS_REFCURSOR,
    V_S_RESPUESTA OUT VARCHAR2,
    V_S_CODIGO OUT NUMBER
)
AS
    V_TAREA_NOMBRE VARCHAR2(50) := 'GESTION FUNCIONARIO';
    V_EMPLEADOCONTROL_DESC VARCHAR2(100) := 'REGISTRO OCUPACION AUTOMATICA';
    V_SUM_HORAS_TAREAS NUMBER := 0;
    V_SUM_HORAS_EMPCON NUMBER := 0;
    V_TIEMPOREAL_SUM NUMBER := 0;
    V_HORASEMPCON_SUM NUMBER := 0;
BEGIN
    BEGIN
        -- Sumar el tiempo real de las tareas
        SELECT SUM(TIEMPOREAL) INTO V_TIEMPOREAL_SUM 
        FROM DF_TAREA 
        WHERE CODEMPLEADOASIGNADO = P_CODEMPLEADOASIGNADO 
        AND FECHAINIREAL BETWEEN P_FECHA_INI AND P_FECHA_FIN
        AND FECHAFINREAL BETWEEN P_FECHA_INI AND P_FECHA_FIN;

        -- Sumar las horas de la tabla DF_EMPLEADOCONTROL
        SELECT SUM(HORAS) INTO V_HORASEMPCON_SUM 
        FROM DF_EMPLEADOCONTROL
        WHERE CODEMPLEADO = P_CODEMPLEADOASIGNADO 
        AND FECHAHORAINICIO BETWEEN P_FECHA_INI AND P_FECHA_FIN 
        AND FECHAHORAFIN BETWEEN P_FECHA_INI AND P_FECHA_FIN;

        -- Verificar si las horas reportadas cumplen con las horas obligatorias
        IF V_TIEMPOREAL_SUM >= P_HORAS_OBLIGATORIAS AND V_HORASEMPCON_SUM >= P_HORAS_OBLIGATORIAS THEN
            V_S_RESPUESTA := 'El empleado ya tiene las horas del mes completas';
        ELSE
            V_S_RESPUESTA := 'El empleado tiene las siguientes horas reportadas';
        END IF;

        OPEN P_CURSOR_TOT_HORAS FOR
            SELECT V_TIEMPOREAL_SUM as V_TIEMPOREAL_SUM, V_HORASEMPCON_SUM as V_HORASEMPCON_SUM, V_S_RESPUESTA as V_S_RESPUESTA FROM DUAL;

        V_S_CODIGO := 200;
    EXCEPTION
        WHEN OTHERS THEN
            V_S_CODIGO := 500;
            V_S_RESPUESTA := 'Error: ' || SQLERRM;
    END;
END SP_GET_HORASMES_REPOR_EMPLEADO;
