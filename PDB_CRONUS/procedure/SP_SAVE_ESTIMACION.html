PROCEDURE SP_SAVE_ESTIMACION (
    P_ID                    IN DF_ESTIMACION.ID%TYPE,
    P_ENTREGABLE           IN DF_ESTIMACION.ENTREGABLE%TYPE,
    P_PERFIL               IN DF_ESTIMACION.PERFIL%TYPE,
    P_PREREQUISITO         IN DF_ESTIMACION.PREREQUISITO%TYPE,
    P_CANTIDAD             IN DF_ESTIMACION.CANTIDAD%TYPE,
    P_FECHAINICIO          IN DF_ESTIMACION.FECHAINICIO%TYPE,
    P_FECHAFIN             IN DF_ESTIMACION.FECHAFIN%TYPE,
    P_COSTO                IN DF_ESTIMACION.COSTO%TYPE,
    P_PORCENTAJE           IN DF_ESTIMACION.PORCENTAJE%TYPE,
    P_CODPROYECTO          IN DF_ESTIMACION.CODPROYECTO%TYPE,
    P_URLSHAREPOINT        IN DF_ESTIMACION.URLSHAREPOINT%TYPE,
    P_ID_PROYECTO_ESTIMACION IN DF_ESTIMACION.ID_PROYECTO_ESTIMACION%TYPE,
    V_ID_CREADO            OUT DF_ESTIMACION.ID%TYPE,
    V_S_RESPUESTA          OUT VARCHAR2,
    V_S_CODIGO             OUT NUMBER
) AS
    V_EXISTE NUMBER;
BEGIN
    -- Verificar si existe el registro
    SELECT COUNT(*)
      INTO V_EXISTE
      FROM DF_ESTIMACION
     WHERE ID = P_ID;

    IF V_EXISTE > 0 THEN
        -- Actualizar
        UPDATE DF_ESTIMACION
           SET ENTREGABLE             = P_ENTREGABLE,
               PERFIL                 = P_PERFIL,
               PREREQUISITO           = P_PREREQUISITO,
               CANTIDAD               = P_CANTIDAD,
               FECHAINICIO            = P_FECHAINICIO,
               FECHAFIN               = P_FECHAFIN,
               COSTO                  = P_COSTO,
               PORCENTAJE             = P_PORCENTAJE,
               CODPROYECTO            = P_CODPROYECTO,
               URLSHAREPOINT          = P_URLSHAREPOINT,
               ID_PROYECTO_ESTIMACION = P_ID_PROYECTO_ESTIMACION
         WHERE ID = P_ID;

        V_ID_CREADO   := P_ID;
        V_S_RESPUESTA := 'Registro actualizado exitosamente';
    ELSE
        -- Insertar
        INSERT INTO DF_ESTIMACION (
            ENTREGABLE,
            PERFIL,
            PREREQUISITO,
            CANTIDAD,
            FECHAINICIO,
            FECHAFIN,
            COSTO,
            PORCENTAJE,
            CODPROYECTO,
            URLSHAREPOINT,
            ID_PROYECTO_ESTIMACION
        ) VALUES (
            P_ENTREGABLE,
            P_PERFIL,
            P_PREREQUISITO,
            P_CANTIDAD,
            P_FECHAINICIO,
            P_FECHAFIN,
            P_COSTO,
            P_PORCENTAJE,
            P_CODPROYECTO,
            P_URLSHAREPOINT,
            P_ID_PROYECTO_ESTIMACION
        )
        RETURNING ID INTO V_ID_CREADO;

        V_S_RESPUESTA := 'Registro insertado exitosamente';
    END IF;

    COMMIT;
    V_S_CODIGO := 200;

EXCEPTION
    WHEN OTHERS THEN
        V_S_CODIGO    := 500;
        V_S_RESPUESTA := 'Error interno: ' || SQLERRM;
        ROLLBACK;
END SP_SAVE_ESTIMACION;
