TRIGGER_NAME TRIGGER_TYPE TRIGGER_OWNER TRIGGERING_EVENT STATUS TABLE_NAME
DF_NOVEDAD_TRG AFTER EACH ROW PDB_CRONUS INSERT OR UPDATE ENABLED DF_NOVEDAD 



TRIGGER "PDB_CRONUS"."DF_NOVEDAD_TRG" 
  AFTER INSERT OR UPDATE ON df_novedad
    REFERENCING OLD AS OLD NEW AS NEW
    FOR EACH ROW
BEGIN
  IF :NEW.Tipo_Novedad = 23 OR :OLD.Tipo_Novedad = 23 THEN 
    IF INSERTING AND :NEW.Tipo_Novedad = 23 AND :NEW.Aprobado = 1 AND :NEW.Estado = 1 THEN
      dbms_output.put_line ('*** INICIO Calculando Horas extras - Insert');
      DF_Horas_Extras_PKG.PR_Procesar_Horas_Extras (:NEW.ID, :NEW.Ini_Novedad, :NEW.Fin_Novedad, :NEW.Aprobado);
      dbms_output.put_line ('***    Fin proceso de Calculo - Insert');
    END IF;
    IF UPDATING THEN
      IF :NEW.Estado = 9 THEN
        dbms_output.put_line ('*** INICIO Calculando Horas extras - Borrar');
        DF_Horas_Extras_PKG.PR_Borrar_Horas_Extras (:OLD.ID);
        dbms_output.put_line ('***    Fin proceso de Calculo - Borrar');
      END IF;

      CASE
        WHEN    ((:OLD.Tipo_Novedad  = 23 AND :OLD.Aprobado = 1) AND (:NEW.Tipo_Novedad  = 23 AND :NEW.Aprobado = 1 ))
             OR ((:OLD.Tipo_Novedad  = 23 AND :OLD.Aprobado = 0) AND (:NEW.Tipo_Novedad  = 23 AND :NEW.Aprobado = 1 ))
             OR ((:OLD.Tipo_Novedad <> 23 AND :OLD.Aprobado = 0) AND (:NEW.Tipo_Novedad  = 23 AND :NEW.Aprobado = 1 ))
             OR ((:OLD.Tipo_Novedad <> 23 AND :OLD.Aprobado = 1) AND (:NEW.Tipo_Novedad  = 23 AND :NEW.Aprobado = 1 )) THEN
          dbms_output.put_line ('*** INICIO Calculando Horas extras - Actualizar');
          DF_Horas_Extras_PKG.PR_Procesar_Horas_Extras (:OLD.ID, :NEW.Ini_Novedad, :NEW.Fin_Novedad, :NEW.Aprobado);
          dbms_output.put_line ('***    Fin proceso de Calculo - Actualizar');
        WHEN    ((:OLD.Tipo_Novedad  = 23 AND :OLD.Aprobado = 1) AND (:NEW.Tipo_Novedad  = 23 AND :NEW.Aprobado = 0 ))
             OR ((:OLD.Tipo_Novedad  = 23 AND :OLD.Aprobado = 1) AND (:NEW.Tipo_Novedad <> 23                       )) THEN
          DF_Horas_Extras_PKG.PR_Borrar_Horas_Extras (:OLD.ID);
      END CASE;
    END IF;
  END IF;
END;
