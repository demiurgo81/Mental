TRIGGER DF_TAREA_AUDIT_TRG
AFTER INSERT OR UPDATE OR DELETE ON DF_TAREA
FOR EACH ROW
DECLARE
    v_operacion VARCHAR2(10);
BEGIN
    IF INSERTING THEN
        v_operacion := 'INSERT';
        INSERT INTO DF_TAREA_HIST (
            ID_HISTORICO, ID, CODTAREAPADRE, CODTAREATIPO, CODTAREAESTADO,
            CODEMPLEADOCREO, CODEMPLEADOASIGNADO, CODPROYECTO,
            JERARQUIA, NOMBRE, NIVEL, DESCRIPCION,
            FECHAINIESTIMADA, FECHAFINESTIMADA, FECHAINIREAL, FECHAFINREAL,
            TIEMPOESTIMADO, TIEMPOREAL, FECHACREACION, FECHAMODIFICACION,
            ESLOGRO, ESTADO, FECHA_COMPROMISO, FECHA_SOLICITUD,
            DESCRIPCION_SOLICITUD, CODLINEAPRODUCTO,
            OPERACION, USUARIO_AUDIT, FECHA_AUDIT
        ) VALUES (
            SEQ_DF_TAREA_HIST.NEXTVAL, :NEW.ID, :NEW.CODTAREAPADRE, :NEW.CODTAREATIPO, :NEW.CODTAREAESTADO,
            :NEW.CODEMPLEADOCREO, :NEW.CODEMPLEADOASIGNADO, :NEW.CODPROYECTO,
            :NEW.JERARQUIA, :NEW.NOMBRE, :NEW.NIVEL, :NEW.DESCRIPCION,
            :NEW.FECHAINIESTIMADA, :NEW.FECHAFINESTIMADA, :NEW.FECHAINIREAL, :NEW.FECHAFINREAL,
            :NEW.TIEMPOESTIMADO, :NEW.TIEMPOREAL, :NEW.FECHACREACION, :NEW.FECHAMODIFICACION,
            :NEW.ESLOGRO, :NEW.ESTADO, :NEW.FECHA_COMPROMISO, :NEW.FECHA_SOLICITUD,
            :NEW.DESCRIPCION_SOLICITUD, :NEW.CODLINEAPRODUCTO,
            v_operacion, SYS_CONTEXT('USERENV', 'SESSION_USER'), CURRENT_TIMESTAMP
        );

    ELSIF UPDATING THEN
        v_operacion := 'UPDATE';
        INSERT INTO DF_TAREA_HIST VALUES (
            SEQ_DF_TAREA_HIST.NEXTVAL, :NEW.ID, :NEW.CODTAREAPADRE, :NEW.CODTAREATIPO, :NEW.CODTAREAESTADO,
            :NEW.CODEMPLEADOCREO, :NEW.CODEMPLEADOASIGNADO, :NEW.CODPROYECTO,
            :NEW.JERARQUIA, :NEW.NOMBRE, :NEW.NIVEL, :NEW.DESCRIPCION,
            :NEW.FECHAINIESTIMADA, :NEW.FECHAFINESTIMADA, :NEW.FECHAINIREAL, :NEW.FECHAFINREAL,
            :NEW.TIEMPOESTIMADO, :NEW.TIEMPOREAL, :NEW.FECHACREACION, :NEW.FECHAMODIFICACION,
            :NEW.ESLOGRO, :NEW.ESTADO, :NEW.FECHA_COMPROMISO, :NEW.FECHA_SOLICITUD,
            :NEW.DESCRIPCION_SOLICITUD, :NEW.CODLINEAPRODUCTO,
            v_operacion, SYS_CONTEXT('USERENV', 'SESSION_USER'), CURRENT_TIMESTAMP
        );

    ELSIF DELETING THEN
        v_operacion := 'DELETE';
        INSERT INTO DF_TAREA_HIST VALUES (
            SEQ_DF_TAREA_HIST.NEXTVAL, :OLD.ID, :OLD.CODTAREAPADRE, :OLD.CODTAREATIPO, :OLD.CODTAREAESTADO,
            :OLD.CODEMPLEADOCREO, :OLD.CODEMPLEADOASIGNADO, :OLD.CODPROYECTO,
            :OLD.JERARQUIA, :OLD.NOMBRE, :OLD.NIVEL, :OLD.DESCRIPCION,
            :OLD.FECHAINIESTIMADA, :OLD.FECHAFINESTIMADA, :OLD.FECHAINIREAL, :OLD.FECHAFINREAL,
            :OLD.TIEMPOESTIMADO, :OLD.TIEMPOREAL, :OLD.FECHACREACION, :OLD.FECHAMODIFICACION,
            :OLD.ESLOGRO, :OLD.ESTADO, :OLD.FECHA_COMPROMISO, :OLD.FECHA_SOLICITUD,
            :OLD.DESCRIPCION_SOLICITUD, :OLD.CODLINEAPRODUCTO,
            v_operacion, SYS_CONTEXT('USERENV', 'SESSION_USER'), CURRENT_TIMESTAMP
        );
    END IF;
END;
