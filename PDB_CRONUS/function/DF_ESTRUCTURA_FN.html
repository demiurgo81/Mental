FUNCTION "DF_ESTRUCTURA_FN" 
 (JERARQ  VARCHAR2, tipoestructura number, tiporetorno number DEFAULT null)
 RETURN VARCHAR IS
     EMPLEADO VARCHAR2(10);
     ESTRUCTURA VARCHAR2(500);
 BEGIN
    SELECT DISTINCT CODEMPLEADO_RESPONSABLE INTO EMPLEADO 
    FROM DF_ESTRUCTURAORGANIZACIONAL EO 
    WHERE EO.CODESTRUCTURATIPO = tipoestructura AND ID IN(SUBSTR(JERARQ,1,10)*1,SUBSTR(JERARQ,11,10)*1,SUBSTR(JERARQ,21,10)*1,SUBSTR(JERARQ,31,10)*1);
    SELECT DISTINCT NOMBRE INTO ESTRUCTURA 
    FROM DF_ESTRUCTURAORGANIZACIONAL EO 
    WHERE EO.CODESTRUCTURATIPO = tipoestructura AND ID IN(SUBSTR(JERARQ,1,10)*1,SUBSTR(JERARQ,11,10)*1,SUBSTR(JERARQ,21,10)*1,SUBSTR(JERARQ,31,10)*1);

    IF tiporetorno IS NOT NULL THEN
        RETURN ESTRUCTURA; 
    END IF;

 RETURN EMPLEADO;
 END DF_ESTRUCTURA_FN;
