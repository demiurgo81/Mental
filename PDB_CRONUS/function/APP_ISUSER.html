FUNCTION "APP_ISUSER" ( V_USER IN VARCHAR2, V_PASSWORD IN VARCHAR2 )
RETURN VARCHAR2
IS
--DECLARE
   ISUSER number;
   TOKEN VARCHAR2(500);
   MASKC VARCHAR2(500);
   SQL_STR VARCHAR2(1000);
   --V_USER VARCHAR2(30);
   --V_PASSWORD VARCHAR2(30);
BEGIN
    --V_USER := 'ADMIN';
    --V_PASSWORD := 'ADMIN1';

    ISUSER := 0;
    TOKEN := '';
    MASKC := 'ASDFGHJKL156789';

    SELECT ID_USER INTO ISUSER FROM APP_USERS  
    WHERE "USER" = V_USER
    AND "PASS"=V_PASSWORD
    AND "STATE" = 'A';
    DBMS_OUTPUT.PUT_LINE(ISUSER);
    IF ISUSER>0 THEN
        SELECT  UTL_RAW.cast_to_varchar2(UTL_ENCODE.base64_encode(UTL_RAW.bit_xor(UTL_RAW.cast_to_raw(V_USER), UTL_RAW.cast_to_raw(MASKC)))) INTO TOKEN FROM dual;
        --APP_NEWSESSION(ISUSER, TOKEN);
        DBMS_OUTPUT.PUT_LINE(TOKEN);
        RETURN TOKEN;
    ELSE
        DBMS_OUTPUT.PUT_LINE('VACIO');
        RETURN '';
    END IF;
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('NO DATA');
        RETURN '';
    WHEN OTHERS THEN
    raise_application_error(-20001,'An error was encountered - '||SQLCODE||' -ERROR- '||SQLERRM);
END;
