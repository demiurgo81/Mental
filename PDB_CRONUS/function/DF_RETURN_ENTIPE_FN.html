FUNCTION df_return_entipe_FN (
    ORIGINAL IN VARCHAR2
) RETURN VARCHAR2 DETERMINISTIC
AS
BEGIN
    -- Recorrer los registros de df_clasifica que cumplen la condición,
    -- ordenados por prioridad.
    FOR rec IN (
        SELECT BUSCA, REEMPLAZA
        FROM df_clasifica
        WHERE OPERACION = 'TIPOENT'
        ORDER BY PRIORIDAD ASC
    ) LOOP
        -- Verificar si el valor de BUSCA no es NULL
        -- y si está contenido en el parámetro ORIGINAL.
        -- INSTR realiza una búsqueda sensible a mayúsculas y minúsculas.
        IF rec.BUSCA IS NOT NULL AND INSTR(ORIGINAL, rec.BUSCA) > 0 THEN
            -- Si se encuentra una coincidencia, retornar REEMPLAZA inmediatamente.
            RETURN rec.REEMPLAZA;
        END IF;
    END LOOP;

    -- Si el bucle termina sin encontrar coincidencias, retornar el valor por defecto.
    RETURN 'DESARROLLO';

EXCEPTION
    WHEN OTHERS THEN
        -- En caso de cualquier error inesperado durante la ejecución,
        -- se retorna NULL. Considera un manejo de errores más específico
        -- o logging si es necesario para tu entorno.
        -- DBMS_OUTPUT.PUT_LINE('Error en df_return_entipe_FN: ' || SQLERRM);
        RETURN NULL;
END df_return_entipe_FN;
